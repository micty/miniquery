 21     /**@inner*/
 22     function Mapper(key, value) {
 23 
 24         //分配本实例对应的容器
 25         var guid = $.String.random();
 26         this[guidKey] = guid;
 27 
 28         guid$type$object[guid] = {
 29             'guid': {},     //针对有 guid 属性的对象作特殊存取，以求一步定位。
 30             'false': {},    //针对 false|null|undefined|''|0|NaN
 31             'string': {},
 32             'number': {},
 33             'boolean': {}, //只针对 true
 34             'object': {},
 35             'function': {},
 36             'undefined': {} //这个用不到
 37         };
 38 
 39         if (arguments.length > 0) { //这里不要判断 key，因为 key 可以为任何值。
 40             this.set(key, value);
 41         }

 46     function getString(obj) {
 47 
 48         //函数的 length 属性表示形参的个数，而且是只读的。 
 49         //详见 https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/Function/length
 50         //函数的 name 属性表示函数的名称，而且是只读的。 匿名函数的名称为空字符串。 
 51         //详见 https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/Function/name
 52         if (typeof obj == 'function') {
 53 
 54             var a = [];
 55 
 56             if ('name' in obj) {
 57                 a.push(obj.name);
 58             }
 59 
 60             if ('length' in obj) {
 61                 a.push(obj.length);
 62             }
 63 
 64             if (a.length > 0) {
 65                 return a.join('#');
 66             }
 67         }
 68 
 69 
 70         return $.Object.getType(obj); //返回相应构造器的名称，不要用　toString，因为可能会变
 71 
 72     }

 88         /**
 89         * 设置一对映射关系。
 90         * @param src 映射关系的键，可以是任何类型。
 91         * @param target 映射关系要关联的值，可是任何类型。
 92         * @return 返回第二个参数 target。
 93         * @example
 94             var obj = { a: 1, b: 2 };
 95             var fn = function(a, b) {
 96                 console.log(a+b);
 97             };
 98             var mapper = new $.Mapper();
 99             mapper.set(obj, fn);
100             mapper.set('a', 100);
101             mapper.set(100, 'abc');
102             mapper.set('100', 'ABC');
103             mapper.set(null, 'abc');
104             
105         */
106         set: function (src, target) {
107 
108             var all = guid$type$object[this[guidKey]];
109 
110             if (!src) { // false|null|undefined|''|0|NaN
111                 all['false'][String(src)] = target;
112                 return target;
113             }
114 
115             var type = typeof src;
116 
117             if (type == 'object' && guidKey in src) { //针对含有 guid 属性的对象作优先处理
118                 var guid = src[guidKey];
119                 all['guid'][guid] = target; //一步定位
120                 return;
121             }
122 
123 
124             switch (type) {
125                 case 'string':
126                 case 'number':
127                 case 'boolean':
128                     all[type][String(src)] = target;
129                     break;
130 
131                 case 'object':
132                 case 'function':
133                     var key = getString(src); //这里确保 key 一定是一个 string
134                     var list = all[type][key];
135                     if (list) { //已存在对应字符串的列表
136                         var pair = $.Array.findItem(list, function (pair, index) {
137                             return pair[0] === src;
138                         });
139 
140                         if (pair) { //已存在，
141                             pair[1] = target; //改写值
142                         }
143                         else { //未找到，创建新的，添加进去一对二元组 [src, target]
144                             list.push([src, target]);
145                         }
146                     }
147                     else { //未存在，则创建并添加
148                         list = [[src, target]];
149                     }
150 
151                     all[type][key] = list; //回写
152             }
153 
154             return target;
155         },

158         * 批量设置映射关系。

176         /**
177         * 获取一对映射关系所关联的值。<br />
178         * 注意：根据映射关系的键查找所关联的值时，对键使用的是全等比较，即区分数据类型。
179         * @param src 映射关系的键，可以是任何类型。
180         * @return 返回映射关系所关联的值。
181         * @example
182             var obj = { a: 1, b: 2 };
183             var fn = function(a, b) {
184                 console.log(a+b);
185             };
186             
187             var mapper = new $.Mapper();
188             mapper.set(obj, fn);
189             
190             var myFn = mapper.get(obj); //获取到之前关联的 fn
191             myFn(100, 200);
192         */
193         get: function (src) {
194 
195             var all = guid$type$object[this[guidKey]];
196 
197             // false|null|undefined|''|0|NaN
198             if (!src) {
199                 return all['false'][String(src)];
200             }
201 
202 
203             var type = typeof src;
204 
205             if (type == 'object' && guidKey in src) { //针对含有 guid 属性的对象作优先处理
206                 var guid = src[guidKey];
207                 return all['guid'][guid]; //一步定位
208             }
209 
210 
211 
212             switch (type) {
213                 //值类型的，直接映射
214                 case 'string':
215                 case 'number':
216                 case 'boolean':
217                     return all[type][String(src)];
218 
219 
220                 //引用类型的，通过 key 映射到一个二维数组，每个二维数组项为 [src, target]
221                 case 'object':
222                 case 'function':
223                     var key = getString(src);
224                     var list = all[type][key];
225                     if (list) { //已存在对应字符串的列表
226                         var pair = $.Array.findItem(list, function (pair, index) {
227                             return pair[0] === src;
228                         });
229 
230                         if (pair) {
231                             return pair[1];
232                         }
233                     }
234             }
235 
236             return undefined;
237         },

241         * 根据给定的键移除一对映射关系。
242         * 注意：根据映射关系的键查找所关联的值时，对键使用的是全等比较，即区分数据类型。
243         * @param src 映射关系的键，可以是任何类型。
244         * @example
245             var obj = { a: 1, b: 2 };
246             var fn = function(a, b) {
247                 console.log(a+b);
248             };

255             console.log( typeof fn); // undefined

257         remove: function (src) {
258 
259             var all = guid$type$object[this[guidKey]];
260 
261             // false|null|undefined|''|0|NaN
262             if (!src) {
263                 delete all['false'][String(src)];
264                 return;
265             }
266 
267             var type = typeof src;
268 
269             if (type == 'object' && guidKey in src) { //针对含有 guid 属性的对象作优先处理
270                 var guid = src[guidKey];
271                 delete all['guid'][guid]; //一步定位
272                 return;
273             }
274 
275 
276             switch (type) {
277                 //值类型的，直接映射
278                 case 'string':
279                 case 'number':
280                 case 'boolean':
281                     delete all[type][String(src)];
282                     break;
283 
284                 //引用类型的，通过 key 映射到一个二维数组，每个二维数组项为 [src, target]
285                 case 'object':
286                 case 'function':
287                     var key = getString(src);
288                     var list = all[type][key];
289                     if (list) { //已存在对应字符串的列表
290                         //移除 src 的项
291                         all[type][key] = $.Array.grep(list, function (pair, index) {
292                             return pair[0] !== src;
293                         });
294                     }
295             }

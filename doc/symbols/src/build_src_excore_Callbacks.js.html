  4 /**
  5 * 回调列表类。
  6 * @class
  7 */
  8 MiniQuery.Callbacks = (function ($, Mapper) {
  9 
 10 
 11 var guidKey = Mapper.getGuidKey();
 12 var eventName = '__callback__' + $.String.random();
 13 var slice = Array.prototype.slice;
 14 
 15 /**
 16 * @inner
 17 * 构造函数
 18 */
 19 function Callbacks() {
 20         
 21     this[guidKey] = $.String.random();
 22 }
 23 
 24 
 25 $.extend(Callbacks.prototype, { /**@lends MiniQuery.Callbacks#*/
 26 
 27     /**
 28     * 添加一个回调函数到回调列表中。
 29     * @param {function} fn 要添加的回调函数
 30     * @param {boolean} [isOnce=false] 指示回调函数是否只给调用一次。
 31         如果要让回调函数给调用一次后被移除，请指定为 true；否则不指定或指定为 false。
 32     */
 33     add: function (fn, isOnce) {
 34         var args = [this, eventName].concat(slice.call(arguments, 0));
 35         $.Event.bind.apply(null, args);
 36     },
 37 
 38     /**
 39     * 添加一个只调用一次的回调函数到回调列表中。
 40     * @param {function} fn 要添加的回调函数。
 41         该回调函数给调用一次后会被移除。
 42     */
 43     once: function (fn) {
 44         var args = [this, eventName].concat(slice.call(arguments, 0));
 45         $.Event.once.apply(null, args);
 46     },
 47 
 48     /**
 49     * 从回调列表中移除指定的回调函数。
 50     * @param {function} fn 要移除的回调函数。
 51     */
 52     remove: function (fn) {
 53         var args = [this, eventName].concat(slice.call(arguments, 0));
 54         $.Event.unbind.apply(null, args);
 55     },
 56 
 57     /**
 58     * 调用回调列表中的回调函数，并可选地传递一些参数。
 59     * @return {Array} 返回由被调用的回调函数的返回值所组成的数组。
 60     */
 61     fire: function () {
 62         var args = slice.call(arguments, 0);
 63         return $.Event.trigger(this, eventName, args);
 64     },
 65 
 66     /**
 67     * 切换到指定的上下文来调用回调列表中的回调函数，并可选传递一些参数。
 68     * @param {Object} context 要切换到上下文对象。
 69         指定该参数后，回调函数中的 this 均指向该上下文对象。
 70     * @return {Array} 返回由被调用的回调函数的返回值所组成的数组。
 71     * @example
 72         var callbacks = new $.Callbacks();
 73         callbacks.add(function (a, b) {
 74             console.log(this.value, a, b);
 75         });
 76         callbacks.fireWith({ value: 100 }, 'aa', 'bb'); //输出 100, 'aa', 'bb'
 77     */
 78     fireWith: function (context) {
 79 
 80         var hasGuid = guidKey in context;
 81 
 82         var old = context[guidKey];
 83         context[guidKey] = this[guidKey];
 84 
 85         var args = slice.call(arguments, 1); //从索引为 1 开始后的所有参数都当作要传递给回调函数的参数
 86         var values = $.Event.trigger(context, eventName, args);
 87 
 88         if (hasGuid) {
 89             context[guidKey] = old;
 90         }
 91         else {
 92             delete context[guidKey];
 93         }
 94 
 95         return values;
 96 
 97         
 98     },
 99 
100     /**
101     * 调用回调列表中的回调函数直到某个回调函数返回指定的值为止，并可选地传递一些参数。
102     * @param {Object} stopValue 要使回调列表停止调用的值。
103         回调列表依次调用列表中的回调函数，当回调函数返回值为 stopValue 指定的值时，将会停止调用后面的。
104     * @return {Array} 返回由被调用的回调函数的返回值所组成的数组。
105     */
106     fireStop: function (stopValue) {
107         var args = slice.call(arguments, 1); //从索引为 1 开始后的所有参数都当作要传递给回调函数的参数
108         return $.Event.triggerStop(this, eventName, args, stopValue);
109     },
110 
111     /**
112     * 判断回调列表中是否包含指定的回调函数。
113     * @param {function} fn 要判断的回调函数。
114     * @return {boolean} 如果回调列表中包含指定的回调函数，则返回 true；否则返回 false。
115     */
116     has: function (fn) {
117         var args = [this, eventName].concat(slice.call(arguments, 0));
118         return $.Event.has.apply(null, args);
119     },
120 
121 
122     /**
123     * 给回调列表添加/移除指定的回调函数。
124     * 如果回调列表中存在该回调函数，则移除它；否则会添加它。
125     * @param {function} fn 回调函数。 
126     * @param {boolean} [isOnce=false] 指示回调函数是否只给调用一次。
127         如果要让回调函数给调用一次后被移除，请指定为 true；否则不指定或指定为 false。
128     */
129     toggle: function (fn, isOnce) {
130         var args = [this, eventName].concat(slice.call(arguments, 0));
131         $.Event.toggle.apply(null, args);
132     },
133 
134     /**
135     * 给回调列表只添加一次指定的回调函数。
136     * 只有当回调列表中不存在该回调函数时才添加它；否则会忽略。
137     * @param {function} fn 要添加的回调函数。 
138     * @param {boolean} [isOnce=false] 指示回调函数是否只给调用一次。
139         如果要让回调函数给调用一次后被移除，请指定为 true；否则不指定或指定为 false。
140     */
141     unique: function (fn, isOnce) {
142         var args = [this, eventName].concat(slice.call(arguments, 0));
143         $.Event.unique.apply(null, args);
144     }
145 
146 });
147 
148 
149 return Callbacks;
150

155 })(MiniQuery, MiniQuery.Mapper);

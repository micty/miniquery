  4 /**
  5 * 自定义事件工具类。
  6 * 该模式也叫观察者模式，该模式的另外一个别名是 订阅/发布 模式。
  7 * 设计这种模式背后的主要动机是促进松散耦合。
  8 * 在这种模式中，并不是一个对象调用另一对象的方法，
  9 * 而是一个对象订阅另一个对象的特定活动并在该活动发生改变后获得通知。
 10 * 订阅者也称之为观察者，而被观察的对象成为发布者。
 11 * 当发生了一个重要的事件时，发布者将通知（调用）所有订阅者并且以事件对象的形式传递消息。
 12 * @class
 13 * @param {Object} obj 要进行绑定事件的目标对象。
 14 * @return {MiniQuery.Event} 返回一个 MiniQuery.Event 的实例。
 15 */
 16 MiniQuery.Event = function Event(obj) {
 17     return new MiniQuery.Event.prototype.init(obj);
 18 };

 30     /**
 31     * 给指定的对象绑定指定类型的事件处理函数。
 32     * @param {Object} obj 要进行绑定事件的目标对象。
 33     * @param {string} eventName 要绑定的事件名称。
 34     * @param {function} fn 事件处理函数。 
 35         在处理函数内部， this 指向参数 obj 对象。
 36     * @param {boolean} [isOnce=false] 指示是否只执行一次事件处理函数，
 37         如果指定为 true，则执行事件处理函数后会将该处理函数从事件列表中移除。
 38     * @example
 39         var obj = { value: 100 };
 40         $.Event.bind(obj, 'show', function(){
 41             console.log(this.value); // this 指向 obj
 42         });
 43         $.Event.bind(obj, {
 44             myEvent: function(){
 45                 console.log(this.value);
 46             },
 47             add: function(value1, value2){
 48                 this.value += (value1 * value2);
 49             }
 50         });
 51         $.Event.trigger(obj, 'show'); //输出 100
 52         $.Event.trigger(obj, 'add', [2, 3]);
 53         $.Event.trigger(obj, 'myEvent'); //输出 106
 54     */
 55     bind: function (obj, eventName, fn, isOnce) {
 56 
 57         var all = mapper.get(obj); //获取 obj 所关联的全部事件的容器 {}
 58         if (!all) { // 首次对 obj 进行绑定事件 
 59             all = {};
 60             mapper.set(obj, all);
 61         }
 62 
 63         switch (typeof eventName) {
 64 
 65             case 'string':  //标准的，单个绑定
 66                 bindItem(eventName, fn);
 67                 break;
 68 
 69             case 'object':  //此时类似为 bind(obj, {click: fn, myEvent: fn}, isOnce)
 70                 if (!$.Object.isPlain(eventName)) {
 71                     throw new Error('当别参数 eventName 为 object 类型时，必须指定为纯对象 {}');
 72                 }
 73 
 74                 isOnce = fn; //参数位置修正
 75                 $.Object.each(eventName, function (key, value) {
 76                     bindItem(key, value);   //批量绑定
 77                 });
 78                 break;
 79 
 80             default:
 81                 throw new Error('无法识别参数 eventName 的类型');
 82                 break;
 83 
 84         }
 85 
 86         isOnce = !!isOnce; //转成 boolean
 87 
 88 
 89         //一个内部的共用函数
 90         function bindItem(eventName, fn) {
 91             if (typeof fn != 'function') {
 92                 throw new Error('参数 fn 必须为一个 function');
 93             }
 94 
 95             var list = all[eventName] || [];
 96 
 97             list.push({
 98                 fn: fn,
 99                 isOnce: isOnce
100             });
101 
102             all[eventName] = list;
103         }
104 
105     },

117         This.bind.apply(null, args);

121     /**
122     * 给指定的对象解除绑定指定类型的事件处理函数。
123     * @param {Object} obj 要进行解除绑定事件的目标对象。
124     * @param {string} [eventName] 要解除绑定的事件名称。
125         如果不指定该参数，则移除所有的事件。
126         如果指定了该参数，其类型必须为 string，否则会抛出异常。
127     * @param {function} [fn] 要解除绑定事件处理函数。
128         如果不指定，则移除 eventName 所关联的所有事件。
129     */
130     unbind: function (obj, eventName, fn) {
131 
132         var all = mapper.get(obj); //获取 obj 所关联的全部事件的容器 {}
133         if (!all) { //尚未存在对象 obj 所关联的事件列表
134             return;
135         }
136 
137 
138         //未指定事件名，则移除所有的事件
139         if (eventName === undefined) {
140             mapper.remove(obj);
141             return;
142         }
143 
144         //指定了事件名
145         if (typeof eventName != 'string') {
146             throw new Error('如果指定了参数 eventName，则其类型必须为 string');
147         }
148 
149         var list = all[eventName];
150         if (!list) { //尚未存在该事件名所对应的事件列表
151             return;
152         }
153 
154         //未指定事件处理函数，则移除该事件名的所有事件处理函数
155         if (!fn) {
156             all[eventName] = [];
157             return;
158         }
159 
160         //移除所有 fn 的项
161         all[eventName] = $.Array.grep(list, function (item, index) {
162             return item.fn !== fn;
163         });
164     },

166     /**
167     * 触发指定的对象上的特定类型事件。
168     * @param {Object} obj 要触发事件的目标对象。
169     * @param {string} eventName 要触发的事件名称。
170     * @param {Array} [args] 要向事件处理函数传递的参数数组。
171     * @return {Array} 返回所有事件处理函数的返回值所组成的一个数组。
172     */
173     trigger: function (obj, eventName, args) {
174 
175         var returns = [];
176 
177         var all = mapper.get(obj); // all = {...}
178         if (!all) {
179             return returns;
180         }
181 
182 
183         var list = all[eventName]; //取得回调列表
184         if (!list) {
185             return returns;
186         }
187 
188         args = args || [];
189 
190 
191         //依次执行回调列表中的函数，并且移除那些只需要执行一次的
192         all[eventName] = $.Array.grep(list, function (item, index) {
193 
194             var value = item.fn.apply(obj, args); //让 fn 内的 this 指向 obj
195             returns.push(value);
196 
197             return !item.isOnce;
198         });
199 
200         return returns;
201     },

203     /**
204     * 触发指定的对象上的特定类型事件，当事件处理函数返回指定的值时将停止继续调用。
205     * @param {Object} obj 要触发事件的目标对象。
206     * @param {string} eventName 要触发的事件名称。
207     * @param {Array} [args] 要向事件处理函数传递的参数数组。
208     * @param [stopValue=false] 要停止继续调用时的返回值。
209         当事件处理函数返回参数 stopValue 所指定的值时，将停止调用后面的处理函数。
210     * @return {Array} 返回已调用的事件处理函数的返回值所组成的一个数组。
211     */
212     triggerStop: function (obj, eventName, args, stopValue) {
213 
214         var returns = [];
215 
216         var all = mapper.get(obj); // all = {...}
217         if (!all) {
218             return returns;
219         }
220 
221 
222         var list = all[eventName]; //取得回调列表
223         if (!list) {
224             return returns;
225         }
226 
227         args = args || [];
228 
229         if (arguments.length == 3) { //不传 stopValue 时，默认为 false
230             stopValue = false;
231         }
232 
233         var items = [];
234         
235 
236         for (var i = 0, len = list.length; i < len; i++) {
237             var item = list[i];
238             var value = item.fn.apply(obj, args); //让 fn 内的 this 指向 obj
239 
240             if (!item.isOnce) {
241                 items.push(item);
242             }
243 
244             returns.push(value);
245 
246             if (value === stopValue) {
247                 break;
248             }
249         }
250 
251         list = list.slice(i + 1);
252         all[eventName] = items.concat(list);
253 
254         return returns;
255     },

257     /**
258     * 检测指定的对象上是否包含特定类型的事件。
259     * @param {Object} obj 要检测的目标对象。
260     * @param {string} [eventName] 要检测的事件名称。
261         当不指定时，则判断是否包含了任意类型的事件。
262     * @param {function} [fn] 要检测的事件处理函数。
263     * @return {boolean} 返回一个布尔值，该布尔值指示目标对象上是否包含指否类型的事件以及指定的处理函数。
264         如果是则返回 true；否则返回 false。
265     */
266     has: function (obj, eventName, fn) {
267 
268         var all = mapper.get(obj); // all = {...}
269         if (!all) {   //尚未绑定任何类型的事件
270             return false;
271         }
272 
273         if (eventName === undefined) {   //未指定事件名，则判断是否包含了任意类型的事件
274             if ($.Object.isEmpty(all)) { // 空对象 {}
275                 return false;
276             }
277 
278             var hasEvent = false;
279 
280             $.Object.each(all, function (eventName, list) {
281                 if (list && list.length > 0) {
282                     hasEvent = true;
283                     return false; // break, 只有在回调函数中明确返回 false 才停止循环。
284                 }
285             });
286 
287             return hasEvent;
288         }
289 
290 
291         //指定了事件名
292         if (typeof eventName != 'string') {
293             throw new Error('如果指定了参数 eventName，则其类型必须为 string');
294         }
295 
296         var list = all[eventName]; //取得回调列表
297         if (!list || list.length == 0) {
298             return false;
299         }
300 
301         if (fn === undefined) { //未指定回调函数
302             return true;
303         }
304 
305         //从列表中搜索该回调函数
306         return $.Array.find(list, function (item, index) {
307             return item.fn === fn;
308         });
309 
310     },

314     * 如果目标对象的指定类型事件中存在该处理函数，则移除它；否则会添加它。
315     * @param {Object} obj 要进行绑定/解除绑定事件的目标对象。
316     * @param {string} eventName 要绑定/解除绑定的事件名称。
317     * @param {function} fn 事件处理函数。 
318     * @param {boolean} [isOnce=false] 指示是否只执行一次事件处理函数，
319         如果指定为 true，则执行事件处理函数后会将该处理函数从事件列表中移除。
320     * @example
321         var obj = { value: 100 };
322         var fn = function() {
323             console.log(this.value); // this 指向 obj
324         }

326         $.Event.trigger(obj, 'show'); //输出 100
327 
328         $.Event.toggle(obj, 'show', fn); //因为 'show' 事件列表中已存在 fn 函数，所以会给移除
329         $.Event.trigger(obj, 'show'); //fn 函数已给移除，因为不产生输出
330     */

332 
333         if (This.has(obj, eventName, fn)) {
334             This.unbind(obj, eventName, fn);
335         }
336         else {
337             This.bind(obj, eventName, fn, isOnce);
338         }

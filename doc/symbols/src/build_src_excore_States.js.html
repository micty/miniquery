  3 /**
  4 * 有限状态机类
  5 * @class
  6 * 有限状态机是一个非常有用的模型，可以模拟世界上大部分事物。
  7 * 简单说，它有三个特征：
  8 *   1.状态总数（state）是有限的。
  9 *   2.任一时刻，只处在一种状态之中。
 10 *   3.状态转换规则是确定的。
 11 * 它对 JavaScript 的意义在于，很多对象可以写成有限状态机。
 12 * 当对象的状态发生变化时，可以触发特定事件进行回调函数的调用。
 13 * 有限状态机的写法，逻辑清晰，表达力强，有利于封装事件。
 14 * 一个对象的状态越多、发生的事件越多，就越适合采用有限状态机的写法。
 15 */
 16 MiniQuery.States = (function ($, Mapper) {
 17 
 18 
 19     var guidKey = Mapper.getGuidKey();
 20     var mapper = new Mapper();
 21 
 22 
 23     /**
 24     * @inner
 25     * 构造函数
 26     */
 27     function States(current, list) {
 28 
 29 
 30         this[guidKey] = $.String.random();
 31 
 32 
 33         var states = {};//状态集合
 34         var paths = {}; //路径集合
 35 
 36         mapper.set(this, {
 37             states: states,
 38             paths: paths,
 39             histories: [current] //经历过的状态
 40         });
 41 
 42         var self = this;
 43         var bind = $.Event.bind;
 44         list = normalize(list);
 45 
 46         $.Array.each(list, function (item, index) {
 47 
 48             var from = item.from;
 49             var to = item.to;
 50             var fn = item.fn;
 51 
 52             if (from && to) {
 53                 bind(self, from + '->' + to, fn);
 54                 paths[from + '->' + to] = true;
 55                 states[from] = true;
 56                 states[to] = true;
 57             }
 58             else if (from) {
 59                 bind(self, from + '->', fn);
 60                 states[from] = true;
 61             }
 62             else if (to) {
 63                 bind(self, '->' + to, fn);
 64                 states[to] = true;
 65             }
 66             else {
 67                 bind(self, '->', fn);
 68             }
 69 
 70         });
 71 
 72     }
 73 
 74 
 75     /**
 76     * @inner 
 77     * 静态方法
 78     * 标准化参数 states，以获得统一的描述格式
 79     */
 80     function normalize(states) {
 81 
 82         var list = [];
 83 
 84         if ($.Object.isPlain(states)) { //此时为 { ... }
 85 
 86             $.Object.each(states, function (key, fn) {
 87 
 88                 var pair = key.split('->');
 89                 var from = pair[0];
 90                 var to = pair[1];
 91 
 92                 if (from.indexOf('|') > 0 || to.indexOf('|') > 0) { // from 和 to 中至少有一个含有 '|'
 93                     from = from.split('|');
 94                     to = to.split('|');
 95                     var a = combine(from, to, fn);
 96                     list = list.concat(a);
 97                 }
 98                 else {
 99                     list.push({
100                         from: from,
101                         to: to,
102                         fn: fn
103                     });
104                 }
105             
106             });
107         }
108         else if ($.Object.isArray(states)) { // 此时为 [ ... ]
109 
110             $.Array.each(states, function (item, index) {
111 
112                 var from = item.from;
113                 var to = item.to;
114                 var fn = item.fn;
115 
116                 var from_isArray = from instanceof Array;
117                 var to_isArray = to instanceof Array;
118 
119                 if (from_isArray || to_isArray) { // from 和 to 中至少有一个为数组
120 
121                     if (!from_isArray) { //此时 to 必为 []
122                         from = [from];
123                     }
124                     else if (!to_isArray) { //此时 from 必为 []
125                         to = [to];
126                     }
127 
128                     var a = combine(from, to, fn);
129                     list = list.concat(a);
130                 }
131                 else {
132                     list.push(item);
133                 }
134 
135             });
136         }
137 
138         return list;
139     }

142     /**
143     * @inner
144     */
145     function combine(A, B, fn) {
146     
147         var groups = $.Array.descartes(A, B);
148     
149         return $.Array.map(groups, function (item, index) {
150             return {
151                 from: item[0],
152                 to: item[1],
153                 fn: fn
154             };
155         });
156     }

162     $.extend(States.prototype, { /**@lends MiniQuery.States#*/
163     
164         /**
165         * 把当前状态转换到指定的状态。
166         * @param {string} name 要转换到的目标状态。
167         * @param {Array} [args] 要向状态转换时触发的处理函数传递的参数数组。
168         * @example
169         *
170         */
171         to: function (name, args) {
172 
173             var current = this.current();
174 
175             if (!this.has(current, name)) {
176                 throw new Error($.String.format('不存在从状态 {0} 到状态 {1} 的路径', current, name));
177             }
178 
179             mapper.get(this).histories.push(name); //先改变状态，再触发事件
180 
181             $.Event.trigger(this, current + '->', args);
182             $.Event.trigger(this, current + '->' + name, args);
183             $.Event.trigger(this, '->' + name, args);
184             $.Event.trigger(this, '->', args);
185 
186 
187         },
188 
189         backward: function (args) {
190             var histories = mapper.get(this).histories;
191             var index = histories.length - 2;
192             if (index >= 0) {
193                 this.to(histories[index], args);
194             }
195         },
196 
197         current: function () {
198             var histories = mapper.get(this).histories;
199             return histories[histories.length - 1];
200         },
201 
202         has: function (from, to) {
203 
204             var states = mapper.get(this).states;
205             if (to === undefined) { //此时为 has(from) 即确定是否包含某个状态
206                 return !!states[from];
207 
208             }
209 
210             //此时为 has(from, to) 即是否存在从 from 到 to 的路径
211             var paths = mapper.get(this).paths;
212             return !!paths[from + '->' + to];
213         },
214 
215         states: function () {
216             var states = mapper.get(this).states;
217             return $.Object.getKeys(states);
218         },
219 
220         paths: function () {
221             var paths = mapper.get(this).paths;
222             return $.Object.getKeys(paths);
223         },
224 
225         histories: function () {
226             var histories = mapper.get(this).histories;
227             return histories.slice(0);
228         },
229 
230         length: function () {
231             return this.states().length;
232         }
233 
234     });
235 
236     //for test
237     //States.mapper = mapper;
238 
239     return States;

 11     /**
 12     * 用工厂模式创建一个类。
 13     * @param {Object} instances 类中的实例成员，即 prototype 的中成员。
 14     * @param {function} [instances.constructor] 构造函数，要创建的类的真正的构造函数；
 15         如果不提供，则使用一个默认的空函数实现。
 16     * @param {Object} [statics] 类中的静态成员。
 17         如果想给类提供一些静态成员，则可指定此参数。
 18     * @return {function} 返回一个类的构造器。
 19     * @example
 20         var Person = $.Class.create( { //实例成员
 21             //构造函数
 22             constructor: function(name, age) {
 23                 this.name = name;
 24                 this.age = age;
 25                 this.type = Person.getType();
 26             },
 27                 
 28             sayHi: function() {
 29                 console.log(' hi, my type is ' + this.type );
 30                 console.log(' and my name is ' + this.name );
 31                 console.log(' and I am ' + this.age + ' years old');
 32             }
 33     
 34         }, { //静态成员
 35             getType: function() {
 36                 return 'Person';
 37             }
 38         });
 39             
 40         var man = new Person('micty', 29);
 41         man.sayHi();
 42         console.log( Person.getType() );

 44     create: function (instances, statics) {
 45         var prototype = instances;
 46 
 47         //如果未提供构造函数，则使用一个默认的实现
 48         prototype.constructor = prototype.constructor || function () { };
 49         prototype.constructor.prototype = prototype;
 50 
 51         $.extend(prototype.constructor, statics); //静态成员
 52 
 53         return prototype.constructor;
 54     },
 55 
 56     /**
 57     * 用继续一个类的方式去创建一个新的类。
 58     * @param {function} SuperClass 要继续的超类，即父类。
 59     * @param {Object} instances 类中的实例成员，即 prototype 的中成员。
 60     * @param {function} [instances.constructor] 构造函数，要创建的类的真正的构造函数；
 61         如果不提供，则使用一个默认的空函数实现。
 62     * @param {Object} [statics] 类中的静态成员。
 63         如果想给类提供一些静态成员，则可指定此参数。
 64     * @return {function} 返回一个类的构造器。
 65     * @example
 66         var Person = $.Class.create( { //实例成员
 67             //构造函数
 68             constructor: function(name, age) {
 69                 this.name = name;
 70                 this.age = age;
 71                 this.type = Person.getType();
 72             },
 73                 
 74             sayHi: function() {
 75                 console.log(' hi, my type is ' + this.type );
 76                 console.log(' and my name is ' + this.name );
 77                 console.log(' and I am ' + this.age + ' years old');
 78             }
 79         }, { //静态成员
 80             getType: function() {
 81                 return 'Person';
 82             }
 83         });
 84             
 85         var Student = $.Class.inherit(Person, {
 86             constructor: function(name, age, school) {
 87                 Person.call(this, name, age);
 88                 this.school = school;
 89                 this.type = Student.getType();
 90             },
 91                 
 92             sayHi: function() {
 93                 Person.prototype.sayHi.call(this);
 94                 console.log(' and I am at ' + this.school + ' school' );
 95             }
 96         }, {
 97             
 98             getType: function() {
 99                 return 'Student';
100             }
101         });
102             
103         var pp = new Person('abc', 31);
104         pp.sayHi();
105         console.log(Person.getType());
106             
107         var st = new Student('micty', 29, 'SZU');
108         st.sayHi();
109         console.log( Student.getType() );
110         console.log( st instanceof Student );   //true
111         console.log( st instanceof Person );    //true
112         console.log( st instanceof Object );    //true
113             
114     */
115     inherit: function (SuperClass, instances, statics) {
116         instances = $.extend(new SuperClass(), instances);
117         statics = $.extend({}, SuperClass, statics);
118 
119         return This.create(instances, statics);
120     }

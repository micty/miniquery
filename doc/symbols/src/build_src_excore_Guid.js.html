  3 /**
  4 * 全局唯一标识符工具类（GUID: Globally Unique Identifier）。
  5 * 提供全局 ID 生成和管理的方法，可用于给组件创建 ID。
  6 * @namespace
  7 */
  8 MiniQuery.Guid = (function ($, This) {
  9 
 10 
 11 
 12 var anonymousName = '__anonymousName_' + $.String.random();
 13 
 14 var group$counter = {}; //分组计数器，从 1 开始
 15 var guids = {};         //用于 id 的注册，以判断某个 id 在全局内是否已存在
 16 
 17     
 18 
 19 
 20 return $.extend(This, { /**@lends MiniQuery.Guid */
 21 
 22     /**
 23     * 获取指定分组的下一个计数器。
 24     * 如果尚未存在该分组，则从 1 开始计数。
 25     * @param {string} groupName 分组的名称。 
 26     * @return {int} 返回一个整数。
 27     * @example 
 28         $.Guid.next('group1'); //1
 29         $.Guid.next('group2'); //1
 30         $.Guid.next('group2'); //2
 31         $.Guid.next('group1'); //2
 32         $.Guid.next(); //1
 33         $.Guid.next(); //2
 34     */
 35     next: function (groupName) {
 36 
 37         if (arguments.length == 0) { // next()
 38             groupName = anonymousName;
 39         }
 40 
 41         var id = group$counter[groupName] || 0;
 42         id = id + 1;
 43         group$counter[groupName] = id;
 44 
 45         return id;
 46     },
 47 
 48     /**
 49     * 给指定分组的下计数器清零。
 50     * 如果尚未存在该分组，则先创建该分组。
 51     * @param {string} groupName 分组的名称。 
 52     * @example 
 53         $.Guid.reset('group1');
 54         $.Guid.reset('group2');
 55         $.Guid.reset();
 56     */
 57     reset: function (groupName) {
 58         if (arguments.length == 0) { // reset()
 59             groupName = anonymousName;
 60         }
 61 
 62         group$counter[groupName] = 0;
 63     },
 64 
 65     /**
 66     * 获取指定分组的并且按指定字符串格式化的 id 值。
 67     * 如果尚未存在该分组，则从 1 开始计数。
 68     * @param {string} groupName 分组的名称。 
 69     * @param {string} [formater='{0}'] 要填充的格式化字符串。
 70     * @return {string} 返回格式化后的 id 字符串。
 71     * @example 
 72         $.Guid.get('group1', 'label_{0}_{1}'); // 'label_1_2'
 73         $.Guid.get('group1', 'label_{0}_{1}'); // 'label_3_4'
 74         $.Guid.get('group2', 'button_{0}_{1}'); // 'button_1_2'
 75         $.Guid.get('group2', 'button_{0}_{0}'); // 'button_3_3'
 76     */
 77     get: function (groupName, formater) {
 78 
 79         formater = formater || '{0}';
 80 
 81         var reg = /\{\d+\}/g;
 82         var items = formater.match(reg); //把所有的 {0}、{1} 等提取出来
 83 
 84         var ids = $.Array.map(items, function (item, index) {
 85             return This.next(groupName);
 86         });
 87 
 88         var id = $.String.format(formater, ids);
 89         This.add(id);
 90 
 91         return id;
 92     },
 93 
 94     /**
 95     * 把指定的 id 值添加(注册)到 guid 列表中，以表示该 id 已被占用。
 96     * @param {string} id 字符串的 ID 值。
 97     * @example 
 98         $.Guid.add('myId');
 99     */
100     add: function (id) {
101         guids[id] = true;
102     },
103 
104     /**
105     * 把指定的 id 值从 guid 列表中移除，以表示该 id 已被释放。
106     * @param {string} id 字符串的 ID 值。
107     * @example 
108         $.Guid.remove('myId');
109     */
110     remove: function (id) {
111         delete guids[id];
112     },
113 
114     /**
115     * 判断指定的 id 值已给占用。
116     * @param {string} id 字符串的 ID 值。
117     * @example 
118         $.Guid.exist('myId');
119     */
120     exist: function (id) {
121         return !!guids[id];
122     }
123 });
124 
125 
126 })(MiniQuery, {});

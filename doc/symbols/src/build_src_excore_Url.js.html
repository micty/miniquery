 16     * 获取指定 Url 的查询字符串中指定的键所对应的值。
 17     * @param {string} url 要进行获取的 url 字符串。
 18     * @param {string} [key] 要检索的键。
 19     * @param {boolean} [ignoreCase=false] 是否忽略参数 key 的大小写。 默认区分大小写。
 20         如果要忽略 key 的大小写，请指定为 true；否则不指定或指定为 false。
 21         当指定为 true 时，将优先检索完全匹配的键所对应的项；若没找到然后再忽略大小写去检索。
 22     * @retun {string|Object|undefined} 返回一个查询字符串值。
 23         当不指定参数 key 时，则获取全部查询字符串，返回一个等价的 Object 对象。
 24         当指定参数 key 为一个空字符串，则获取全部查询字符串，返回一个 string 类型值。
 25     * @example
 26         $.Url.getQueryString('http://www.demo.com?a=1&b=2#hash', 'a');  //返回 '1'
 27         $.Url.getQueryString('http://www.demo.com?a=1&b=2#hash', 'c');  //返回 undefined
 28         $.Url.getQueryString('http://www.demo.com?a=1&A=2#hash', 'A');  //返回 2
 29         $.Url.getQueryString('http://www.demo.com?a=1&b=2#hash', 'A', true);//返回 1
 30         $.Url.getQueryString('http://www.demo.com?a=1&b=2#hash', '');   //返回 'a=1&b=2'
 31         $.Url.getQueryString('http://www.demo.com?a=1&b=2#hash');       //返回 {a: '1', b: '2'}
 32         $.Url.getQueryString('http://www.demo.com?a=&b=');              //返回 {a: '', b: ''}
 33         $.Url.getQueryString('http://www.demo.com?a&b');                //返回 {a: '', b: ''}
 34         $.Url.getQueryString('http://www.demo.com?a', 'a');             //返回 ''
 35     */
 36     getQueryString: function (url, key, ignoreCase) {
 37 
 38         var beginIndex = url.indexOf('?');
 39         if (beginIndex < 0) { //不存在查询字符串
 40             return;
 41         }
 42 
 43         var endIndex = url.indexOf('#');
 44         if (endIndex < 0) {
 45             endIndex = url.length;
 46         }
 47 
 48         var qs = url.slice(beginIndex + 1, endIndex);
 49         if (key === '') { //获取全部查询字符串的 string 类型
 50             return decodeURIComponent(qs);
 51         }
 52 
 53 
 54         var obj = $.Object.parseQueryString(qs);
 55 
 56         if (key === undefined) { //未指定键，获取整个 Object 对象
 57             return obj;
 58         }
 59 
 60         if (!ignoreCase || key in obj) { //区分大小写或有完全匹配的键
 61             return obj[key];
 62         }
 63 
 64         //以下是不区分大小写
 65         key = key.toString().toLowerCase();
 66 
 67         for (var name in obj) {
 68             if (name.toLowerCase() == key) {
 69                 return obj[name];
 70             }
 71         }
 72

 81     * @param {string|Object} key 要添加的查询字符串的键。
 82         当传入一个 Object 对象时，会对键值对进行递归组合编码成查询字符串。
 83     * @param {string} [value] 要添加的查询字符串的值。
 84     * @retun {string} 返回组装后的新的 Url。
 85     * @example
 86         //返回 'http://www.demo.com?a=1&b=2&c=3#hash'
 87         $.Url.setQueryString('http://www.demo.com?a=1&b=2#hash', 'c', 3);  
 88         
 89         //返回 'http://www.demo.com?a=3&b=2&d=4#hash'
 90         $.Url.setQueryString('http://www.demo.com?a=1&b=2#hash', {a: 3, d: 4});

 92     addQueryString: function (url, key, value) {
 93 
 94         var qs = This.getQueryString(url) || {}; //先取出原来的
 95         
 96         if (typeof key == 'object') {
 97             $.Object.extend(qs, key);
 98         }
 99         else {
100             qs[key] = value;
101         }
102 
103 
104         //过滤掉值为 null 的项
105         var obj = {};
106 
107         for (var key in qs) {
108             var value = qs[key];
109 
110             if (value === null) {
111                 continue;
112             }
113             else {
114                 obj[key] = value;
115             }
116 
117         }
118 
119         return This.setQueryString(url, obj);
120

147     * 把指定的 Url 和查询字符串组装成一个新的 Url。
148     * 注意，该方法会删除之前的查询字符串。
149     * @param {string} url 组装前的 url。
150     * @param {string|Object} key 要设置的查询字符串的键。
151         当传入一个 Object 对象时，会对键值对进行递归组合编码成查询字符串。
152     * @param {string} [value] 要添加的查询字符串的值。
153     * @retun {string} 返回组装后的新的 Url。
154     * @example
155         //返回 'http://www.demo.com?c=3#hash'
156         $.Url.setQueryString('http://www.demo.com?a=1&b=2#hash', 'c', 3);

158         //返回 'http://www.demo.com?a=3&d=4#hash'

161     setQueryString: function (url, key, value) {
162         
163         var type = typeof key;
164         var isValueType = (/^(string|number|boolean)$/).test(type);
165 
166         var qs = '';
167 
168         if (arguments.length == 2 && isValueType) { //setQueryString(url, qs);
169             qs = encodeURIComponent(key);
170         }
171         else {
172             var obj = type == 'object' ? key : $.Object.make(key, value);
173             qs = $.Object.toQueryString(obj);
174         }
175         
176 
177 
178         var hasQuery = url.indexOf('?') > -1;
179         var hasHash = url.indexOf('#') > -1;
180         var a;
181 
182         if (hasQuery && hasHash) {
183             a = url.split(/\?|#/g);
184             return a[0] + '?' + qs + '#' + a[2];
185         }
186 
187         if (hasQuery) {
188             a = url.split('?');
189             return a[0] + '?' + qs;
190         }
191 
192         if (hasHash) {
193             a = url.split('#');
194             return a[0] + '?' + qs + '#' + a[1];
195         }
196 
197         return url + '?' + qs;

217     hasQueryString: function (url, key, ignoreCase) {

228

230             return true;
231         }
232 
233         if (ignoreCase) { //明确指定了忽略大小写
234 
235             key = key.toString().toLowerCase();
236             for (var name in obj) {
237                 if (name.toLowerCase() == key) {
238                     return true;
239                 }
240             }
241         }

243         //区分大小写，但没找到
244         return false;

250     * 获取指定 Url 的哈希中指定的键所对应的值。
251     * @param {string} url 要进行获取的 Url 字符串。
252     * @param {string} [key] 要检索的键。

256     * @retun {string|Object|undefined} 返回一个查询字符串值。
257         当不指定参数 key 时，则获取全部哈希值，对其进行 unescape 解码，
258         然后返回一个等价的 Object 对象。
259         当指定参数 key 为一个空字符串，则获取全部哈希(不解码)，返回一个 string 类型值。
260     * @example
261         $.Url.getHash('http://www.demo.com?query#a%3D1%26b%3D2', 'a');  //返回 '1'

265         $.Url.getHash('http://www.demo.com?query#a%3D1%26b%3D2', '');   //返回 'a%3D1%26b%3D2'
266         $.Url.getHash('http://www.demo.com?query#a%3D1%26b%3D2');       //返回 {a: '1', b: '2'}

268         $.Url.getHash('http://www.demo.com??query#a%26b');              //返回 {a: '', b: ''}
269         $.Url.getHash('http://www.demo.com?query#a', 'a');              //返回 ''

295             return obj[key];
296         }
297 
298 
299         //以下是不区分大小写
300         key = key.toString().toLowerCase();
301 
302         for (var name in obj) {
303             if (name.toLowerCase() == key) {
304                 return obj[name];
305             }

308     /**
309     * 把指定的哈希设置到指定的 Url 上。
310     * 该方法会对哈希进行 escape 编码，再设置到 Url 上，以避免哈希破坏原有的 Url。
311     * 同时原有的哈希会移除掉而替换成新的。
312     * @param {string} url 要设置的 url 字符串。
313     * @param {string|number|boolean|Object} key 要设置的哈希的键。
314         当传入一个 Object 对象时，会对键值对进行递归编码成查询字符串， 然后用 escape 编码来设置哈希。
315         当传入的是一个 string|number|boolean 类型，并且不传入第三个参数， 则直接用 escape 编码来设置哈希。
316     * @param {string} [value] 要添加的哈希的值。
317     * @retun {string} 返回组装后的新的 Url 字符串。
318     * @example
319         //返回 'http://www.demo.com?#a%3D1'
320         $.Url.setHash('http://www.demo.com', 'a', 1);  
321         
322         //返回 'http://www.demo.com?query#a%3D3%26d%3D4'
323         $.Url.setHash('http://www.demo.com?query#a%3D1%26b%3D2', {a: 3, d: 4});  
324 
325         //返回 'http://www.demo.com?query#a%3D3%26d%3D4'
326         $.Url.setHash('http://www.demo.com?query#a%3D1%26b%3D2', 'a=3&b=4'); 
327         
328     */
329     setHash: function (url, key, value) {
330         
331         var type = typeof key;
332         var isValueType = (/^(string|number|boolean)$/).test(type);
333 
334 
335         var hash = '';
336 
337         if (arguments.length == 2 && isValueType) {
338             hash = String(key);
339         }
340         else {
341             var obj = type == 'object' ? key : $.Object.make(key, value);
342             hash = $.Object.toQueryString(obj);
343         }
344 
345 
346         hash = escape(hash); //要进行编码，避免破坏原有的 Url
347 
348         var index = url.lastIndexOf('#');
349         if (index > -1) {
350             url = url.slice(0, index);
351         }
352 
353         return url + '#' + hash;
354

  9 /**
 10 * 针对 IE 创建最优版本的 XMLDocument 对象。
 11 * @inner
 12 */
 13 function createDocument() {
 14     var cache = arguments.callee;
 15 
 16     if (!cache['version']) { //首次创建
 17         var versions = [
 18             'MSXML2.DOMDocument.6.0',
 19             'MSXML2.DOMDocument.3.0',
 20             'MSXML2.DOMDocument'
 21         ];
 22 
 23         for (var i = 0, len = versions.length; i < len; i++) {
 24             try {
 25                 var xmldoc = new ActiveXObject(versions[i]);
 26                 cache['version'] = versions[i]; //缓存起来
 27                 return xmldoc;
 28             }
 29             catch (ex) { //跳过
 30 
 31             }
 32         }
 33     }
 34 
 35     return new ActiveXObject(cache['version']);
 36 }

 39 /**
 40 * 解析一个 XML 节点的属性集合成一个键/值形式的 Object 对象。
 41 * 可以指定第二个参数是否为深层解析，即属性值中包含查询字符串编码时，可以进一步解析成对象。
 42 * @inner
 43 */
 44 function parseAttributes(node, deep) {
 45 
 46     var obj = {};
 47     var parseQueryString = $.Object.parseQueryString;
 48 
 49     //把类数组对象转成真正的数组
 50     $.Array(node.attributes).each(function (item, index) {
 51 
 52         if (item.specified) { //兼容性写法，过滤出自定义特性，可用于 HTML 节点的 attributes
 53 
 54             var value = item.value;
 55 
 56             if (deep && value.indexOf('=') > 0) { //深层次解码
 57                 obj[item.name] = parseQueryString(value);
 58             }
 59             else {
 60                 obj[item.name] = value;
 61             }
 62         }
 63 
 64     });
 65 
 66     return obj;
 67 }

 69 /**
 70 * 跨浏览器解析 XML 数据(字符串)，返回一个 XMLDocument 对象。
 71 * @inner
 72 */
 73 function parseString(data) {
 74     var xmldoc = null;
 75     var impl = document.implementation;
 76 
 77     if (window.DOMParser) { //标准
 78 
 79         xmldoc = (new DOMParser()).parseFromString(data, 'text/xml');
 80         var errors = xmldoc.getElementsByTagName('parsererror');
 81         if (errors.length > 0) {
 82             throw new Error('XML 解析错误: ' + errors[0].textContent);
 83         }
 84     }
 85     else if (impl.hasFeature('LS', '3.0')) { // DOM3
 86     
 87         var parser = impl.createLSParser(impl.MODE_SYNCHRONOUS, null);
 88         var input = impl.createInput();
 89         input.stringData = data;
 90         xmldoc = parser.parse(input); //如果解析错误，则抛出异常
 91     }
 92     else { // IE
 93     
 94         xmldoc = createDocument();
 95         xmldoc.loadXML(data);
 96         if (xmldoc.parseError.errorCode != 0) {
 97             throw new Error('XML 解析错误: ' + xmldoc.parseError.reason);
 98         }
 99     }
100 
101     if (!xmldoc) {
102         throw new Error('没有可用的 XML 解析器');
103     }
104 
105     return xmldoc;
106 }

108 /**
109 * 把一个 Object 对象转成等价的 XML 字符串。
110 *
111 * 注意：传入的 Object 对象中，简单属性表示该节点自身的属性；
112         数组表示该节点的子节点集合；
113 *       属性值只能是 string、number、boolean 三种值类型。
114 * @inner
115 */
116 function Object_to_String(obj, name) {
117     var fn = arguments.callee;
118 
119     if (!name) { //处理(重载) Object_to_String(obj) 的情况
120     
121         for (name in obj) {
122             return fn(obj[name], name);
123         }
124 
125         throw new Error('参数 obj 中不包含任何成员');
126     }
127 
128 
129     //正常情况 Object_to_String(obj, name)
130 
131     var attributes = [];
132     var children = [];
133 
134     for (var key in obj) {
135         if ($.Object.isArray(obj[key])) { //处理子节点
136         
137             $.Array.each(obj[key], function (child, index) {
138                 children.push(fn(child, key));
139             });
140             continue;
141         }
142 
143         //处理属性
144         var type = typeof obj[key];
145         if (type == 'string' || type == 'number' || type == 'boolean') {
146             var value = String(obj[key]).replace(/"/g, '\\"');
147             attributes.push($.String.format('{0}="{1}"', key, value));
148         }
149         else {
150             throw new Error('非法数据类型的属性值: ' + key);
151         }
152     }
153 
154     return $.String.format('<{name} {attributes}>{children}</{name}>', {
155         name: name,
156         attributes: attributes.join(' '),
157         children: children.join(' \r\n')
158     });
159 }

261         if (!obj || typeof obj != 'object') {

287     /**
288     * 把一个 XML 字符串或 XMLDocument 对象或 XML 节点解析成等价结构的 Object 对象
289     * @param {string|XMLDocument|XMLNode} node 
290         要进行解析的 XML 字符串或 XMLDocument 对象或 XML 节点。<br />
291         注意：表示 XML 节点中的属性名不能跟直接子节点中的任何一个节点名相同。
292     * @param {boolean} deep 
293         指示是否对节点值中进一步按查询字符串的解析成等价的对象。<br />
294         如 "a=1&b=2&c=A%3D100%26B%3D200" 会被解析成对象 {a:1, b:2, c:{A:100, B:200}}
295     * @return {Object} 返回一个等价的对象。<br />
296         返回的 Object 对象中，属性表示该节点自身的属性；数组表示该节点的子节点集合。
297     *@example
298     <xmp>
299         var xml = '
300             <Person num="2" code="0"> \
301                 <user id="1" name="micty" age="28"> \
302                     <book id="1" name="C++" price="100"></book> \

307                     <book id="1" name="CPP" price="100"></book> \
308                     <book id="2" name="Linux" price="156"></book> \
309                 </user> \
310             </Person>';
311     </xmp>
312         var obj = $.XML.toObject(xml); 
313     得到：
314         obj = {
315             Person: {
316                 num: "2", code: "0",
317                 user: [
318                 {
319                     id: "1", name: "micty", age: "28",
320                     book: [
321                         { id: "1", name: "C++", price: "100" },
322                         { id: "2", name: "C#.NET", price: "256" },
323                         { id: "3", name: "JavaScript", price: "218" }
324                     ]
325                 },
326                 {
327                     id: "2", name: "solomon", age: "25",
328                     book: [
329                         { id: "1", name: "CPP", price: "100" },
330                         { id: "2", name: "Linux", price: "156" }
331                     ]
332                 }]
333             }
334         };
335     */
336     toObject: function (node, deep) {
337 
338         var fn = arguments.callee;  //引用自身，递归用到
339 
340         if (typeof node == 'string') { //传入的是 XML 的字符串
341             var data = node;
342             var xmlDoc = parseString(data); //把字符串解析成 XMLDocument 对象
343             return fn(xmlDoc, deep);
344         }
345 
346         if (node && node.documentElement) { //传入的是 XMLDocument 对象
347             var xmlDoc = node;
348             var obj = {};
349             obj[xmlDoc.documentElement.nodeName] = fn(xmlDoc.documentElement, deep); //取根节点进行解析
350             return obj;
351         }
352 
353 
354         //以下处理的是 XML 节点的情况
355 
356         if (!node || !node.nodeName) {
357             throw new Error('参数 node 错误：非法的 XML 节点');
358         }
359 
360 
361         var obj = parseAttributes(node, deep); //把节点属性转成键值对 obj
362 
363         var childNodes = $.Array.parse(node.childNodes); //把类数组的子节点列表转成真正的数组
364 
365         //处理 <abc ...>xxx</abc> 这样的情况：obj.value = xxx;
366         if (childNodes.length == 1) { //只有一个子节点
367             var leaf = childNodes[0];
368             if (leaf.nodeType == 3) { // TextNode 文本节点
369                 obj['value'] = leaf.nodeValue; //增加一个默认字段 value
370                 return obj;
371             }
372         }
373 
374         //过虑出真正的元素节点。IE 中 node 节点 没有 children 属性，因此用 childNodes 是兼容的写法
375         $.Array(childNodes).grep(function (item, index) {
376             return item.nodeType === 1; //元素节点
377 
378         }).each(function (child, index) {
379             var name = child.nodeName; //标签名，如 div
380 
381             if (!obj[name]) { //同类标签名，汇合到同一个数组中
382                 obj[name] = [];
383             }
384 
385             obj[name].push(fn(child));
386         });
387 
388         return obj;
389 
390     }

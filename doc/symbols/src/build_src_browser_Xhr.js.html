 15     create: function () {

 17         //下面使用了惰性载入函数的技巧，即在第一次调用时检测了浏览器的能力并重写了接口

 19

 21 
 22         } : window.ActiveXObject ? function () { //IE
 23 
 24             var cache = arguments.callee;
 25             var key = 'version';
 26 
 27             if (!cache[key]) { //首次创建
 28 
 29                 var versions = [
 30                     'MSXML2.XMLHttp.6.0',
 31                     'MSXML2.XMLHttp.3.0',
 32                     'MSXML2.XMLHttp'
 33                 ];
 34 
 35                 for (var i = 0, len = versions.length; i < len; i++) {
 36                     try {
 37                         var xhr = new ActiveXObject(versions[i]);
 38                         cache[key] = versions[i];
 39                         return xhr;
 40                     }
 41                     catch (ex) { //跳过
 42 
 43                     }
 44                 }
 45             }
 46 
 47             return new ActiveXObject(cache[key]);
 48 
 49         } : function () {
 50             throw new Error('没有可用的 XHR 对象');
 51         };
 52 
 53 
 54         This.create = fn;
 55 
 56         return fn();
 57     }

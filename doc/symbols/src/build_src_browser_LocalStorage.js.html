  3 /**
  4 * 本地存储工具类
  5 * @namespace
  6 */
  7 MiniQuery.LocalStorage = (function (localStorage) {
  8 
  9 
 10     if (!localStorage) { //不支持
 11         return null;
 12     }
 13 
 14 
 15     var id = '__MiniQuery.LocalStorage__';
 16     var all = localStorage.getItem(id) || null;
 17     all = JSON.parse(all) || {};
 18 
 19 
 20     function set(key, value) {
 21 
 22         if (key in all && all[key] === value) { //已存在
 23             return;
 24         }
 25 
 26         all[key] = value;
 27 
 28         var json = JSON.stringify(all);
 29         localStorage.setItem(id, json);
 30         
 31     }
 32 
 33 
 34 
 35     function get(key) {
 36 
 37         return all[key];
 38 
 39     }
 40 
 41     function remove(key) {
 42         delete all[key];
 43         var json = JSON.stringify(all);
 44         localStorage.setItem(id, json);
 45     }
 46 
 47     function clear() {
 48         all = {};
 49         var json = JSON.stringify(all);
 50         localStorage.setItem(id, json);
 51     }
 52 
 53 
 54     function each(fn) {
 55 
 56         for (var key in all) {
 57             fn(key, all[key]);
 58         }
 59 
 60     }
 61 
 62     function length() {
 63         return keys().length;
 64     }
 65 
 66     function keys() {
 67         if (typeof Object.keys == 'function') {
 68             return Object.keys(all);
 69         }
 70 
 71         var a = [];
 72         for (var key in all) {
 73             a.push(key);
 74         }
 75 
 76         return a;
 77     }
 78 
 79     function key(index) {
 80 
 81         return keys()[index];
 82     }
 83 
 84 
 85 
 86 
 87 
 88 
 89     return {
 90         set: set,
 91         get: get,
 92         remove: remove,
 93         clear: clear,
 94         each: each,
 95         length: length,
 96         keys: keys,
 97         key: key
 98     };
 99 
100 
101 
102 })(localStorage);

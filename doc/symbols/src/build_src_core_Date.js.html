 30     *   1.Date 实例<br />
 31     *   2.string 字符串，包括调用 Date 实例的 toString 方法得到的字符串，也包括以下格式: 
 32         <pre>
 33             yyyy-MM-dd
 34             yyyy.MM.dd
 35             yyyy/MM/dd
 36             yyyy_MM_dd
 37                 
 38             HH:mm:ss
 39                 
 40             yyyy-MM-dd HH:mm:ss

 45     * @return 返回一个日期时间的实例。
 46     * @example
 47         $.Date.parse('2013-04-29 09:31:20');

 49     parse: function (value) {
 50         if (value instanceof Date) {
 51             if (isNaN(value.getTime())) {
 52                 //throw new Error('参数是非法的日期实例');
 53                 return null;
 54             }
 55 
 56             return value;
 57         }
 58 
 59         if (typeof value != 'string') {
 60             //throw new Error('不支持该类型的参数：' + typeof value);
 61             return null;
 62         }
 63 
 64 
 65         //标准方式
 66         var date = new Date(value);
 67         if (!isNaN(date.getTime())) {
 68             return date;
 69         }
 70 
 71         /*
 72             自定义方式：
 73                 yyyy-MM-dd
 74                 yyyy.MM.dd
 75                 yyyy/MM/dd
 76                 yyyy_MM_dd
 77                 
 78                 HH:mm:ss
 79                 
 80                 yyyy-MM-dd HH:mm:ss
 81                 yyyy.MM.dd HH:mm:ss
 82                 yyyy/MM/dd HH:mm:ss
 83                 yyyy_MM_dd HH:mm:ss
 84                 
 85         */

 87         function GetDate(s) {
 88             var now = new Date();
 89 
 90             var separator =
 91                     s.indexOf('.') > 0 ? '.' :
 92                     s.indexOf('-') > 0 ? '-' :
 93                     s.indexOf('/') > 0 ? '/' :
 94                     s.indexOf('_') > 0 ? '_' : null;
 95 
 96             if (!separator) {
 97                 //throw new Error('无法识别的日期格式：' + s);
 98                 return null;
 99 
100             }
101 
102             var ps = s.split(separator);
103 
104             return {
105                 yyyy: ps[0],
106                 MM: ps[1] || 0,
107                 dd: ps[2] || 1
108             };
109         }

111         function GetTime(s) {
112             var separator = s.indexOf(':') > 0 ? ':' : null;
113             if (!separator) {
114                 //throw new Error('无法识别的时间格式：' + s);
115                 return null;
116 
117             }
118 
119             var ps = s.split(separator);
120 
121             return {
122                 HH: ps[0] || 0,
123                 mm: ps[1] || 0,
124                 ss: ps[2] || 0
125             };
126         }
127 
128 
129         var parts = value.split(' ');
130         if (!parts[0]) {
131             //throw new Error('无法识别的格式：' + value);
132             return null;
133 
134         }
135 
136         var date = parts[0].indexOf(':') > 0 ? null : parts[0];
137         var time = parts[0].indexOf(':') > 0 ? parts[0] : (parts[1] || null);
138 
139         if (date || time) {
140             if (date && time) {
141                 var d = GetDate(date);
142                 var t = GetTime(time);
143                 return new Date(d.yyyy, d.MM - 1, d.dd, t.HH, t.mm, t.ss);
144             }
145 
146             if (date) {
147                 var d = GetDate(date);
148                 return new Date(d.yyyy, d.MM - 1, d.dd);
149             }
150 
151             if (time) {
152                 var now = new Date();
153                 var t = GetTime(time);
154                 return new Date(now.getFullYear(), now.getMonth(), now.getDate(), t.HH, t.mm, t.ss);
155             }
156         }
157 
158         //throw new Error('无法识别的格式：' + value);
159         return null;

166     /**
167     * 把日期时间格式化指定格式的字符串。
168     * @param {Date} datetime 要进行格式化的日期时间。
169     * @param {string} formater 格式化的字符串。<br />
170         其中保留的占位符有：
171     <pre>
172         'yyyy': 4位数年份
173         'yy': 2位数年份
174         'MM': 2位数的月份(01-12)
175         'M': 1位数的月份(1-12)
176         'dddd': '星期日|一|二|三|四|五|六'
177         'dd': 2位数的日份(01-31)
178         'd': 1位数的日份(1-31)
179         'HH': 24小时制的2位数小时数(00-23)
180         'H': 24小时制的1位数小时数(0-23)
181         'hh': 12小时制的2位数小时数(00-12)
182         'h': 12小时制的1位数小时数(0-12)
183         'mm': 2位数的分钟数(00-59)
184         'm': 1位数的分钟数(0-59)
185         'ss': 2位数的秒钟数(00-59)
186         's': 1位数的秒数(0-59)
187         'tt': 上午：'AM'；下午: 'PM'
188         't': 上午：'A'；下午: 'P'
189         'TT': 上午： '上午'； 下午: '下午'
190         'T': 上午： '上'； 下午: '下'
191     </pre>
192     * @return {string} 返回一个格式化的字符串。
193     * @example
194         //返回当前时间的格式字符串，类似 '2013年4月29日 9:21:59 星期一'
195         $.Date.format(new Date(), 'yyyy年M月d日 h:m:s dddd')
196     */
197     format: function (datetime, formater) {
198 
199         var year = datetime.getFullYear();
200         var month = datetime.getMonth() + 1;
201         var date = datetime.getDate();
202         var hour = datetime.getHours();
203         var minute = datetime.getMinutes();
204         var second = datetime.getSeconds();
205 
206         var padLeft = function (value, length) {
207             return $.String.padLeft(value, length, '0');
208         };
209 
210 
211         var isAM = hour <= 12;
212 
213         //这里不要用 {} 来映射，因为 for in 的顺序不确定
214         var maps = [
215             ['yyyy', padLeft(year, 4)], 
216             ['yy', String(year).slice(2)],
217             ['MM', padLeft(month, 2)],
218             ['M', month],
219             ['dddd', '星期' + ('日一二三四五六'.charAt(datetime.getDay()))],
220             ['dd', padLeft(date, 2)],
221             ['d', date],
222             ['HH', padLeft(hour, 2)],
223             ['H', hour],
224             ['hh', padLeft(isAM ? hour : hour - 12, 2)],
225             ['h', isAM ? hour : hour - 12],
226             ['mm', padLeft(minute, 2)],
227             ['m', minute],
228             ['ss', padLeft(second, 2)],
229             ['s', second],
230             ['tt', isAM ? 'AM' : 'PM'],
231             ['t', isAM ? 'A' : 'P'],
232             ['TT', isAM ? '上午' : '下午'],
233             ['T', isAM ? '上' : '下']
234         ];
235 
236 
237         var s = formater;
238 
239         var replaceAll = $.String.replaceAll;
240         for (var i = 0, len = maps.length; i < len; i++) {
241 
242             var item = maps[i];
243             s = replaceAll(s, item[0], item[1]);
244         }
245 
246         return s;

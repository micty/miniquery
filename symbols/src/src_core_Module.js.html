  2 /**
  3 * 模块管理器类
  4 * @class
  5 */
  6 var Module = (function () {
  7 
  8     var guidKey = '__guid__';
  9     var guid$meta = {};
 10 
 11     /**
 12     * 构造器。
 13     * @inner
 14     */
 15     function Module(config) {
 16 
 17         var guid = Math.random().toString().slice(2);
 18         this[guidKey] = guid;
 19 
 20         config = config || {
 21             seperator: '/',
 22             shortcut: true,
 23         };
 24 
 25         
 26 
 27         var meta = {
 28             'id$module': {},
 29 
 30             'seperator': config.seperator,
 31             'shortcut': config.shortcut,
 32             'crossover': config.crossover,
 33         };
 34 
 35         guid$meta[guid] = meta;
 36 
 37     }
 38 
 39 
 40     //实例方法
 41     
 42     Module.prototype = /**@lends Module#*/ {
 43         constructor: Module,
 44 
 45         /**
 46         * 定义指定名称的模块。
 47         * @param {string} id 模块的名称。
 48         * @param {Object|function} factory 模块的导出函数或对象。
 49         */
 50         define: function define(id, factory) {
 51 
 52             var guid = this[guidKey];
 53             var meta = guid$meta[guid];
 54 
 55             var id$module = meta.id$module;
 56 
 57             id$module[id] = {
 58                 factory: factory,
 59                 exports: null,      //这个值在 require 后可能会给改写
 60                 required: false,    //指示是否已经 require 过
 61                 exposed: false,     //默认对外不可见
 62             };
 63 
 64         },
 65 
 66 
 67         /**
 68         * 加载指定的模块。
 69         * @param {string} id 模块的名称。
 70         * @return 返回指定的模块。
 71         */
 72         require: function (id) {
 73 
 74             var guid = this[guidKey];
 75             var meta = guid$meta[guid];
 76             var id$module = meta.id$module;
 77 
 78             var crossover = meta.crossover;
 79             var seperator = meta.seperator;
 80 
 81             //如 'List/API' 或 '/List/API'
 82             if (!crossover && id.lastIndexOf(seperator) > 0) { 
 83                 throw new Error('配置已经设定了不允许跨级加载模块。');
 84             }
 85 
 86             //指定了允许使用短名称，并且以分隔符开头，如 '/' 开头，如　'/API'
 87             if (meta.shortcut && id.indexOf(seperator) == 0) { 
 88                 var parentId = this.findId(arguments.callee.caller); //如 'List'
 89                 if (!parentId) {
 90                     throw new Error('require 时如果指定了以 "' + seperator + '" 开头的短名称，则必须用在 define 的函数体内');
 91                 }
 92 
 93                 id = parentId + id; //完整名称，如 'List/API'
 94             }
 95 
 96 
 97 
 98             var module = id$module[id];
 99             if (!module) { //不存在该模块
100                 return;
101             }
102 
103             if (module.required) { //已经 require 过了
104                 return module.exports;
105             }
106 
107 
108             //首次 require
109 
110             module.required = true; //更改标志，指示已经 require 过一次
111 
112             var factory = module.factory;
113 
114             if (typeof factory != 'function') { //非工厂函数，则直接导出
115                 module.exports = factory;
116                 return factory;
117             }
118 
119             //factory 是个工厂函数
120             var require = arguments.callee.bind(this); //引用自身，并且作为静态方法调用
121             var exports = {};
122             var mod = {
123                 id: id,
124                 exports: exports,
125             };
126 
127             exports = factory(require, mod, exports);
128             if (exports === undefined) {    //没有通过 return 来返回值，
129                 exports = mod.exports;      //则要导出的值只能在 mod.exports 里
130             }
131 
132             module.exports = exports;
133             return exports;
134 
135         },
136 
137         /**
138         * 根据工厂函数反向查找对应的模块 id。
139         */
140         findId: function (factory) {
141 
142             var guid = this[guidKey];
143             var meta = guid$meta[guid];
144 
145             var id$module = meta.id$module;
146            
147             for (var id in id$module) {
148                 var module = id$module[id];
149                 if (module.factory === factory) {
150                     return id;
151                 }
152             }
153         },
154 
155         
156         /**
157         * 设置或获取对外暴露的模块。
158         * 通过此方法，可以控制指定的模块是否可以通过 MiniQuery.require(id) 来加载到。
159         * @param {string|Object} id 模块的名称。
160             当指定为一个 {} 时，则表示批量设置。
161             当指定为一个字符串时，则单个设置。
162         * @param {boolean} [exposed] 模块是否对外暴露。
163             当参数 id 为字符串时，且不指定该参数时，表示获取操作，
164             即获取指定 id 的模块是否对外暴露。
165         * @return {boolean}
166         */
167         expose: function (id, exposed) {
168 
169             var guid = this[guidKey];
170             var meta = guid$meta[guid];
171             var id$module = meta.id$module;
172 
173             //内部方法: get 操作
174             function get(id) {
175                 var module = id$module[id];
176                 if (!module) {
177                     return;
178                 }
179 
180                 return module.exposed;
181             }
182 
183             //内部方法: set 操作
184             function set(id, exposed) {
185                 var module = id$module[id];
186                 if (module) {
187                     module.exposed = !!exposed;
188                 }
189             }
190 
191             //set 操作
192             if (typeof id == 'object') { // expose({ })，批量 set
193                 var id$exposed = id;
194                 for (var id in id$exposed) {
195                     var exposed = id$exposed[id];
196                     set(id, exposed);
197                 }
198                 return;
199             }
200 
201             if (arguments.length == 2) { // expose('', true|false)，单个 set
202                 set(id, exposed);
203                 return;
204             }
205 
206             //get 操作
207             return get(id);
208         },
209 
210         /**
211         * 销毁本实例。
212         */
213         destroy: function () {
214             var guid = this[guidKey];
215             delete this[guidKey];
216             delete guid$meta[guid];
217         },
218     };
219 
220 
221 
222     return Module;
223 
224 
225 })();

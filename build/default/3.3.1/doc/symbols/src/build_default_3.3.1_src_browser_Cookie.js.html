  2 /**
  3 * Cookie 工具
  4 * @namespace
  5 * @name Cookie
  6 */
  7 define('Cookie', function (require, module, exports) {
  8 
  9 
 10     //缓存 toObject 中的结果
 11     var cookie$object = {
 12         'true': {},
 13         'false': {}
 14     };
 15 
 16 
 17     module.exports = exports = /**@lends Cookie*/ {
 18 
 19         /**
 20         * 把一个 cookie 字符串解析成等价的 Object 对象。
 21         * @param {String} cookie 要进行解析的 cookie 字符串。
 22         * @param {boolean} [deep=false] 指定是否要进行深层次解析。
 23             如果要对 cookie 中的值进行查询字符串方式的深层次解析，请指定 true；
 24             否则请指定 false 或不指定。
 25         * @return {Object} 返回一个解析后的等价的 Object 对象。
 26         * @example 
 27             var obj = Cookie.toObject('A=1; B=2; C=a=100&b=200', true); //深层次解析
 28             //得到
 29             obj = {A: 1, B: 2, C: {a: 100, b: 200}};
 30     
 31             var obj = Cookie.toObject('A=1; B=2; C=a=100&b=200'); //浅解析
 32             //得到
 33             obj = {A: 1, B: 2, C: 'a=100&b=200'};
 34 
 35             Cookie.toObject('a=1; b=2');
 36             Cookie.toObject('a=1; b=2', true);
 37             Cookie.toObject();
 38             Cookie.toObject(true);
 39     
 40         */
 41         toObject: function (cookie, deep) {
 42 
 43             if (typeof cookie != 'string') { //此时为 toObject(true|false) 或 toObject()
 44                 deep = cookie;
 45                 cookie = document.cookie;
 46             }
 47 
 48             if (!cookie) {
 49                 return {};
 50             }
 51 
 52             deep = !!deep; //转成 true|false，因为有以它为键的存储
 53 
 54             var obj = cookie$object[deep][cookie];
 55             if (obj) { //缓存中找到
 56                 return obj;
 57             }
 58 
 59 
 60             obj = {};
 61 
 62             var $Object = require('Object');
 63             var $Array = require('Array');
 64 
 65             var parseQueryString = $Object.parseQueryString;
 66 
 67             $Array.each(cookie.split('; '), function (item, index) {
 68 
 69                 var pos = item.indexOf('=');
 70 
 71                 var name = item.slice(0, pos);
 72                 name = decodeURIComponent(name);
 73 
 74 
 75                 var value = item.slice(pos + 1);
 76 
 77                 if (deep && value.indexOf('=') > -1) { //指定了深层次解析，并且还包含 '=' 号
 78                     value = parseQueryString(value); //深层次解析复合值
 79                 }
 80                 else {
 81                     value = decodeURIComponent(value);
 82                 }
 83 
 84 
 85                 if (name in obj) {
 86                     var a = obj[name];
 87                     if (a instanceof Array) {
 88                         a.push(value);
 89                     }
 90                     else {
 91                         obj[name] = [a, value]
 92                     }
 93                 }
 94                 else {
 95                     obj[name] = value;
 96                 }
 97 
 98             });
 99 
100             cookie$object[deep][cookie] = obj; //缓存起来
101 
102             return obj;
103         },
104 
105 
106         /**
107         * 从当前 document.cookie 中获取指定名称和键所对应的值。
108         * @param {boolean} [name] 要获取的项的名称。
109             当不指定该参数时，全量返回 document.cookie 字符串。
110         * @param {String} [key] 要获取的项的键。
111             当不指定该参数时，返回参数 name 对应的项。
112         * @return 返回指定项的值。
113         * @example 
114             Cookie.get();
115             Cookie.get(true);
116             Cookie.get('A');
117             Cookie.get('A', true);
118             Cookie.get('A', 'b');
119         */
120         get: function (name, key) {
121 
122             if (name === undefined) { //此时为 get()
123                 return exports.toObject(); //返回一个浅解析的全量 Object
124             }
125 
126             if (name === true) { //此时为 get(true)
127                 return exports.toObject(true); //返回一个深解析的全量 Object
128             }
129 
130 
131             //下面的 name 为一个具体的值
132 
133             if (key === undefined) { //此时为 get(name)
134                 var obj = exports.toObject(); //浅解析
135                 return obj[name];
136             }
137 
138 
139             //下面的指定了 key
140 
141             var obj = exports.toObject(true); //深解析
142             var value = obj[name];
143 
144             if (key === true) {
145                 return value;
146             }
147 
148             //下面的 key 为一个具体的值
149 
150             if (value instanceof Array) { //同一个 name 得到多个值，主要是由于 path 不同导致的
151 
152                 var $Array = require('Array');
153 
154                 //过滤出含有 key 的 object 项
155                 var items = $Array.grep(value, function (item, index) {
156                     return item &&
157                         typeof item == 'object' &&
158                         key in item;
159                 });
160 
161                 if (items.length == 0) {
162                     return;
163                 }
164 
165                 if (items.length == 1) { //只有一个
166                     return items[0][key];
167                 }
168 
169                 return $Array.keep(items, function (item, index) {
170                     return item[key];
171                 });
172             }
173 
174 
175             if (value && typeof value == 'object') {
176                 return value[key];
177             }
178 
179         },
180 
181 
182         /**
183         * 给当前的 document.cookie 设置一个 Cookie。
184         * @param {Document} document 要进行操作的 Document 对象。
185         * @param {String} name 要设置的 Cookie 名称。
186         * @param {String|Object} value 要设置的 Cookie 值。
187             当传入一个 Object 对象时，会对它进行查询字符串的编码以获取一个 String 类型值。
188         * @param {String|Number|Date} [expires] 过期时间。
189             参数 expires 接受以下格式的字符串：
190                 y: 年
191                 M: 月
192                 w: 周
193                 d: 天
194                 h: 小时
195                 m: 分钟
196                 s: 秒
197                 ms: 毫秒
198             或传入一个 $.Date.parse 识别的格式字符串，并会被解析成一个实际的日期实例。
199         * @example 
200             //设置一个 A=100 的 Cookie，过期时间为12天后
201             Cookie.set('A', 100, '12d'); 
202     
203             //设置一个 B=200 的 Cookie，过期时间为2周后
204             Cookie.set('B', 200, '2w'); 
205     
206             //设置一个 C=300 的 Cookie，过期时间为 '2014-9-10'
207             Cookie.set('C', 300, '2014-9-10'); 
208     
209             Cookie.set({
210                 name: 'A',
211                 value: 100,
212                 expires: '2w',
213                 path: '/',
214                 domain: 'localhost',
215                 secure: true
216             });
217         */
218         set: function (name, value, expires, path, domain, secure) {
219 
220             var $Object = require('Object');
221 
222 
223             if ($Object.isPlain(name)) { // 此时为 set({ ... });
224                 var config = name;
225                 name = config.name;
226                 value = config.value;
227                 expires = config.expires;
228                 path = config.path;
229                 domain = config.domain;
230                 secure = config.secure;
231             }
232 
233             name = encodeURIComponent(name);
234             if (!name) { // 空字符串
235                 throw new Error('参数 name 不能为空字符串');
236             }
237 
238 
239             if ($Object.isPlain(value)) {
240                 value = $Object.toQueryString(value);
241             }
242             else {
243                 value = encodeURIComponent(value);
244             }
245 
246             var cookie = name + '=' + value + '; ';
247 
248             if (expires) {
249                 var Expires = require(module, 'Expires');
250                 expires = Expires.parse(expires);
251                 cookie += 'expires=' + expires.toGMTString() + '; '; //不推荐使用 toGMTString 方法
252             }
253 
254             if (path) {
255                 cookie += 'path=' + path + '; ';
256             }
257 
258             if (domain) {
259                 cookie += 'domain=' + domain + '; ';
260             }
261 
262             if (secure) {
263                 cookie += 'secure';
264             }
265 
266             document.cookie = cookie;
267         },
268 
269 
270         /**
271         * 从当前的 document.cookie 中移除指定名称的 Cookie。
272         * @param {String} [name] 要移除的 Cookie 的名称。
273             当不指定参数 name 时，则会把所有的 Cookie 都移除。
274         * @example 
275             //给 document 名称为 A 的 Cookie 移除
276             Cookie.remove(document, 'A'); 
277     
278             //把 document 的所有 Cookie 都移除
279             Cookie.remove(document); 
280         */
281         remove: function (name, path, domain, secure) {
282 
283             var $Object = require('Object');
284 
285             var config = $Object.isPlain(name) ? name : {
286                 name: name,
287                 path: path,
288                 domain: domain,
289                 secure: secure
290             };
291 
292             config.expires = new Date(0);
293 
294             name = config.name;
295 
296             if (name === undefined) { //未指定名称，则全部移除
297                 var obj = exports.toObject();
298                 $Object.each(obj, function (name, value) {
299                     config.name = name;
300                     exports.set(config);
301                 });
302 
303                 return;
304             }
305 
306             exports.set(config);
307 
308         }
309 
310 
311     };
312 
313 });

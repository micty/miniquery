 10     var defaults = {
 11         url: '',
 12         id: '',
 13         charset: 'utf-8',
 14         document: window.document,
 15         onload: null
 16     };

 19     /**
 20     * 加载单个
 21     * @inner
 22     */
 23     function loadSingle(url, charset, document, onload) {
 24 
 25         var id;
 26 
 27         if (typeof url == 'object') { //传入的是一个 {} 
 28             var config = url;
 29 
 30             id = config.id;
 31             url = config.url;
 32             charset = config.charset;
 33             document = config.document;
 34             onload = config.onload;
 35         }
 36 
 37 
 38         var script = document.createElement('script');
 39         script.type = 'text/javascript';
 40 
 41         if (id) {
 42             script.id = id;
 43         }
 44 
 45         if (charset) {
 46             script.charset = charset;
 47         }
 48 
 49 
 50         if (onload) { //指定了回调函数，则设置它
 51 
 52             if (script.readyState) { //IE
 53 
 54                 /**@ignore*/
 55                 script.onreadystatechange = function () {
 56 
 57                     var readyState = script.readyState;
 58 
 59                     if (readyState == 'loaded' || readyState == 'complete') {
 60                         script.onreadystatechange = null; //避免重复执行回调
 61                         onload();
 62                     }
 63                 };
 64             }
 65             else { //标准
 66                 script.onload = onload;
 67             }
 68 
 69         }
 70 
 71 
 72         script.src = url;
 73         document.head.appendChild(script);
 74     }
 75 
 76     /**
 77     * 顺序加载批量
 78     * @inner
 79     */
 80     function loadBatch(urls, charset, document, fn) {
 81 
 82         var index = 0;
 83 
 84         (function () {
 85 
 86             var next = arguments.callee;
 87             var url = urls[index];
 88 
 89             loadSingle(url, charset, document, function () {
 90                 index++;
 91 
 92                 if (index < urls.length) {
 93                     next();
 94                 }
 95                 else {
 96                     fn && fn();
 97                 }
 98             });
 99 
100         })();
101 
102 
103     }

106     /**
107     * 单个写入
108     * @inner
109     */
110     function document_write(url, charset, document) {
111 
112         var $String = require('String');
113 
114         var html = $String.format('<script type="text/javascript" src="{src}" {charset} ><\/script>', {
115             'src': url,
116             'charset': charset ? $String.format('charset="{0}"', charset) : ''
117         });
118 
119         document.write(html);
120     }

123     module.exports = exports = /**@lends Script*/ {
124 
125         /**
126         * 跨浏览器动态加载 JS 文件，并在加载完成后执行指定的回调函数。
127         * @param {string|Array} params.url 
128             要加载的 JS 文件的 url 地址，如果要批量加载，则为一个地址数组。
129         * @param {string} [params.charset="utf-8"] 
130             要加载的 JS 文件的字符编码，默认为 utf-8。
131         * @param {Document} [params.document=window.document] 
132             要加载的 JS 文件的上下文环境的 document，默认为当前窗口的 document 对象。
133         * @param {function} [params.onload] 
134             加载成功后的回调函数。
135         * @example
136             Script.load({
137                 url: 'a.js',
138                 charset: 'utf-8',
139                 document: document,
140                 id: 'myScript',
141                 onload: function (){ }
142             });
143 
144             Script.load('a.js', 'utf-8', document, function(){});
145             Script.load('a.js', 'utf-8', function(){});
146             Script.load('a.js', document, function(){});
147             Script.load('a.js', function(){});
148 
149             //批量加载
150             Script.load(['a.js', 'b.js'], function(){});
151         */
152         load: function (params) {
153 
154             var $Object = require('Object');
155 
156             var obj = $Object.extend({}, defaults); //复制一份
157 
158             //注意，params 有可能是个数组，不能用 typeof 为 'object'
159             if ($Object.isPlain(params)) { //纯对象 {}
160                 $Object.extend(obj, params);
161             }
162             else {
163                 obj.url = params;
164 
165                 switch (typeof arguments[1]) {
166                     case 'string':
167                         obj.charset = arguments[1];
168                         break;
169                     case 'object':
170                         obj.document = arguments[1];
171                         break;
172                     case 'function':
173                         obj.onload = arguments[1];
174                         break;
175                 }
176 
177                 switch (typeof arguments[2]) {
178                     case 'object':
179                         obj.document = arguments[2];
180                         break;
181                     case 'function':
182                         obj.onload = arguments[2];
183                         break;
184                 }
185 
186                 if (arguments[3]) {
187                     obj.onload = arguments[3];
188                 }
189             }
190 
191             var url = obj.url;
192 
193             if (typeof url == 'string') {
194                 loadSingle(obj);
195             }
196             else if (url instanceof Array) {
197                 loadBatch(url, obj.charset, obj.document, obj.onload);
198             }
199             else {
200                 throw new Error('参数 params.url 必须为 string 或 string 的数组');
201             }
202 
203 
204         },
205 
206         /**
207         * 创建一个 script 标签，并插入 JavaScript 代码。
208         * @param {string} params.code 要插入的 JS 代码。
209         * @param {string} [params.id] 创建的 script 标签中的 id。
210         * @param {Document} [params.document=window.document] 
211             创建的 script 标签的上下文环境的 document。默认为当前窗口的 document 对象。
212         * @example
213             Script.insert({
214                 code: 'alert(0);',
215                 id: 'myScript',
216                 document: document
217             });
218             Script.insert('alert(0);', 'myScript', document);
219             Script.insert('alert(0);', 'myScript');
220             Script.insert('alert(0);', document);
221         */
222         insert: function (params) {
223             var obj = {
224                 code: '',
225                 id: '',
226                 document: window.document
227             };
228 
229             var $Object = require('Object');
230 
231 
232             if ($Object.isPlain(params)) {
233                 $Object.extend(obj, params);
234             }
235             else {
236                 obj.code = params;
237 
238                 switch (typeof arguments[1]) {
239                     case 'string':
240                         obj.id = arguments[1];
241                         break;
242                     case 'object':
243                         obj.document = arguments[1];
244                         break;
245                 }
246 
247                 if (arguments[2]) {
248                     obj.document = arguments[2];
249                 }
250             }
251 
252             var script = obj.document.createElement('script');
253             script.type = 'text/javascript';
254 
255             if (obj.id) {
256                 script.id = obj.id;
257             }
258 
259             try { // 标准，IE 除外
260                 script.appendChild(obj.document.createTextNode(obj.code));
261             }
262             catch (ex) { // IE，但不限于 IE
263                 script.text = obj.code;
264             }
265 
266             obj.document.getElementsByTagName('head')[0].appendChild(script);
267         },
268 
269         /**
270         * 用 document.write 的方式加载 JS 文件。
271         * @param {string|Array} params.url 
272             要加载的 JS 文件的 url 地址，如果要批量加载，则为一个地址数组。
273         * @param {string} [params.charset="utf-8"] 
274             要加载的 JS 文件的字符编码，默认为 utf-8。
275         * @param {Document} [params.document=window.document] 
276             要加载的 JS 文件的上下文环境的 document，默认为当前窗口的 document 对象。
277         * @example
278             Script.write({
279                 url: 'a.js',
280                 charset: 'utf-8',
281                 document: document
282             });
283             Script.write('a.js', 'utf-8', document);
284             Script.write('a.js', 'utf-8');
285             Script.write('a.js', document);
286         */
287         write: function (params) {
288             var obj = {
289                 url: '',
290                 charset: '',
291                 document: window.document
292             };
293 
294             var $Object = require('Object');
295 
296             if ($Object.isPlain(params)) {
297                 $Object.extend(obj, params);
298             }
299             else {
300                 obj.url = params;
301                 switch (typeof arguments[1]) {
302                     case 'string':
303                         obj.charset = arguments[1];
304                         break;
305                     case 'object':
306                         obj.document = arguments[1];
307                         break;
308                 }
309 
310                 if (arguments[2]) {
311                     obj.document = arguments[2];
312                 }
313 
314             }
315 
316             if ($Object.isArray(obj.url)) {
317                 var urls = obj.url;
318                 for (var i = 0, len = urls.length; i < len; i++) {
319                     document_write(urls[i], obj.charset, obj.document);
320                 }
321             }
322             else {
323                 document_write(obj.url, obj.charset, obj.document);
324             }
325         }
326     };

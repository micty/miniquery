 36         * 以安全的方式给 MiniQuery 使用一个新的命名空间。

 38         * 当 global 中未存在指定的命名空间或参数中指定了要全量覆盖时，则使用全量覆盖的方式，

 41         * 该方式是最安全的方式，也是默认和推荐的方式。
 42         */
 43         use: function (namespace, overwrite) {
 44 
 45             if (!(namespace in global) || overwrite) { //未存在或明确指定了覆盖
 46                 global[namespace] = exports; //全量覆盖
 47                 return;
 48             }
 49 
 50             //已经存在，则拷贝不冲突的成员部分
 51             var obj = global[namespace];
 52 
 53             for (var key in exports) {
 54 
 55                 if (!(key in obj)) { //只拷贝不冲突的部分

 57                 }
 58             }

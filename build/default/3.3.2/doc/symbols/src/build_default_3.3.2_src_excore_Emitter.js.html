  9     var $ = require('$');
 10     var $Array = require('Array');
 11     var $Object = require('Object');
 12     var $String = require('String');
 13     var Mapper = require('Mapper');
 14 
 15     var Tree = require(module, 'Tree'); //完整名称为 Emitter/Tree
 16 
 17     var mapper = new Mapper();

 21     //绑定事件。
 22     //实例的私有方法，必须用 bind.apply(this, []) 的方式来调用。
 23     function bind(isOneOff, name, fn) {
 24 
 25         var meta = mapper.get(this);
 26         var all = meta.all;
 27 
 28         // 单名称情况 on(name, fn)，专门成一个分支，为了优化
 29         if (typeof name == 'string' && typeof fn == 'function') {
 30             Tree.add(all, [name], {
 31                 'fn': fn,
 32                 'isOneOff': isOneOff,
 33             });
 34             return;
 35         }
 36 
 37         //多名称情况
 38         var args = $.toArray(arguments).slice(1);
 39 
 40         //重载 bind(isOneOff, name0, name1, ..., nameN, {...}) 的情况
 41         //先尝试找到 {} 所在的位置
 42         var index = $Array.findIndex(args, function (item, index) {
 43             return typeof item == 'object';
 44         });
 45 
 46         if (index >= 0) {
 47             var obj = args[index];
 48             var names = args.slice(0, index);
 49             var list = $Object.linearize(obj);
 50 
 51             $Array.each(list, function (item, index) {
 52 
 53                 var keys = names.concat(item.keys);
 54 
 55                 Tree.add(all, keys, {
 56                     'fn': item.value,
 57                     'isOneOff': isOneOff,
 58                 });
 59             });
 60             return;
 61         }
 62 
 63 
 64         //重载 bind(isOneOff, name0, name1, ..., nameN, fn) 的情况
 65         //尝试找到回调函数 fn 所在的位置
 66         var index = $Array.findIndex(args, function (item, index) {
 67             return typeof item == 'function';
 68         });
 69 
 70         if (index < 0) {
 71             throw new Error('参数中必须指定一个回调函数');
 72         }
 73 
 74         fn = args[index]; //回调函数
 75         var names = args.slice(0, index); //前面的都当作是名称
 76 
 77         Tree.add(all, names, {
 78             'fn': fn,
 79             'isOneOff': isOneOff,
 80         });
 81     }
 82

 84     //触发事件。
 85     //实例的私有方法，必须用 fire.apply(this, []) 的方式来调用。
 86     function fire(config) {
 87 
 88         var names = config.names;
 89         if (!names || names.length == 0) {
 90             throw new Error('必须至少指定一个事件名称。');
 91         }
 92 
 93         var meta = mapper.get(this);
 94         var all = meta.all;
 95 
 96         var list = Tree.getList(all, names);
 97         if (!list || list.length == 0) {
 98             return [];
 99         }
100 
101 
102         function stop(list) {
103 
104             if (!('stop' in config)) {
105                 stop = function () {
106                     return false;
107                 };
108                 return false;
109             }
110 
111             var fn = config.stop;
112             if (typeof fn == 'function') {
113                 stop = fn;
114                 return fn(list) === true;
115             }
116 
117             stop = function (list) { 
118                 return list.slice(-1)[0] === fn; //取最后一个值进行判断
119             };
120 
121             return list.slice(-1)[0] === fn;
122         }
123 
124 
125         //这里要特别注意，在执行回调的过程中，回调函数里有可能会去修改回调列表，
126         //而此处又要去移除那些一次性的回调（即只执行一次的），
127         //为了避免破坏回调函数里的修改结果，这里要边移除边执行回调，而且每次都要
128         //以原来的回调列表为准去查询要移除的项的确切位置。
129 
130         var items = list.slice(0); //复制一份，因为回调列表可能会在执行回调过程发生变化。
131         var returns = [];
132         var len = items.length;
133         var context = meta.context || null;
134         var args = config.args || [];
135 
136         for (var i = 0; i < len; i++) {
137             var item = items[i];
138 
139             if (item.isOneOff) { //只需执行一次的
140                 var index = $Array.indexOf(list, item); //找到该项在回调列表中的索引位置
141                 if (index >= 0) {
142                     list.splice(index, 1); //直接从原数组删除
143                 }
144             }
145 
146             var value = item.fn.apply(context, args); //让 fn 内的 this 指向 obj
147             returns.push(value);
148 
149             //返回值符合设定的停止值，则停止后续的调用
150             if (stop(returns) === true) {
151                 break;
152             }
153         }
154 
155         return returns;
156 
157     }

163     /**
164     * 构造器。
165     * @param {Object} [context=null] 事件处理函数中的 this 上下文对象。
166     *   如果不指定，则默认为 null。
167     */
168     function Emitter(context) {
169 
170         var id = 'Emitter-' + $String.random();
171         Mapper.setGuid(this, id);
172 
173         var meta = {
174             'context': context,
175             'all': {},
176         };
177 
178         mapper.set(this, meta);
179 
180     }
181 
182     //实例方法
183     Emitter.prototype = /**@lends Emitter.prototype */ {
184         constructor: Emitter,
185 
186         /**
187         * 绑定指定名称的事件处理函数。
188         * @param {string} name 要绑定的事件名称。
189         * @param {function} fn 事件处理函数。 
190             在处理函数内部， this 指向构造器参数 context 对象。
191         * @example
192             var emitter = new Emitter();
193             emitter.on('click', function () {
194 
195             });
196         */
197         on: function (name, fn) {
198             var args = $.toArray(arguments);
199             args = [false].concat(args);
200             bind.apply(this, args);
201         },
202 
203         /**
204         * 绑定指定名称的一次性事件处理函数。
205         * @param {string} name 要绑定的事件名称。
206         * @param {function} fn 事件处理函数。 
207             在处理函数内部， this 指向构造器参数 context 对象。
208         */
209         one: function (name, fn) {
210             var args = $.toArray(arguments);
211             args = [true].concat(args);
212             bind.apply(this, args);
213         },
214 
215 
216         /**
217         * 解除绑定指定名称的事件处理函数。
218         * @param {string} [name] 要解除绑定的事件名称。
219             如果不指定该参数，则移除所有的事件。
220             如果指定了该参数，其类型必须为 string，否则会抛出异常。
221         * @param {function} [fn] 要解除绑定事件处理函数。
222             如果不指定，则移除 name 所关联的所有事件。
223         */
224         off: function (name, fn) {
225 
226             var meta = mapper.get(this);
227             var all = meta.all;
228 
229             //未指定事件名，则移除所有的事件
230             if (name === undefined) {
231                 meta.all = {};
232                 return;
233             }
234 
235             //多名称情况: fire(name0, name1, name2, ..., nameN, fn)
236             var args = $.toArray(arguments);
237 
238             //找到回调函数所在的位置
239             var index = $Array.findIndex(args, function (item, index) {
240                 return typeof item == 'function';
241             });
242 
243             if (index < 0) {
244                 index = args.length;
245             }
246 
247             fn = args[index];
248 
249             var names = args.slice(0, index);
250             var node = Tree.getNode(all, names);
251             if (!node) { //尚未存在该名称所对应的节点
252                 return;
253             }
254 
255             var list = node.list;
256             if (list.length == 0) {
257                 return;
258             }
259 
260             if (fn) {
261                 node.list = $Array.grep(list, function (item, index) {
262                     return item.fn !== fn;
263                 });
264             }
265             else { //未指定处理函数，则清空列表
266                 list.length = 0;
267             }
268 
269         },
270 
271         /**
272         * 已重载。
273         * 触发指定名称的事件，并可向事件处理函数传递一些参数。
274         * 如果指定了 stop 字段，则当事件处理函数返回指定的值时将停止调用后面的处理函数。
275         * @param {Object} config 配置对象。
276         * @param {Array} config.names 事件名称列表。
277         * @param {Array} config.args 要传递给事件处理函数的参数数据。
278         * @param config.stop 当事件处理函数的返回值满足一定时，将停止调用后面的处理函数。
279             当 stop 为函数时，则需要在 stop 函数内明确返回 true 才停止。
280             否则，事件处理的返回值跟 stop 完全相等时才停步。
281         * @return {Array} 返回所有事件处理函数的返回值所组成的一个数组。
282         * @example
283             var emitter = new Emitter();
284             emitter.on('click', 'name', function (a, b) {
285                 console.log(a, b);
286             });
287             emitter.fire('click', 'name', [100, 200]);
288 
289             emitter.fire({
290                 names: ['click', 'name'],
291                 args: [100, 200],
292                 stop: 100,
293             });
294         */
295         fire: function (config) {
296 
297             if (typeof config == 'object') { //重载 fire({...})
298                 return fire.apply(this, [config]);
299             }
300 
301             //多名称情况: fire(name0, name1, name2, ..., nameN, params)
302             var args = $.toArray(arguments);
303 
304             //找到回调函数所在的位置
305             var index = $Array.findIndex(args, function (item, index) {
306                 return item instanceof Array;
307             });
308 
309             if (index < 0) {
310                 index = args.length;
311             }
312            
313             return fire.apply(this, [{
314                 'names': args.slice(0, index),
315                 'args': args[index]
316             }]);
317 
318         },
319 
320         /**
321         * 检测是否包含指定名称的事件。
322         * @param {string} [name] 要检测的事件名称。
323             当不指定时，则判断是否包含了任意类型的事件。
324         * @param {function} [fn] 要检测的事件处理函数。
325         * @return {boolean} 返回一个布尔值，该布尔值指示目标对象上是否包含指否类型的事件以及指定的处理函数。
326             如果是则返回 true；否则返回 false。
327         */
328         has: function (name, fn) {
329 
330             var meta = mapper.get(this);
331             var all = meta.all;
332 
333             if (arguments.length == 0) { //未指定名称，则判断是否包含了任意类型的事件
334                 return !$Object.isEmpty(all);
335             }
336 
337             //多名称情况: fire(name0, name1, name2, ..., nameN, fn)
338             var args = $.toArray(arguments);
339 
340             //找到回调函数所在的位置
341             var index = $Array.findIndex(args, function (item, index) {
342                 return typeof item == 'function';
343             });
344 
345             if (index < 0) {
346                 index = args.length;
347             }
348 
349             fn = args[index];
350 
351             var names = args.slice(0, index);
352             var list = Tree.getList(all, names);
353 
354             if (!list || list.length == 0) { //尚未存在该名称所对应的节点
355                 return false;
356             }
357 
358             if (!fn) {
359                 return true;
360             }
361 
362             //指定了回调函数
363             return $Array.find(list, function (item, index) {
364                 return item.fn === fn;
365             });
366 
367 
368         },
369 
370         /**
371         * 销毁本实例对象。
372         */
373         destroy: function () {
374             this.off();
375             mapper.remove(this);
376         },
377 
378 
379     };
380 
381     
382     module.exports = Emitter;

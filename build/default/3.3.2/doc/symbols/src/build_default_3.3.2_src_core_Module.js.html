  2 /**
  3 * 模块管理器类
  4 * @class
  5 */
  6 var Module = (function () {
  7 
  8     var guidKey = '__guid__';
  9     var guid$meta = {};
 10 
 11 
 12     /**
 13     * 构造器。
 14     * @inner
 15     */
 16     function Module(config) {
 17 
 18         var guid = Math.random().toString().slice(2);
 19         this[guidKey] = guid;
 20 
 21         config = config || {
 22             seperator: '/',
 23             shortcut: true,
 24         };
 25 
 26         var meta = {
 27             'id$module': {},
 28             'seperator': config.seperator,
 29             'shortcut': config.shortcut,
 30             'crossover': config.crossover,
 31         };
 32 
 33         guid$meta[guid] = meta;
 34 
 35     }
 36 
 37     //实例方法
 38     Module.prototype = /**@lends Module#*/ {
 39         constructor: Module,
 40 
 41         /**
 42         * 定义指定名称的模块。
 43         * @param {string} id 模块的名称。
 44         * @param {Object|function} factory 模块的导出函数或对象。
 45         */
 46         define: function define(id, factory) {
 47 
 48             var guid = this[guidKey];
 49             var meta = guid$meta[guid];
 50 
 51             var id$module = meta.id$module;
 52 
 53 
 54             id$module[id] = {
 55                 factory: factory,
 56                 exports: null,      //这个值在 require 后可能会给改写
 57                 required: false,    //指示是否已经 require 过
 58                 exposed: false,     //默认对外不可见
 59                 module: null,       //用于检测在 define 中加载下级模块，即 require(module, id) 时用到
 60             };
 61 
 62 
 63         },
 64 
 65         /**
 66         * 加载指定的模块。
 67         * 已重载 require(moudle, id)，用于加载 module 的直接下级子模块。
 68         * @param {string} id 模块的名称。
 69         * @return 返回指定的模块。
 70         */
 71         require: function (id) {
 72 
 73             var guid = this[guidKey];
 74             var meta = guid$meta[guid];
 75             var id$module = meta.id$module;
 76 
 77             var module = null;
 78 
 79             if (typeof id == 'object') { // 重载 require(module, id)
 80                 module = id; //此时 module 相对于 id 为父模块
 81 
 82                 if (!module) {
 83                     throw new Error('当作为 require(module, id) 调用时，第一个参数 module 不能为空。');
 84                 }
 85 
 86                 var parentId = module.id;
 87                 var parentMod = id$module[parentId];
 88                 if (!parentMod) {
 89                     throw new Error('不存在 module.id 为 ' + parentId + ' 的父模块。');
 90                 }
 91 
 92                 //防止用户手动构造 module 对象以达到跨界调用的目的。
 93                 //传进来的 module 必须是 define 函数中的 factory 函数中的第二个参数 module 原始对象。
 94                 if (module !== parentMod.module) { 
 95                     throw new Error('当作为 require(module, id) 调用时，第一个参数 module 必须为 define 中的工厂函数第二个原始参数 module 对象。');
 96                 }
 97 
 98                 id = arguments[1];
 99             }
100 
101 
102             var crossover = meta.crossover;
103             var seperator = meta.seperator;
104 
105             //如 'List/API' 或 '/List/API'
106             if (!crossover && id.lastIndexOf(seperator) > 0) { 
107                 throw new Error('配置已经设定了不允许跨级加载模块。');
108             }
109 
110             if (module) {
111                 id = [parentId, id].join(seperator);
112             }
113 
114 
115             module = id$module[id];
116             if (!module) { //不存在该模块
117                 return;
118             }
119 
120             if (module.required) { //已经 require 过了
121                 return module.exports;
122             }
123 
124 
125             //首次 require
126 
127             module.required = true; //更改标志，指示已经 require 过一次
128 
129             var factory = module.factory;
130 
131             if (typeof factory != 'function') { //非工厂函数，则直接导出
132                 module.exports = factory;
133                 return factory;
134             }
135 
136             //factory 是个工厂函数
137             var require = arguments.callee.bind(this); //引用自身，并且作为静态方法调用
138             var exports = {};
139             var mod = {
140                 id: id,
141                 exports: exports,
142             };
143 
144             module.module = mod; //保存到元数据中
145 
146             exports = factory(require, mod, exports);
147             if (exports === undefined) {    //没有通过 return 来返回值，
148                 exports = mod.exports;      //则要导出的值只能在 mod.exports 里
149             }
150 
151             module.exports = exports;
152             return exports;
153 
154         },
155 
156         
157         /**
158         * 设置或获取对外暴露的模块。
159         * 通过此方法，可以控制指定的模块是否可以通过 MiniQuery.require(id) 来加载到。
160         * @param {string|Object} id 模块的名称。
161             当指定为一个 {} 时，则表示批量设置。
162             当指定为一个字符串时，则单个设置。
163         * @param {boolean} [exposed] 模块是否对外暴露。
164             当参数 id 为字符串时，且不指定该参数时，表示获取操作，
165             即获取指定 id 的模块是否对外暴露。
166         * @return {boolean}
167         */
168         expose: function (id, exposed) {
169 
170             var guid = this[guidKey];
171             var meta = guid$meta[guid];
172             var id$module = meta.id$module;
173 
174             //内部方法: get 操作
175             function get(id) {
176                 var module = id$module[id];
177                 if (!module) {
178                     return;
179                 }
180 
181                 return module.exposed;
182             }
183 
184             //内部方法: set 操作
185             function set(id, exposed) {
186                 var module = id$module[id];
187                 if (module) {
188                     module.exposed = !!exposed;
189                 }
190             }
191 
192             //set 操作
193             if (typeof id == 'object') { // expose({ })，批量 set
194                 var id$exposed = id;
195                 for (var id in id$exposed) {
196                     var exposed = id$exposed[id];
197                     set(id, exposed);
198                 }
199                 return;
200             }
201 
202             if (arguments.length == 2) { // expose('', true|false)，单个 set
203                 set(id, exposed);
204                 return;
205             }
206 
207             //get 操作
208             return get(id);
209         },
210 
211         /**
212         * 销毁本实例。
213         */
214         destroy: function () {
215             var guid = this[guidKey];
216             delete this[guidKey];
217             delete guid$meta[guid];
218         },
219     };
220 
221     return Module;
222 
223 })();

227     seperator: '/',
228     crossover: true,
229     shortcut: true,
230 });
231 
232 //提供快捷方式
233 var define = mod.define.bind(mod);
234 var require = mod.require.bind(mod);
235 var expose = mod.expose.bind(mod);

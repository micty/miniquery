  3 if (!Function.prototype.bind) {
  4     Function.prototype.bind = function (thisArg) {
  5         // this 指向的是要绑定的函数。
  6         if (typeof this !== "function") {
  7             throw new TypeError("Function.prototype.bind - 要绑定的对象只能是函数。");
  8         }
  9 
 10         var params = [].slice.call(arguments, 1);
 11         var self = this;
 12 
 13         return function () {
 14             var args = [].slice.call(arguments, 0);
 15             args = params.concat(args);
 16             return self.apply(thisArg, args);
 17         };
 18     };
 19 }

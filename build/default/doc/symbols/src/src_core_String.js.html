 11         /**
 12         * 用指定的值去填充一个字符串。
 13         * 当不指定字符串的填充标记时，则默认为 {}。
 14         * @param {String} string 要进行格式填充的字符串模板。
 15         * @param {Object} obj 要填充的键值对的对象。
 16         * @return 返回一个用值去填充后的字符串。
 17         * @example
 18             $String.format('{id}{type}', {id: 1, type: 'app'});
 19             $String.format('{2}{0}{1}', 'a', 'b', 'c');
 20         */
 21         format: function (string, obj, arg2) {
 22             var s = string;
 23             var replaceAll = exports.replaceAll;
 24 
 25             if (typeof obj == 'object') {
 26                 for (var key in obj) {
 27                     s = replaceAll(s, '{' + key + '}', obj[key]);
 28                 }
 29             }
 30             else {
 31                 var args = [].slice.call(arguments, 1);
 32                 for (var i = 0, len = args.length; i < len; i++) {
 33                     s = replaceAll(s, '{' + i + '}', args[i]);
 34                 }
 35             }
 36 
 37             return s;
 38         },

 59         * @param {String} string 要进行替换的目标字符串。
 60         * @param {String} startTag 区间的开始标记。
 61         * @param {String} endTag 区间的结束标记
 62         * @param {String} newString 要进行替换的新子串，新值。
 63         * @return {String} 返回一个替换后的字符串。<br />

 69         replaceBetween: function (string, startTag, endTag, newString) {
 70             var startIndex = string.indexOf(startTag);
 71             if (startIndex < 0) {
 72                 return string;
 73             }
 74 
 75             var endIndex = string.indexOf(endTag);
 76             if (endIndex < 0) {
 77                 return string;
 78             }
 79 
 80             var prefix = string.slice(0, startIndex);
 81             var suffix = string.slice(endIndex + endTag.length);
 82 
 83             return prefix + newString + suffix;

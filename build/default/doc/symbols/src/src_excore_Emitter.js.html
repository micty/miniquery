  2 /**
  3 * 自定义多级事件类
  4 * @class
  5 * @name Emitter
  6 */
  7 define('Emitter', function (require, module, exports) {
  8 
  9     var $ = require('$');
 10     var $Array = require('Array');
 11     var $Object = require('Object');
 12     var $String = require('String');
 13     var Mapper = require('Mapper');
 14 
 15     var Tree = require('/Tree'); //完整名称为 Emitter/Tree
 16 
 17     var mapper = new Mapper();
 18 
 19 
 20 
 21     //绑定事件。
 22     //实例的私有方法，必须用 bind.apply(this, []) 的方式来调用。
 23     function bind(isOneOff, name, fn) {
 24 
 25         var meta = mapper.get(this);
 26         var all = meta.all;
 27 
 28         // 单名称情况 on(name, fn)，专门成一个分支，为了优化
 29         if (typeof name == 'string' && typeof fn == 'function') {
 30             Tree.add(all, [name], {
 31                 'fn': fn,
 32                 'isOneOff': isOneOff,
 33             });
 34             return;
 35         }
 36 
 37         //多名称情况
 38         var args = $.toArray(arguments).slice(1);
 39 
 40         //重载 bind(isOneOff, name0, name1, ..., nameN, {...}) 的情况
 41         //先尝试找到 {} 所在的位置
 42         var index = $Array.findIndex(args, function (item, index) {
 43             return typeof item == 'object';
 44         });
 45 
 46         if (index >= 0) {
 47             var obj = args[index];
 48             var names = args.slice(0, index);
 49             var list = $Object.linearize(obj);
 50 
 51             $Array.each(list, function (item, index) {
 52 
 53                 var keys = names.concat(item.keys);
 54 
 55                 Tree.add(all, keys, {
 56                     'fn': item.value,
 57                     'isOneOff': isOneOff,
 58                 });
 59             });
 60             return;
 61         }
 62 
 63 
 64         //重载 bind(isOneOff, name0, name1, ..., nameN, fn) 的情况
 65         //尝试找到回调函数 fn 所在的位置
 66         var index = $Array.findIndex(args, function (item, index) {
 67             return typeof item == 'function';
 68         });
 69 
 70         if (index < 0) {
 71             throw new Error('参数中必须指定一个回调函数');
 72         }
 73 
 74         fn = args[index]; //回调函数
 75         var names = args.slice(0, index); //前面的都当作是名称
 76 
 77         Tree.add(all, names, {
 78             'fn': fn,
 79             'isOneOff': isOneOff,
 80         });
 81     }
 82 
 83 
 84 
 85 
 86 
 87     /**
 88     * 构造器。
 89     * @param {Object} [context=null] 事件处理函数中的 this 上下文对象。
 90     *   如果不指定，则默认为 null。
 91     */
 92     function Emitter(context) {
 93 
 94         var id = 'Emitter-' + $String.random();
 95         Mapper.setGuid(this, id);
 96 
 97         var meta = {
 98             'context': context,
 99             'all': {},
100         };
101 
102         mapper.set(this, meta);
103 
104     }
105 
106     //实例方法
107     Emitter.prototype = { /**@lends Emitter.prototype */
108         constructor: Emitter,
109 
110         /**
111         * 绑定指定名称的事件处理函数。
112         * @param {string} name 要绑定的事件名称。
113         * @param {function} fn 事件处理函数。 
114             在处理函数内部， this 指向构造器参数 context 对象。
115         * @example
116         */
117         on: function (name, fn) {
118 
119             var args = $.toArray(arguments);
120             args = [false].concat(args);
121             bind.apply(this, args);
122 
123         },
124 
125         /**
126         * 绑定指定名称的一次性事件处理函数。
127         * @param {string} name 要绑定的事件名称。
128         * @param {function} fn 事件处理函数。 
129             在处理函数内部， this 指向构造器参数 context 对象。
130         * @example
131         */
132         one: function (name, fn) {
133             var args = $.toArray(arguments);
134             args = [true].concat(args);
135             bind.apply(this, args);
136         },
137 
138 
139         /**
140         * 解除绑定指定名称的事件处理函数。
141         * @param {string} [name] 要解除绑定的事件名称。
142             如果不指定该参数，则移除所有的事件。
143             如果指定了该参数，其类型必须为 string，否则会抛出异常。
144         * @param {function} [fn] 要解除绑定事件处理函数。
145             如果不指定，则移除 name 所关联的所有事件。
146         */
147         off: function (name, fn) {
148 
149             var meta = mapper.get(this);
150             var all = meta.all;
151 
152             //未指定事件名，则移除所有的事件
153             if (name === undefined) {
154                 meta.all = {};
155                 return;
156             }
157 
158             //多名称情况: fire(name0, name1, name2, ..., nameN, fn)
159             var args = $.toArray(arguments);
160 
161             //找到回调函数所在的位置
162             var index = $Array.findIndex(args, function (item, index) {
163                 return typeof item == 'function';
164             });
165 
166             if (index < 0) {
167                 index = args.length;
168             }
169 
170             fn = args[index];
171 
172             var names = args.slice(0, index);
173             var node = Tree.getNode(all, names);
174             if (!node) { //尚未存在该名称所对应的节点
175                 return;
176             }
177 
178             var list = node.list;
179             if (list.length == 0) {
180                 return;
181             }
182 
183             if (fn) {
184                 node.list = $Array.grep(list, function (item, index) {
185                     return item.fn !== fn;
186                 });
187             }
188             else { //未指定处理函数，则清空列表
189                 list.length = 0;
190             }
191 
192         },
193 
194         /**
195         * 触发指定名称的事件，并可向事件处理函数传递一些参数。
196         * @param {string} name 要触发的事件名称。
197         * @param {Array} [params] 要向事件处理函数传递的参数数组。
198         * @return {Array} 返回所有事件处理函数的返回值所组成的一个数组。
199         */
200         fire: function (name, params) {
201 
202             var meta = mapper.get(this);
203             var all = meta.all;
204             var context = meta.context;
205 
206             //多名称情况: fire(name0, name1, name2, ..., nameN, params)
207             var args = $.toArray(arguments);
208 
209             //找到回调函数所在的位置
210             var index = $Array.findIndex(args, function (item, index) {
211                 return item instanceof Array;
212             });
213 
214             if (index < 0) {
215                 index = args.length;
216             }
217 
218             params = args[index] || [];
219 
220             var names = args.slice(0, index);
221             var list = Tree.getList(all, names);
222             var returns = [];
223 
224             if (!list || list.length == 0) {
225                 return returns;
226             }
227 
228             //这里要特别注意，在执行回调的过程中，回调函数里有可能会去修改回调列表，
229             //而此处又要去移除那些一次性的回调（即只执行一次的），
230             //为了避免破坏回调函数里的修改结果，这里要边移除边执行回调，而且每次都要
231             //以原来的回调列表为准去查询要移除的项的确切位置。
232 
233             var items = list.slice(0); //复制一份，因为回调列表可能会在执行回调过程发生变化。
234 
235             $Array.each(items, function (item, index) {
236 
237                 if (item.isOneOff) { //只需执行一次的
238                     var index = $Array.indexOf(list, item); //找到该项在回调列表中的索引位置
239                     if (index >= 0) {
240                         list.splice(index, 1); //直接从原数组删除
241                     }
242                 }
243 
244                 var value = item.fn.apply(context, params); //让 fn 内的 this 指向 obj
245                 returns.push(value);
246 
247             });
248 
249             return returns;
250 
251         },
252 
253         /**
254         * 检测是否包含指定名称的事件。
255         * @param {string} [name] 要检测的事件名称。
256             当不指定时，则判断是否包含了任意类型的事件。
257         * @param {function} [fn] 要检测的事件处理函数。
258         * @return {boolean} 返回一个布尔值，该布尔值指示目标对象上是否包含指否类型的事件以及指定的处理函数。
259             如果是则返回 true；否则返回 false。
260         */
261         has: function (name, fn) {
262 
263             var meta = mapper.get(this);
264             var all = meta.all;
265 
266             if (arguments.length == 0) { //未指定名称，则判断是否包含了任意类型的事件
267                 return !$Object.isEmpty(all);
268             }
269 
270             //多名称情况: fire(name0, name1, name2, ..., nameN, fn)
271             var args = $.toArray(arguments);
272 
273             //找到回调函数所在的位置
274             var index = $Array.findIndex(args, function (item, index) {
275                 return typeof item == 'function';
276             });
277 
278             if (index < 0) {
279                 index = args.length;
280             }
281 
282             fn = args[index];
283 
284             var names = args.slice(0, index);
285             var list = Tree.getList(all, names);
286 
287             if (!list || list.length == 0) { //尚未存在该名称所对应的节点
288                 return false;
289             }
290 
291             if (!fn) {
292                 return true;
293             }
294 
295             //指定了回调函数
296             return $Array.find(list, function (item, index) {
297                 return item.fn === fn;
298             });
299 
300 
301         },
302 
303 
304     };
305 
306     module.exports = Emitter;

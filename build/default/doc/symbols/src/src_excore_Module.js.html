  2 /**
  3 * 模块管理器类。
  4 * 主要提供给页面定义页面级别的私有模块。
  5 * @class
  6 * @name Module
  7 */
  8 define('Module', function (require, module, exports) {
  9 
 10     var $ = require('$');
 11 
 12     var mod = new Module(); //默认的、静态的
 13 
 14     //提供静态的调用方式
 15     module.exports = $.extend(Module, { /**@lends Module*/
 16 
 17         /**
 18         * @static
 19         * @function
 20         * @borrows Module.protoype.define
 21         */
 22         'define': mod.define.bind(mod),
 23         'require': mod.require.bind(mod)
 24     });
 25 
 26 });

  2 /**
  3 * 本地存储工具类
  4 * @namespace
  5 * @name LocalStorage
  6 */
  7 define('LocalStorage', function (require, module, exports) {
  8 
  9     var localStorage = window.localStorage;
 10 
 11     if (!localStorage) { //不支持
 12         return null; //须显式的返回 null，以告诉 require 加载器已加载过
 13     }
 14 
 15 
 16     var id = '__MiniQuery.LocalStorage__';
 17     var all = localStorage.getItem(id) || null;
 18     all = JSON.parse(all) || {};
 19 
 20 
 21 
 22     module.exports = exports = { /**@lends LocalStorage*/
 23 
 24         set: function (key, value) {
 25 
 26             if (key in all && all[key] === value) { //已存在
 27                 return;
 28             }
 29 
 30             all[key] = value;
 31 
 32             var json = JSON.stringify(all);
 33             localStorage.setItem(id, json);
 34 
 35         },
 36 
 37         get: function (key) {
 38             return all[key];
 39         },
 40 
 41         remove: function (key) {
 42             delete all[key];
 43             var json = JSON.stringify(all);
 44             localStorage.setItem(id, json);
 45         },
 46 
 47         clear: function () {
 48             all = {};
 49             var json = JSON.stringify(all);
 50             localStorage.setItem(id, json);
 51         },
 52 
 53         each: function (fn) {
 54             for (var key in all) {
 55                 fn(key, all[key]);
 56             }
 57         },
 58 
 59         length: function () {
 60             return exports.keys().length;
 61         },
 62 
 63         keys: function () {
 64             if (typeof Object.keys == 'function') {
 65                 return Object.keys(all);
 66             }
 67 
 68             var a = [];
 69             for (var key in all) {
 70                 a.push(key);
 71             }
 72 
 73             return a;
 74         },
 75 
 76         key: function (index) {
 77             return exports.keys()[index];
 78         }
 79     };
 80 
 81 
 82 });

  8     function getDateItem(s) {
  9         var now = new Date();
 10 
 11         var separator =
 12                 s.indexOf('.') > 0 ? '.' :
 13                 s.indexOf('-') > 0 ? '-' :
 14                 s.indexOf('/') > 0 ? '/' :
 15                 s.indexOf('_') > 0 ? '_' : null;
 16 
 17         if (!separator) {
 18             return null;
 19         }
 20 
 21         var ps = s.split(separator);
 22 
 23         return {
 24             'yyyy': ps[0],
 25             'MM': ps[1] || 0,
 26             'dd': ps[2] || 1
 27         };
 28     }
 29 
 30     function getTimeItem(s) {
 31         var separator = s.indexOf(':') > 0 ? ':' : null;
 32         if (!separator) {
 33             return null;
 34         }
 35 
 36         var ps = s.split(separator);
 37 
 38         return {
 39             'HH': ps[0] || 0,
 40             'mm': ps[1] || 0,
 41             'ss': ps[2] || 0
 42         };
 43     }

 58         /**
 59         * 把参数 value 解析成等价的日期时间实例。
 60         * @param {Date|string} value 要进行解析的参数，可接受的类型为：
 61         *   1.Date 实例
 62         *   2.string 字符串，包括调用 Date 实例的 toString 方法得到的字符串；也包括以下格式: 
 63                 yyyy-MM-dd
 64                 yyyy.MM.dd
 65                 yyyy/MM/dd
 66                 yyyy_MM_dd
 67                 HH:mm:ss
 68                 yyyy-MM-dd HH:mm:ss
 69                 yyyy.MM.dd HH:mm:ss
 70                 yyyy/MM/dd HH:mm:ss
 71                 yyyy_MM_dd HH:mm:ss
 72         * @return 返回一个日期时间的实例。
 73             如果解析失败，则返回 null。
 74         * @example
 75             $Date.parse('2013-04-29 09:31:20');
 76         */
 77         parse: function (value) {
 78             if (value instanceof Date) {
 79                 return isNaN(value.getTime()) ? null : value;
 80             }
 81 
 82             if (typeof value != 'string') {
 83                 return null;
 84             }
 85 
 86 
 87             //标准方式
 88             var date = new Date(value);
 89             if (!isNaN(date.getTime())) {
 90                 return date;
 91             }
 92 
 93             /*
 94              自定义方式：
 95                 yyyy-MM-dd
 96                 yyyy.MM.dd
 97                 yyyy/MM/dd
 98                 yyyy_MM_dd
 99                 HH:mm:ss
100                 yyyy-MM-dd HH:mm:ss
101                 yyyy.MM.dd HH:mm:ss
102                 yyyy/MM/dd HH:mm:ss
103                 yyyy_MM_dd HH:mm:ss
104                     
105             */
106 
107             var parts = value.split(' ');
108             var left = parts[0];
109 
110             if (!left) {
111                 return null;
112             }
113 
114             //冒号只能用在时间的部分，而不能用在日期部分
115             var date = left.indexOf(':') > 0 ? null : left;
116             var time = date ? (parts[1] || null) : date;
117 
118             if (!date && !time) { //既没指定日期部分，也没指定时间部分
119                 return null;
120             }
121 
122 
123             if (date && time) {
124                 var d = getDateItem(date);
125                 var t = getTimeItem(time);
126                 return new Date(d.yyyy, d.MM - 1, d.dd, t.HH, t.mm, t.ss);
127             }
128 
129             if (date) {
130                 var d = getDateItem(date);
131                 return new Date(d.yyyy, d.MM - 1, d.dd);
132             }
133 
134             if (time) {
135                 var now = new Date();
136                 var t = getTimeItem(time);
137                 return new Date(now.getFullYear(), now.getMonth(), now.getDate(), t.HH, t.mm, t.ss);
138             }
139             
140         },

142         /**
143         * 把日期时间格式化指定格式的字符串。
144         * @param {Date} datetime 要进行格式化的日期时间。
145         * @param {string} formater 格式化的字符串。 其中保留的占位符有：
146             'yyyy': 4位数年份
147             'yy': 2位数年份
148             'MM': 2位数的月份(01-12)
149             'M': 1位数的月份(1-12)
150             'dddd': '星期日|一|二|三|四|五|六'
151             'dd': 2位数的日份(01-31)
152             'd': 1位数的日份(1-31)
153             'HH': 24小时制的2位数小时数(00-23)
154             'H': 24小时制的1位数小时数(0-23)
155             'hh': 12小时制的2位数小时数(00-12)
156             'h': 12小时制的1位数小时数(0-12)
157             'mm': 2位数的分钟数(00-59)
158             'm': 1位数的分钟数(0-59)
159             'ss': 2位数的秒钟数(00-59)
160             's': 1位数的秒数(0-59)
161             'tt': 上午：'AM'；下午: 'PM'
162             't': 上午：'A'；下午: 'P'
163             'TT': 上午： '上午'； 下午: '下午'
164             'T': 上午： '上'； 下午: '下'
165         * @return {string} 返回一个格式化的字符串。
166         * @example
167             //返回当前时间的格式字符串，类似 '2013年4月29日 9:21:59 星期一'
168             $Date.format(new Date(), 'yyyy年M月d日 h:m:s dddd')
169         */
170         format: function (datetime, formater) {
171 
172             var $String = require('String');
173 
174             var year = datetime.getFullYear();
175             var month = datetime.getMonth() + 1;
176             var date = datetime.getDate();
177             var hour = datetime.getHours();
178             var minute = datetime.getMinutes();
179             var second = datetime.getSeconds();
180 
181             var padLeft = function (value, length) {
182                 return $String.padLeft(value, length, '0');
183             };
184 
185 
186             var isAM = hour <= 12;
187 
188             //这里不要用 {} 来映射，因为 for in 的顺序不确定
189             var maps = [
190                 ['yyyy', padLeft(year, 4)],
191                 ['yy', String(year).slice(2)],
192                 ['MM', padLeft(month, 2)],
193                 ['M', month],
194                 ['dddd', '星期' + ('日一二三四五六'.charAt(datetime.getDay()))],
195                 ['dd', padLeft(date, 2)],
196                 ['d', date],
197                 ['HH', padLeft(hour, 2)],
198                 ['H', hour],
199                 ['hh', padLeft(isAM ? hour : hour - 12, 2)],
200                 ['h', isAM ? hour : hour - 12],
201                 ['mm', padLeft(minute, 2)],
202                 ['m', minute],
203                 ['ss', padLeft(second, 2)],
204                 ['s', second],
205                 ['tt', isAM ? 'AM' : 'PM'],
206                 ['t', isAM ? 'A' : 'P'],
207                 ['TT', isAM ? '上午' : '下午'],
208                 ['T', isAM ? '上' : '下']
209             ];
210 
211 
212             var s = formater;
213             var replaceAll = $String.replaceAll;
214 
215             for (var i = 0, len = maps.length; i < len; i++) {
216                 var item = maps[i];
217                 s = replaceAll(s, item[0], item[1]);
218             }
219 
220             return s;
221         },

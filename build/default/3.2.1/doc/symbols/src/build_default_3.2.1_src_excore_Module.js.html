  2 /**
  3 * 对外提供的模块管理器类。
  4 * 主要提供给上层框架/业务进一步定义属于自己的命名空间的模块。
  5 */
  6 define('Module', function (require, module, exports) {
  7 
  8     var $ = require('$');
  9     var mod = new Module({
 10         seperator: '/',
 11         crossover: true,
 12         shortcut: true,
 13     });
 14 
 15 
 16     $.extend(Module, /**@lends Module*/ {
 17 
 18         /**
 19         * 静态方法。
 20         * @function
 21         * @memberOf Module
 22         */
 23         define: mod.define.bind(mod),
 24 
 25         /**
 26         * 静态方法。
 27         * @function
 28         * @memberOf Module
 29         */
 30         require: mod.require.bind(mod)
 31     });
 32 
 33 
 34     module.exports = exports = Module;
 35 
 36 });

  2 /**
  3 * 本地存储工具类
  4 * @namespace
  5 * @name LocalStorage
  6 */
  7 define('LocalStorage', function (require, module, exports) {
  8 
  9     var storage = window.localStorage;
 10 
 11     if (!storage) { //不支持
 12         return null; //须显式的返回 null，以告诉 require 加载器已加载过
 13     }
 14 
 15 
 16     var id = '__MiniQuery.LocalStorage__';
 17     var all = storage.getItem(id) || null;
 18     all = JSON.parse(all) || {};
 19 
 20 
 21 
 22     module.exports = exports = /**@lends LocalStorage*/ {
 23 
 24         /**
 25         * 设置一对键值。
 26         * @param {string} key 要进行设置的键名称。
 27         * @param value 要进行设置的值，可以是任何类型。
 28         */
 29         set: function (key, value) {
 30 
 31             if (key in all && all[key] === value) { //已存在
 32                 return;
 33             }
 34 
 35             all[key] = value;
 36 
 37             var json = JSON.stringify(all);
 38             storage.setItem(id, json);
 39 
 40         },
 41 
 42         /**
 43         * 根据给定的键获取关联的值。
 44         * @param {string} key 要进行获取的键名称。
 45         * @return 返回该键所关联的值。
 46         */
 47         get: function (key) {
 48             return all[key];
 49         },
 50 
 51         /**
 52         * 移除给定的键所关联的项。
 53         * @param {string} key 要进行移除的键名称。
 54         */
 55         remove: function (key) {
 56             delete all[key];
 57             var json = JSON.stringify(all);
 58             storage.setItem(id, json);
 59         },
 60 
 61         /**
 62         * 清空所有项。
 63         */
 64         clear: function () {
 65             all = {};
 66             var json = JSON.stringify(all);
 67             storage.setItem(id, json);
 68         },
 69 
 70         /**
 71         * 对每一项进行迭代，并调用传入的回调函数。
 72         * @param {function} fn 迭代调用的回调函数。
 73             该函数会接收到两个参数: 
 74             key: 当前键的名称。
 75             value: 当前键所关联的值。
 76         */
 77         each: function (fn) {
 78             for (var key in all) {
 79                 fn(key, all[key]);
 80             }
 81         },
 82 
 83         /**
 84         * 获取所有的项的总个数。
 85         */
 86         length: function () {
 87             return exports.keys().length;
 88         },
 89 
 90         /**
 91         * 获取所有的项的键数组。
 92         */
 93         keys: function () {
 94             if (typeof Object.keys == 'function') {
 95                 return Object.keys(all);
 96             }
 97 
 98             var a = [];
 99             for (var key in all) {
100                 a.push(key);
101             }
102 
103             return a;
104         },
105 
106         /**
107         * 获取所有的项的键数组指定中的项。
108         * @param {number} index 键所对应的索引值。
109         */
110         key: function key(index) {
111             return exports.keys()[index];
112         }
113     };
114 
115 
116 });

  2 /**
  3 * 映射器工具类。
  4 * 实现任意类型的两个变量的关联。

  6 * @name Mapper

  8 define('Mapper', function (require, module, exports) {
  9 
 10     var $ = require('$');
 11     var $String = require('String');
 12     var $Array = require('Array');
 13 
 14     var guidKey = '__guid-' + $String.random();
 15     var guid$all = {}; //容纳所有 Mapper 实例的数据
 16 
 17 
 18     function getString(obj) {
 19 
 20         //函数的 length 属性表示形参的个数，而且是只读的。 
 21         //详见 https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/Function/length
 22         //函数的 name 属性表示函数的名称，而且是只读的。 匿名函数的名称为空字符串。 
 23         //详见 https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/Function/name
 24         if (typeof obj == 'function') {
 25 
 26             var a = [];
 27 
 28             if ('name' in obj) {
 29                 a.push(obj.name);
 30             }
 31 
 32             if ('length' in obj) {
 33                 a.push(obj.length);
 34             }
 35 
 36             if (a.length > 0) {
 37                 return a.join('#');
 38             }
 39         }
 40 
 41         var $Object = require('Object');
 42 
 43         return $Object.getType(obj); //返回相应构造器的名称，不要用　toString，因为可能会变
 44 
 45     }

 51     /**
 52     * 构造器。
 53     * @inner
 54     */
 55     function Mapper() {
 56 
 57         //分配本实例对应的容器
 58         var guid = $String.random();
 59         this[guidKey] = guid;
 60 
 61         guid$all[guid] = {
 62             'guid': {},     //针对有 guid 属性的对象作特殊存取，以求一步定位。
 63             'false': {},    //针对 false|null|undefined|''|0|NaN
 64             'string': {},
 65             'number': {},
 66             'boolean': {},  //只针对 true
 67             'object': {},
 68             'function': {},
 69             'undefined': {} //这个用不到
 70         };
 71     }
 72 
 73     //实例方法
 74     Mapper.prototype = { /**@lends Mapper# */
 75 
 76         constructor: Mapper,
 77         /**
 78         * 根据给定的键和值设置成一对映射关系。
 79         * @param key 映射关系的键，可以是任何类型。
 80         * @param value 映射关系要关联的值，可是任何类型。
 81         * @return 返回第二个参数 value。
 82         * @example
 83             var obj = { a: 1, b: 2 };
 84             var fn = function(a, b) {
 85                 console.log(a+b);
 86             };
 87             var mapper = new $.Mapper();
 88             mapper.set(obj, fn);
 89             mapper.set('a', 100);
 90             mapper.set(100, 'abc');
 91             mapper.set('100', 'ABC');
 92             mapper.set(null, 'abc');
 93             
 94         */
 95         set: function (key, value) {
 96 
 97             var guid = this[guidKey];
 98             var all = guid$all[guid]; //当前实例的容器
 99 
100             var type = typeof key;
101 
102             //针对含有 guid 属性的对象作优先处理
103             if (key && type == 'object' && guidKey in key) { 
104                 var guid = key[guidKey];
105                 all['guid'][guid] = value; //一步定位
106                 return value;
107             }
108 
109             //false
110             if (!key) { // false|null|undefined|''|0|NaN
111                 all['false'][String(key)] = value;
112                 return value;
113             }
114 
115             //值类型
116             if ((/^(string|number|boolean)$/g).test(type)) {
117                 all[type][String(key)] = value;
118                 return value;
119             }
120 
121             //引用类型: object|function
122 
123             var key$list = all[type];
124             key = getString(key); //这里确保 key 一定是一个 string
125             var list = key$list[key];
126 
127             //未存在对应字符串的列表，则创建并添加
128             if (!list) {
129                 key$list[key] = [[key, value]];
130                 return value;
131             }
132 
133 
134             //已存在对应字符串的列表
135             var pair = $Array.findItem(list, function (pair, index) {
136                 return pair[0] === key;
137             });
138 
139             if (pair) { //已存在，
140                 pair[1] = value; //改写值
141             }
142             else { //未找到，创建新的，添加进去一对二元组 [key, value]
143                 list.push([key, value]);
144             }
145 
146             return value;
147         },
148 
149 
150         /**
151         * 根据给定的键去获取其关联的值。
152         * 注意：根据映射关系的键查找所关联的值时，对键使用的是全等比较，即区分数据类型。
153         * @param key 映射关系的键，可以是任何类型。
154         * @return 返回映射关系所关联的值。
155         * @example
156             var obj = { a: 1, b: 2 };
157             var fn = function(a, b) {
158                 console.log(a+b);
159             };
160             
161             var mapper = new $.Mapper();
162             mapper.set(obj, fn);
163             
164             var myFn = mapper.get(obj); //获取到之前关联的 fn
165             myFn(100, 200);
166             console.log(fn === myFn);
167         */
168         get: function (key) {
169 
170             var guid = this[guidKey];
171             var all = guid$all[guid];
172 
173             var type = typeof key;
174 
175             //针对含有 guid 属性的对象作优先处理
176             if (key && type == 'object' && guidKey in key) { 
177                 var guid = key[guidKey];
178                 return all['guid'][guid]; //一步定位
179             }
180 
181             //false
182             if (!key) { // false|null|undefined|''|0|NaN
183                 return all['false'][String(key)];
184             }
185 
186 
187             //值类型，直接映射
188             if ((/^(string|number|boolean)$/g).test(type)) {
189                 return all[type][String(key)];
190             }
191 
192             //引用类型: object|function
193             //通过 key 映射到一个二维数组，每个二维数组项为 [key, value]
194             key = getString(key);
195             var list = all[type][key];
196 
197             if (!list) {
198                 return;
199             }
200 
201 
202             //已存在对应字符串的列表
203             var pair = $Array.findItem(list, function (pair, index) {
204                 return pair[0] === key;
205             });
206 
207             return pair ? pair[1] : undefined;
208 
209         },
210 
211 
212         /**
213         * 根据给定的键移除一对映射关系。
214         * 注意：根据映射关系的键查找所关联的值时，对键使用的是全等比较，即区分数据类型。
215         * @param key 映射关系的键，可以是任何类型。
216         * @example
217             var obj = { a: 1, b: 2 };
218             var fn = function(a, b) {
219                 console.log(a+b);
220             };
221             
222             var mapper = new $.Mapper();
223             mapper.set(obj, fn);
224             
225             mapper.remove(obj);
226             fn = mapper.get(obj);
227             console.log( typeof fn); // undefined
228         */
229         remove: function (key) {
230 
231             var guid = this[guidKey];
232             var all = guid$all[guid];
233 
234             var type = typeof key;
235 
236             //针对含有 guid 属性的对象作优先处理
237             if (key && type == 'object' && guidKey in key) { 
238                 var guid = key[guidKey];
239                 delete all['guid'][guid]; //一步定位
240                 return;
241             }
242 
243             //false
244             if (!key) { // false|null|undefined|''|0|NaN
245                 delete all['false'][String(key)];
246                 return;
247             }
248 
249 
250             //值类型
251             if ((/^(string|number|boolean)$/g).test(type)) {
252                 delete all[type][String(key)];
253                 return;
254             }
255 
256             //引用类型: object|function
257             var key$list = all[type];
258             key = getString(key);
259             var list = key$list[key];
260 
261             if (!list) {
262                 return;
263             }
264 
265             //已存在对应字符串的列表
266             //移除 key 的项
267             key$list[key] = $Array.grep(list, function (pair, index) {
268                 return pair[0] !== key;
269             });
270 
271         },
272 
273 
274         /**
275         * 销毁本实例。
276         * 这会移除所有的映射关系，并且移除本实例内部使用的存放映射关系的容器对象。
277         */
278         destroy: function () {
279             var guid = this[guidKey];
280             delete guid$all[guid];
281             delete this[guidKey]; //把 guid 掉，无法再访问数据
282         }
283 
284     };

293         /**
294         * 获取运行时确定的随机 guid 值所使用的 key。
295         * @return {string} 返回guid 值所使用的 key。
296         */
297         getGuidKey: function () {
298             return guidKey;
299         },
300 
301         /**
302         * 给指定的对象设置一个 guid 值。
303         * @param {Object} 要设置的对象，只要是引用类型即可。
304         * @param {string} [guid] 要设置的 guid 值。
305         *   当不指定时，则分配一个默认的随机字符串。(以 'default-' 开头 )
306         * @return {string} 返回设置后的 guid 值。
307         */
308         setGuid: function (obj, guid) {
309             if (guid === undefined) {
310                 guid = 'default-' + $String.random();
311             }
312 
313             obj[guidKey] = guid;
314             return guid;
315         },
316 
317         /**
318         * 获取指定的对象的 guid 值。
319         * @param {Object} 要获取的对象，只要是引用类型即可。
320         * @return {string} 返回该对象的 guid 值。
321         */
322         getGuid: function (obj) {
323             return obj[guidKey];
324         },
325     });

327     module.exports = Mapper;

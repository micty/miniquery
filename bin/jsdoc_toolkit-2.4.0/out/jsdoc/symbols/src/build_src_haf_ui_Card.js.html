  5 ; (function (Haf, $, Meta) {
  6 
  7 
  8 /**
  9 * Card 容器
 10 */
 11 Haf.define('Haf.view.Card', {
 12     extend: 'Haf.view.Container',
 13     xtype: 'card',
 14     nativeProps: [],
 15 
 16     config: {
 17 
 18     },
 19 
 20     initialize: function (config) {
 21 
 22         this.callSuper(arguments, config);
 23 
 24         var item = this.getItem(0);
 25         if (item) { //默认激活第一项(如果有)
 26             this.setActiveItem(0);
 27         }
 28         else {
 29             Meta.set(this, 'activeIndex', -1);
 30         }
 31 
 32     },
 33 
 34     /**
 35     * 设置指定的组件为当前显示的组件
 36     * @param item
 37     */
 38     setActiveItem: function (item) {
 39 
 40         var index = item; //先假设传进来的是数字，当成 index
 41 
 42         item = this.getItem(item);
 43         if (!item) {
 44             throw Haf.error('容器中不存在该组件');
 45         }
 46 
 47         
 48         if (typeof index != 'number') {
 49             var items = Meta.get(this, 'items');
 50             index = $.Array.indexOf(items, item);
 51         }
 52 
 53         var activeIndex = this.getActiveIndex();
 54 
 55         if (index == activeIndex) {
 56             return;
 57         }
 58 
 59         Meta.set(this, 'activeIndex', index);
 60         this.callNative('showView', item);
 61 
 62         var oldItem = this.getItem(activeIndex);
 63 
 64         this.trigger('activeitemchange', [item, index, oldItem, activeIndex]);
 65 
 66     },
 67 
 68     getActiveItem: function () {
 69         var index = this.getActiveIndex();
 70         if (index < 0) {
 71             return null;
 72         }
 73 
 74         var items = Meta.get(this, 'items');
 75         return items[index] || null;
 76     },
 77 
 78     getActiveIndex: function () {
 79         return Meta.get(this, 'activeIndex');
 80     },
 81 
 82     next: function (isCircled) {
 83 
 84         var activeIndex = this.getActiveIndex();
 85         var count = this.getCount();
 86 
 87 
 88         var index =
 89             activeIndex < 0 ? 0 :
 90             isCircled === true ? $.Math.next(activeIndex, count) :
 91             Math.min(activeIndex + 1, count - 1);
 92 
 93         if (index == activeIndex) {
 94             return;
 95         }
 96 
 97         this.setActiveItem(index);
 98     },
 99 
100     previous: function (isCircled) {
101         var activeIndex = this.getActiveIndex();
102         var count = this.getCount();
103 
104         var index =
105             activeIndex < 0 ? count :
106             isCircled === true ? $.Math.previous(activeIndex, count) :
107             Math.max(activeIndex - 1, 0);
108 
109         if (index == activeIndex) {
110             return;
111         }
112 
113         this.setActiveItem(index);
114     }
115 });

  4 /**
  5 * 内部使用的组件辅助工具类。
  6 * @namespace
  7 */
  8 var ComponentHelper = (function ($, Haf, ClassManager) {
  9 
 10     //  haf.xtype -> native.xtype
 11     var xtype$native = {
 12 
 13         actionsheet: 'ActionSheet',
 14         box: 'Box',
 15         button: 'Button',
 16         card: 'Card',
 17         carousel: 'Carousel',
 18         datepicker: 'DatePicker',
 19         dialog: 'Dialog',
 20         framelayer: 'FrameLayer',
 21         image: 'Image',
 22         label: 'Label',
 23         list: 'List',
 24         mask: 'ScreenMask',
 25         popupwindow: 'PopupWindow',
 26         progressbar: 'ProgressBar',
 27         select: 'Select',
 28         slider: 'Slider',
 29         textarea: 'TextArea',
 30         textinput: 'TextInput',
 31         toast: 'Toast',
 32         toggle: 'Toggle',
 33         webview: 'WebView'
 34     };
 35 
 36     //维护 { className: nativeXtype } 的关系。
 37     var className$nativeXtype = {};
 38 
 39     /**
 40     * 获取指定组件实例的真实的 native 层的 xtype。
 41     * 该方法会从继承关系层次向上追溯，直到找到最近的 native 层的 xtype。
 42     */
 43     function getNativeXtype(self) {
 44 
 45         //先从直接映射表查找
 46         var xtype = xtype$native[self.xtype];
 47         if (xtype) {
 48             return xtype;
 49         }
 50 
 51         //从缓存中查找
 52         var className = self.getType();
 53         xtype = className$nativeXtype[className];
 54         if (xtype) {
 55             return xtype;
 56         }
 57         
 58         //再从继承树查找
 59         var classNames = ClassManager.getSuperNames(self);
 60 
 61         $.Array.each(classNames, function (name, index) {
 62 
 63             var data = ClassManager.getData(name);
 64             xtype = xtype$native[data.xtype];
 65 
 66             if (xtype) {
 67                 return false; //相当于 break
 68             }
 69 
 70         });
 71 
 72         className$nativeXtype[className] = xtype; //缓存起来
 73         return xtype;
 74     }
 75 
 76     
 77     function getNativeProps(self) {
 78 
 79         var props = self.nativeProps;
 80 
 81         if (!props) {
 82             return {};
 83         }
 84 
 85 
 86         var config = self.config;
 87 
 88         var obj =
 89             props instanceof Array ? //是数组，直接从 config 中把同名称的成员过滤出来
 90                 $.Object.filter(config, props) :
 91 
 92             typeof props == 'object' ? //是纯对象，则重新映射
 93                 $.Object.map(props, function (key, item) {
 94 
 95                     //是一个key，表示仅仅是换了名称
 96                     if (typeof item == 'string') {
 97                         return config[item];
 98                     }
 99 
100                     if ($.Object.isPlain(item)) { //此时 item = { name: '', values: {} }
101 
102                         var name = item.name;
103                         if (!(name in config)) { //config 中不包含 item.name 所指定所成员
104                             return undefined;
105                         }
106                         
107                         var value = config[name];
108 
109                         if (value in item.values) {
110                             return item.values[value];
111                         }
112 
113                         throw Haf.error('config.{0} 中的值不符合 nativeProps.{1} 的转换要求', name, key);
114 
115                     }
116 
117                     throw Haf.error('无法识别 nativeProps.{0} 所对应的值', key);
118 
119                 }) : {}; //否则
120                 
121 
122         return $.Object.trim(obj);  // natvie 比较敏感，必须过滤掉非法值 null, undefined
123     }
124 
125 
126 
127     /**
128     * 判断指定的事件名称是否为指定组件的 native 事件。
129     */
130     function isNativeEvent(obj, name) {
131 
132         var events = isValidComponent(obj) ? obj.nativeEvents : obj;
133 
134         return $.Object.isArray(events) ?
135                     $.Array.contains(events, name) :
136                     (name in events);
137     }
138 
139     //根据 haf 的属性名反向查找　native 属性名。
140     function getNativePropName(obj, name) {
141 
142         var props = isValidComponent(obj) ? obj.nativeProps : obj; //就保持这样的重载，更灵活
143 
144         // props 是一个数组，则表示 native 的名称与 config 中的名称一致
145         if ($.Object.isArray(props) && $.Array.contains(props, name)) {
146             return name;
147         }
148 
149         //否则，迭代查找每个成员
150         return $.Object.findKey(props, function (key, item) {
151 
152             return item === name ||
153                 $.Object.isPlain(item) && item.name === name;
154         });
155     }
156 
157     /**
158     * 判断指定的属性名称是否为指定组件的 native 属性。
159     */
160     function isNativeProp(obj, name) {
161 
162         return !!getNativePropName(obj, name);
163     }
164 
165     
166     
167     /**
168     * 判断指定的对象是否为合法的组件。
169     */
170     function isValidComponent(obj) {
171 
172         if (!obj || typeof obj != 'object') { //优化
173             return false;
174         }
175 
176         return ClassManager.isInstanceOf(obj, 'Haf.view.Component' );
177     }
178 
179     
180 
181     //为了复用
182     function onchangeHandler(newConfig, oldConfig) {
183 
184         var self = this;
185         var isPropName = ComponentHelper.isNativeProp;
186         var isStyleName = ComponentStyle.isValidName;
187 
188         $.Object.each(newConfig, function (key, value) {
189 
190             if (value === oldConfig[key]) { //值没发生改变
191                 return;
192             }
193 
194             if (isStyleName(key) || isPropName(self, key)) {
195 
196                 self.updateUI();
197                 return false; //break;
198             }
199         });
200     }
201 
202 
203 
204 
205     return {
206         isValidComponent: isValidComponent,
207         isNativeEvent: isNativeEvent,
208         isNativeProp: isNativeProp,
209 
210         getNativeXtype: getNativeXtype,
211 
212         getNativeProps: getNativeProps,
213         getNativePropName: getNativePropName,
214 
215         onchangeHandler: onchangeHandler
216     };
217 
218 })(Haf, Haf, ClassManager);

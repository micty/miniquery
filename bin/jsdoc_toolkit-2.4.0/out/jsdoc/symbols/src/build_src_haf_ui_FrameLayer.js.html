  4 /**
  5 * 叠加层。
  6 * 一个可叠加的层容器控件，加入到层中的元素都是从左上角原点出现的。
  7 */
  8 Haf.define('Haf.view.FrameLayer', {
  9     extend: 'Haf.view.Container',
 10     xtype: 'framelayer',
 11     nativeEvents: [],
 12     nativeProps: {
 13     },
 14 
 15     config: {
 16         
 17     },
 18 
 19     exchange: function (item1, item2) {
 20         
 21         item1 = this.getItem(item1);
 22         if (!item1) {
 23             throw Haf.error('容器中不包含 item1 的子组件');
 24         }
 25 
 26         item2 = this.getItem(item2);
 27         if (!item2) {
 28             throw Haf.error('容器中不包含 item2 的子组件');
 29         }
 30 
 31         this.callNative('exChangeLayerIndex', item1, item2);
 32     }

  2 /**
  3 * Boolean 工具类
  4 * @namespace
  5 */
  6 define('Boolean', function (require, module, exports) {
  7 
  8     module.exports = exports = { /**@lends MiniQuery.Boolean */
  9 
 10         /**
 11         * 解析指定的参数为 bool 值。
 12         * null、undefined、0、NaN、false、'' 及其相应的字符串形式会转成 false；
 13         * 其它的转成 true
 14         * @param {Object} arg 要进行进换的值，可以是任何类型。
 15         * @return {boolean} 返回一个 bool 值。
 16         * @example
 17             $Boolean.parse(null); //false;
 18             $Boolean.parse('null'); //false;
 19             $Boolean.parse(undefined); //false;
 20             $Boolean.parse('undefined'); //false;
 21             $Boolean.parse(0); //false;
 22             $Boolean.parse('0'); //false;
 23             $Boolean.parse(NaN); //false;
 24             $Boolean.parse('NaN'); //false;
 25             $Boolean.parse(false); //false;
 26             $Boolean.parse('false'); //false;
 27             $Boolean.parse(''); //false;
 28             $Boolean.parse(true); //true;
 29             $Boolean.parse({}); //true;
 30         */
 31         parse: function (arg) {
 32             if (!arg) {// null、undefined、0、NaN、false、''
 33                 return false;
 34             }
 35 
 36             if (typeof arg == 'string' || arg instanceof String) {
 37                 var reg = /^(false|null|undefined|0|NaN)$/g;
 38                 return !reg.test(arg);
 39             }
 40 
 41             return true;
 42         },
 43 
 44         /**
 45         * 解析指定的参数为 int 值：0 或 1。
 46         * null、undefined、0、NaN、false、'' 及其相应的字符串形式会转成 0；
 47         * 其它的转成 1
 48         * @param {Object} 要进行转换的值，可以是任何类型。
 49         * @return {int} 返回一个整型值 0 或 1。
 50         * @example
 51             $Boolean.toInt(null); //0;
 52             $Boolean.toInt('null'); //0;
 53             $Boolean.toInt(undefined); //0;
 54             $Boolean.toInt('undefined'); //0;
 55             $Boolean.toInt(0); //0;
 56             $Boolean.toInt('0'); //0;
 57             $Boolean.toInt(NaN); //0;
 58             $Boolean.toInt('NaN'); //0;
 59             $Boolean.toInt(false); //0;
 60             $Boolean.toInt('false'); //0;
 61             $Boolean.toInt(''); //0;
 62             $Boolean.toInt(true); //1;
 63             $Boolean.toInt({}); //1;
 64         */
 65         toInt: function (arg) {
 66             return exports.parse(arg) ? 1 : 0;
 67         },
 68 
 69         /**
 70         * 反转一个 boolean 值，即 true 变成 false；false 变成 true。
 71         * @param {Object} 要进行反转的值，可以是任何类型。
 72         * @return {int} 返回一个 bool 值。
 73         * @example
 74             $Boolean.reverse(null); //true;
 75             $Boolean.reverse('null'); //true;
 76             $Boolean.reverse(undefined); //true;
 77             $Boolean.reverse('undefined'); //true;
 78             $Boolean.reverse(0); //true;
 79             $Boolean.reverse('0'); //true;
 80             $Boolean.reverse(NaN); //true;
 81             $Boolean.reverse('NaN'); //true;
 82             $Boolean.reverse(false); //true;
 83             $Boolean.reverse('false'); //true;
 84             $Boolean.reverse(''); //true;
 85             $Boolean.reverse(true); //false;
 86             $Boolean.reverse({}); //false;
 87         */
 88         reverse: function (arg) {
 89             return !exports.parse(arg);
 90         },
 91 
 92         /**
 93         * 产生一个随机布尔值。
 94         * @return {boolean} 返回一个随机的 true 或 false。
 95         * @example
 96             $Boolean.random();
 97         */
 98         random: function () {
 99             return !!Math.floor(Math.random() * 2); //产生随机数 0 或 1
100         }
101     };
102 
103 });

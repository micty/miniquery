  2 /**
  3 * 内部用的一个 MiniQuery 容器
  4 * @namespace MiniQuery.$
  5 
  6 * @inner
  7 */
  8 define('$', function (require, module, exports) {
  9 
 10     var slice = [].slice;
 11 
 12     //兼容性写法
 13     var toArray = slice.call.bind ? slice.call.bind(slice) : function ($arguments) {
 14         return slice.call($arguments, 0);
 15     };
 16 
 17 
 18 
 19     module.exports = exports = { /**@lends MiniQuery.$*/
 20 
 21         expando: 'MiniQuery-' + Math.random().toString().slice(2),
 22 
 23         toArray: toArray,
 24 
 25         /**
 26         * 用指定的值去扩展指定的目标对象，返回目标对象。
 27         */
 28         extend: function (target, obj1, obj2) {
 29 
 30             //针对最常用的情况作优化
 31             if (obj1 && typeof obj1 == 'object') {
 32                 for (var key in obj1) {
 33                     target[key] = obj1[key];
 34                 }
 35             }
 36 
 37             if (obj2 && typeof obj2 == 'object') {
 38                 for (var key in obj2) {
 39                     target[key] = obj2[key];
 40                 }
 41             }
 42 
 43             var startIndex = 3;
 44             var len = arguments.length;
 45             if (startIndex >= len) { //已处理完所有参数
 46                 return target;
 47             }
 48 
 49             //更多的情况
 50             for (var i = startIndex; i < len; i++) {
 51                 var obj = arguments[i];
 52                 for (var name in obj) {
 53                     target[name] = obj[name];
 54                 }
 55             }
 56 
 57             return target;
 58         },
 59 
 60         
 61 
 62         /**
 63         * 把数组、类数组合并成一个真正的数组。
 64         */
 65         concat: function () {
 66 
 67             var a = [];
 68             var args = toArray(arguments);
 69 
 70             for (var i = 0, len = args.length; i < len; i++) {
 71 
 72                 var item = args[i];
 73                 if (!(item instanceof Array)) {
 74                     item = toArray(item);
 75                 }
 76 
 77                 a = a.concat(item);
 78             }
 79 
 80             return a;
 81         },
 82 
 83 
 84         require: function (id) {
 85             return expose(id) ? require(id) : null;
 86         },
 87     };
 88 
 89 });

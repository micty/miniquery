  9 * 弹出窗口容器。

 11 Haf.define('Haf.view.PopupWindow', {
 12     extend: 'Haf.view.Container',
 13     xtype: 'popupwindow',
 14     nativeEvents: {
 15         hide: 'dismiss'
 16     },
 17 
 18     nativeProps: {
 19 
 20         losable: 'losable',
 21         mask: 'mask',
 22         maskColor: 'maskColor',
 23         hasBorder: 'hasBorder',
 24 
 25         popAnimationType: {
 26             name: 'animation',
 27             values: {
 28                 top: 2,
 29                 right: 0,
 30                 bottom: 3,
 31                 left: 1
 32             }
 33         }
 34     },
 35 
 36     config: {
 37         losable: true, //是否易消失的，即点击窗口外部会自动关闭本窗口
 38         hasBorder: true,
 39 
 40         mask: {
 41             color: 'black',
 42             opacity: '80%'
 43         }
 44     },
 45 
 46     initialize: function (config) {
 47 
 48         var mask = config.mask;
 49         if (mask) {
 50             var color = 'black';
 51             var opacity = 0.5;
 52 
 53             if ($.Object.isPlain(mask)) {
 54                 color = mask.color || color;
 55                 opacity = mask.opacity === undefined ? opacity : mask.opacity;
 56             }
 57 
 58             this.set({
 59                 mask: true,
 60                 maskColor: Colors.get(color, opacity)
 61 
 62             }, true);
 63         }
 64 
 65         this.callSuper(arguments, config);
 66     },
 67 
 68     showAt: function (anchor, direction) {
 69 
 70         switch (typeof anchor) {
 71             case 'string':
 72                 if (direction === undefined) { //此时为 showAt(alignType)
 73                     var alignType = Aligns[anchor];
 74                     if (alignType === undefined) {
 75                         throw Haf.error('无法识别的 anchor 参数: {0}', alignType);
 76                     }
 77 
 78                     this.callNative('showByAlignType', alignType);
 79                     break;
 80                 }
 81                 //注意，这里没有 break;
 82 
 83             case 'object':  //此时为 showAt(anchor, direction)
 84                 anchor = ComponentManager.normalize(anchor);
 85                 direction = Directions[direction];
 86 
 87                 if (!direction) {
 88                     throw Haf.error('无法识别的 direction 参数: {0}', direction);
 89                 }
 90 
 91                 this.callNative('show', anchor, direction);
 92                 break;
 93 
 94             case 'number': //此时为 showAt(x, y)
 95                 var x = anchor;
 96                 var y = direction;
 97                 this.callNative('showAtPoint', x, y);
 98                 break;
 99 
100             default:
101                 throw Haf.error('无法识别参数 anchor 的类型');
102         }
103 
104     },
105 
106     show: function () {
107         this.showAt('center');
108     },
109 
110     hide: function () {
111         this.callNative('hide');
112     }
113 });

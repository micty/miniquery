 89     /**
 90     * 这是一个模板方法，此处提供 Component 组件的实现。
 91     * 该方法区分普通的 js 事件和 native 层的事件，并且不提供重载。
 92     */
 93     bind: function (eventName, handler, isOnce) {
 94 
 95         var self = this;
 96         var isNativeEvent = ComponentHelper.isNativeEvent(this, eventName);
 97 
 98         // 即普通的 js 事件
 99         if (!isNativeEvent) {
100             $.Event.bind(this, eventName, handler, isOnce);
101             return;
102         }
103 
104         // native 事件。
105         var nativeId = this.getNativeId();
106 
107         var nativeEvents = this.nativeEvents;
108         if ($.Object.isPlain(nativeEvents)) { //设置了新的映射名称，则进行转换
109             eventName = nativeEvents[eventName]; // haf -> hae
110         }
111 
112         //钩子函数
113         function exFn() {
114             var args = $.Array.parse(arguments);
115             handler.apply(self, args); //让 handler 函数中的 this 指向当前实例。
116 
117             if (isOnce) {
118                 UI.removeEventListener(nativeId, eventName, exFn);
119             }
120         }
121 
122         var mapper = Meta.get(this, 'bind.nativeEvents.mapper', new $.Mapper());
123         mapper.set(handler, exFn); // 用 mapper 找到关系： handler -> exFn
124 
125         UI.addEventListener(nativeId, eventName, exFn);
126 
127     },

129     /**
130     * 给本实例解除绑定事件处理函数。
131     * @param {string} [eventName] 要解除绑定的事件名称。
132         如果不指定，则移除所有的事件。
133     * @param {function} [handler] 要解除绑定事件处理函数。
134         如果不指定，则移除 eventName 所关联的所有事件。
135     */
136     off: function (eventName, handler) {
137 
138         var isNativeEvent = ComponentHelper.isNativeEvent(this, eventName);
139         if (!isNativeEvent) {
140             $.Event.unbind(this, eventName, handler);
141             return;
142         }
143 
144         var mapper = Meta.get(this, 'bind.nativeEvents.mapper');
145         if (!mapper) { //尚未绑定过 native 事件
146             return;
147         }
148 
149         var exFn = mapper.get(handler); //handler -> exFn
150         var nativeId = this.getNativeId();
151         UI.removeEventListener(nativeId, eventName, exFn);
152         
153 
154     },

173     /**
174     * 销毁本实例，同时执行一些清理操作。

190     /**
191     * 获取 config 对象中指定成员的值。
192     * 如果获取的成员是属性名/样式名，则从 native 层获取值，并且转换成 haf 层所使用的值。
193     */
194     get: function (key) {
195 
196         var nativeName =
197                 ComponentStyle.isNativeName(key) ? key :
198                 ComponentHelper.getNativePropName(this, key);
199 
200         if (nativeName) { //尝试作为属性名去获取，如果是属性名，则获取 native 层对应的值
201 
202             var nativeValue = this.getNativeProp(nativeName);
203             
204             var item = this.nativeProps[nativeName]; 
205             if ($.Object.isPlain(item)) { //把 nativeValue 转成 haf 层所使用的 value
206 
207                 return $.Object.findKey(item.values, function (key, value) {
208                     return value === nativeValue;
209                 });
210             }
211 
212             return nativeValue;
213         }
214         
215         return this.config[key];
216     },

232     /**
233     * 调用本组件的 native 方法。
234     * 这是一个受保护的方法，仅供子类使用。
235     */
236     callNative: function (methodName) {
237 
238         var nativeId = this.getNativeId();
239         var args = [nativeId, methodName];
240         
241         if (arguments.length > 1) {
242 
243             var isComponent = ComponentHelper.isValidComponent;
244             var list = $.Array.parse(arguments).slice(1);
245             
246             list = $.Array.keep(list, function (item, index) {
247 
248                 return isComponent(item) ? item.getNativeId() : item;
249             });
250 
251             args = args.concat(list);
252         }
253         
254         UI.invokeComponentMethod.apply(UI, args); // native 的调用方式中的是参数个数可变的
255 
256     },

268     toggle: function (isVisible) {
269 
270         if (isVisible === undefined) {
271             var visible = this.get('visible');
272             if (visible === 'show' || visible === undefined) {
273                 this.hide();
274             }
275             else {
276                 this.show();
277             }
278         }
279         else if (isVisible === true) {
280             this.show();
281         }
282         else if (isVisible === false) {
283             this.hide();
284         }
285         else {
286             throw Haf.error('无法识别参数 isVisible 的值');
287         }
288     },

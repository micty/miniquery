 15     var baseName = 'Haf.Base';

 18     //Haf.define(className, data) 中的 data 中的具有特殊作用的保留字段。 
 19     var reservedFields = [ //为方便维护，请按升序存放。
 20         'abstract',
 21         'config',
 22         'configListeners',
 23         'extend',
 24         'listeners',
 25         'requires',
 26         'singleton',

 28         'xtype'

 31     //一对一的关系
 32     var name$data = {};         //维护 { className: data } 这样的原始集合
 33 
 34     var name$superName = {};    //维护 { className: 直接父类 className } 的关系
 35     var xtype$superXtype = {};  //维护 { xtype: 直接父类 xtype } 的关系
 36     var xtype$name = {};        //维护 { xtype: className } 的关系
 37     var name$class = {};        //维护 { className: constructor} 的关系，对 singleton 的特别有用
 38     var name$config = {};       //维护 { className: 自上而下合并后的 config } 的关系，实现字段继承
 39 
 40     //采用预先计算的技术，避免在需要时再去实时计算取结果。
 41     //因为 y = f(x)，当 x 确定时，y 也随之确定，可以先计算出结果放在 { x: y } 映射关系里维护起来。
 42     var name$superDatas = {};   //维护 { className: [父类A.data, ... 当前类.data] } 的关系，其中 A 是 N 的父类

 44 
 45     //一对多的关系
 46     var xtype$childXtypes = {}; //维护 { xtype: [直接子类xtype] } 的关系
 47     var name$childNames = {};   //维护 { clasName: [直接子类className] } 的关系

 50     /**
 51     * 获取指定类名的所有父类名，返回一个数组。
 52     * 该方法从继承层次上向上搜索所有的父类类名，但不包括根类 Haf.Base。
 53     */
 54     function getSuperNames(className, includingSelf) {
 55         
 56         if (typeof className == 'object') { //此时传进来的是一个实例
 57             className = className.getType();
 58         }
 59 
 60         var a = includingSelf ? [ className ] : [];
 61 
 62         var superName = name$superName[className];
 63         while (superName) {
 64 
 65             a.push(superName);
 66             superName = name$superName[superName];
 67         }
 68 
 69         a.length = a.length - 1; //去掉最后一项 Haf.Base
 70 
 71         return a;
 72     }

 74     /**
 75     * 获取指定 xtype 的所有父类的 xtype，返回一个数组
 76     * 该方法从继承层次上向上搜索所有父类的 xtype，但不包括根类 Haf.Base。
 77     */
 78     function getSuperXtypes(xtype, includingSelf) {
 79 
 80         var className = xtype$name[xtype]; //xtype -> className
 81 
 82         var list = getSuperNames(className, includingSelf);
 83 
 84         return $.Array.map(list, function (name, index) {
 85 
 86             var data = getData(name);
 87             var xtype = data.xtype;
 88 
 89             if (xtype) {
 90                 return xtype;
 91             }
 92 
 93             throw Haf.error('{0} 类的 xtype 不存在。', name);
 94         });
 95     }

 98     /**
 99     * 建立起各种映射关系。
100     * @param {string} className 类的名称
101     * @param {Object} data 类的配置数据
102     */
103     function setData(className, data) {
104 
105         //维护 {xtype: className} 的关系
106         var xtype = data.xtype;
107         if (xtype) {
108             if (xtype in xtype$name) {
109                 throw Haf.error('xtype: {0} 已给 {1} 类占用', xtype, xtype$name[xtype]);
110             }
111             xtype$name[xtype] = className;
112         }
113 
114         var superName = data.extend || baseName;
115 
116         //维护 {className: data} 这样的原始集合
117         data.extend = superName;
118         name$data[className] = data;
119 
120         //维护 {className: config} 的集合
121         var superConfig = name$config[superName]; //父类的 config
122         name$config[className] = $.Object.extend({}, superConfig, data.config);
123 
124 
125         //维护 {className: superClassName} 的关系
126         name$superName[className] = superName;
127 
128 
129         var superDatas = name$superDatas[superName] || [];
130         superDatas.push(data);
131         name$superDatas[className] = superDatas;
132 
133 
134 
135         //把当前类的 className 存入 name$childNames 对应的父类数组中
136         var childNames = name$childNames[superName] || [];
137         childNames.push(className);
138         name$childNames[superName] = childNames;
139 
140         if (xtype) { //创建类时指定了 xtype
141 
142             //找到父类的 xtype
143             var superXtype = superName == baseName ?
144                 baseXtype : 
145                 getData(superName).xtype;
146 
147             xtype$superXtype[xtype] = superXtype;
148 
149             //找到父类所关联的子 xtype 数组，并把当前的 xtype 添加进去。
150             var childXtypes = xtype$childXtypes[superXtype] || [];
151             childXtypes.push(xtype);
152             xtype$childXtypes[superXtype] = childXtypes;
153             
154         }
155 
156         //给每个方法分配一个 guid 和元数据。
157         //这会对方法产生一个字段的污染，但不影响使用。
158         $.Object.each(data, function (key, fn) {
159             if (typeof fn != 'function') {
160                 return;
161             }
162 
163             var guid = $.String.format('{0}.{1}', className, key);
164 
165             Meta.setGuid(fn, guid);
166 
167             //设置元数据
168             Meta.set(fn, {
169                 'className': className,
170                 'methodName': key
171             }); 
172         });
173     }

175     /**
176     * 获取指定类的类定义数据。
177     * 该方法可以获取 Haf.define(className, data) 中的 data 对象。
178     * @param {string} className 要获取的类的名称。
179     * @param {number} [upLevel=0] 相对于 className 所在的类(当前类)，沿着继承层次向上的层数。
180         当前类为 0；当前类的父类为 1；当前类的父类的父类为 2；依次类推。
181     * @return {Object} 返回类定义时的数据对象。
182         如果不存在该类对应的定义数据，则返回一个默认的对象。
183     */
184     function getData(className, upLevel) {
185 
186         //传进来的是一个实例，则默认取其 className 元数据
187         if (typeof className == 'object') {
188             className = Meta.get(className, 'className');
189         }
190 
191 
192         upLevel = upLevel || 0; //向上追着
193 
194         while (upLevel > 0) { //向上追溯的层次
195             className = name$superName[className];
196             upLevel--;
197         }
198 
199         return name$data[className] || {
200             extend: baseName,
201             requires: [],
202             config: {},
203             statics: {}
204         };
205     }

207     /**
208     * 获取指定 xtype 所对应的类定义数据。
209     * 该方法可以获取 Haf.define(className, data) 中的 data 对象。
210     * @param {string} xtype 要获取的类的 xtype。
211     * @param {number} [upLevel=0] 相对于 xtype 所在的类(当前类)，沿着继承层次向上的层数。
212         当前类为 0；当前类的父类为 1；当前类的父类的父类为 2；依次类推。
213     * @return {Object} 返回类定义时的数据对象。
214         如果不存在该类对应的定义数据，则返回一个默认的对象。
215     */
216     function getDataByXtype(xtype, upLevel) {
217 
218         var className = xtype$name[xtype]; //xtype -> className
219 
220         if (!className) {
221             throw Haf.error('不存在 xtype 为 {0} 的类', xtype);
222         }
223 
224         return getData(className, upLevel);
225     }

237     /**
238     * 通过类名获取该类的构造器。
239     * 该类必须是通过 Haf.define 的方式定义实现的。
240     * 当不存在该类，则尝试用同步方式动态加载并返回该类的构造器。
241     * @param {string} className 要加载的类的完全名称。
242     * @param {int} [upLevel=0] 向上追溯的层次。
243         0: 表示当前类; 
244         1: 表示父类;
245         2: 表示父类的父类;
246         依次类推
247     * @return {Function} 返回该类的构造器。
248     */
249     function getClass(className, upLevel, autoLoad) {
250 
251         switch (typeof upLevel) {
252 
253             case 'undefined': //此时为  getClass(className)
254                 upLevel = 0; //向上追溯
255                 autoLoad = true;
256                 break;
257 
258             case 'number':  //此时为 getClass(className, upLevel) 或 getClass(className, upLevel, autoLoad)
259                 if (autoLoad === undefined) { //此时为 getClass(className, upLevel)
260                     autoLoad = true;
261                 }
262                 break;
263 
264             case 'boolean': //此时为 getClass(className, autoLoad)
265                 autoLoad = upLevel;
266                 upLevel = 0;
267                 break;
268 
269             default:
270                 throw Haf.error('无法识别参数 upLevel 的类型');
271         }
272 
273         while (upLevel > 0) { //向上追溯的层次
274             className = name$superName[className];
275             upLevel--;
276         }
277 
278         // 这里用原始的，可以获取到 singleton 的构造器。
279         // 而 $.Object.namespace(global, className) 会获取到 singleton 的实例，而不是构造器。
280         // 另外，Haf.Base 不在 name$class 的列表中，而 $.Object.namespace(global, className) 是可以获取到 Haf.Base 的。
281         var F = name$class[className] || $.Object.namespace(global, className);
282         if (!F && autoLoad) {
283 
284             Loader.loadClass(className);
285             F = name$class[className] || $.Object.namespace(global, className);
286         }
287 
288         return F;
289     }

291     function isSingleton(className) {
292 
293         var data = name$data[className];
294         if (!data) {
295             return false;
296         }
297 
298         return data.singleton === true;

302     /**
303     * 根据配置数据去创建当前类的 prototype 实例
304     */
305     function createPrototype($data) {
306 
307         var className = $data.extend;   //父类的类名称
308 
309         var prototype;
310         if (className == 'Haf.Base') {
311             prototype = new Haf.Base();
312         }
313         else {
314             var data = getData(className);  //父类的配置数据
315             var F = getClass(className);    //父类的构造器，会去同步地动态加载(如果还没有)
316             prototype = new F(data.config, true);
317         }
318 
319         //移除保留的成员，只保留用户自定义的成员。
320         var props = $.Object.remove($data, reservedFields);
321 
322         //把当前类的其他成员拷到原型对象上，从而实现继续和重写(同名的覆盖)
323         $.Object.extend(prototype, props);
324 
325         return prototype;
326     }

329     /**
330     * 给指定实例的 config 中的成员绑定事件处理函数。
331     */
332     function bindConfigListeners(self, listeners) {
333         
334         //这里只对 listeners 作迭代，而不是去迭代 config。
335         //因为 listeners 包含的成员是 config 的成员的一个子集，从而可以减少迭代次数。
336         $.Object.each(listeners, function (key, item) {
337 
338             if ($.Object.isArray(item)) { //优化 #3-0
339                 $.Array.each(item, function (item, index) {
340                     bind(item);
341                 });
342             }
343             else {
344                 bind(item);
345             }
346 
347             //内部的公用函数
348             function bind(item) {
349 
350                 if (typeof item == 'function') {
351 
352                     self.onchange(key, item);
353                 }
354                 else if ($.Object.isPlain(item)) {
355 
356                     var when = item.when;
357                     if (when == 'before') {
358                         self.onbeforechange(key, item.fn);
359                     }
360                     else if (when == 'after') {
361                         self.onchange(key, item.fn);
362                     }
363                     else {
364                         throw new Error('未识别的 when 字段: ' + when);
365                     }
366                 }
367                 else {
368                     throw new Error('未识别的配置成员值: ', +key);
369                 }
370             }
371             
372         });
373     }

377     /**
378     * 判断指定的对象是否为指定类的实例。
379     * 该方法会检测指定类及原型链上的类。
380     * @param {Object} obj 要进行检测的对象。
381     */
382     function isInstanceOf(obj, className) {
383 
384         if ( !obj || typeof obj != 'object' ) {
385             return false;
386         }
387 
388         var type = typeof className;
389 
390         //构造器
391         var F = type == 'string' ? getClass(className) :
392             type == 'function' ? className : null;
393 
394         if (!F) {
395             return false;
396         }
397 
398         return obj instanceof F;
399     }

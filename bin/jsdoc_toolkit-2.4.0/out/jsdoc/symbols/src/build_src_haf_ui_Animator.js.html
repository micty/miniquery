  9 //不透明度
 10 var Opacity = (function () {
 11 
 12     function parse(config) {
 13 
 14         var opacity = config.opacity;
 15         var type = typeof opacity;
 16 
 17         switch (type) {
 18 
 19             case 'string':
 20                 opacity = $.Math.parsePercent(opacity);
 21                 if (isNaN(opacity)) {
 22                     throw Haf.error('无法识别 opacity 的值');
 23                 }
 24             case 'number':
 25                 opacity = [1, opacity];
 26                 break;
 27         }
 28 
 29         return $.Object.isArray(opacity) ? {
 30             type: 3,
 31             fromAlpha: opacity[0],
 32             toAlpha: opacity[1]
 33         } : null;
 34            
 35     }
 36 
 37     return {
 38         parse: parse
 39     };
 40 
 41 })(); //结束 Opacity

 44 //缩放
 45 var Scale = (function () {
 46 
 47     function getPair(v) {
 48 
 49         var type = typeof v;
 50         if (type == 'number') {
 51             return [v, v];
 52         }
 53 
 54         if (type == 'string') {
 55             v = $.Math.parsePercent(v);
 56             if (isNaN(v)) {
 57                 throw Haf.error('无法识别参数 v 的值: {0}', v);
 58             }
 59             return [v, v];
 60         }
 61 
 62         if ($.Object.isArray(v)) {
 63             return v;
 64         }
 65 
 66     }
 67 
 68 
 69     function parse(config) {
 70 
 71         var scale = config.scale;
 72         if (scale === undefined) {
 73             return null;
 74         }
 75 
 76 
 77         var from;
 78         var to;
 79         var dest;
 80 
 81         if ($.Object.isPlain(scale)) {
 82             from = scale.from;
 83             to = scale.to;
 84             dest = scale.dest;
 85         }
 86         else {
 87             to = scale;
 88         }
 89 
 90         to = getPair(to);
 91         if (!to) {
 92             return null;
 93         }
 94 
 95         from = getPair(from) || [1, 1];
 96         dest = getPair(dest) || [0, 5, 0.5];
 97 
 98         return {
 99             type: 4,
100 
101             fromXScale: from[0],
102             fromYScale: from[1],
103 
104             toXScale: to[0],
105             toYScale: to[1],
106 
107             destinationX: dest[0],
108             destinationY: dest[1]
109         };
110     }
111 
112     return {
113         parse: parse
114     };
115 })(); //结束 Scale

185 //旋转
186 var Rotate = (function () {
187 
188 
189     function getDegree(degree) {
190 
191         var type = typeof degree;
192         if (type == 'number') {
193             return [0, degree];
194         }
195 
196         if (type == 'string') {
197             var circles = $.Math.parsePercent(degree);
198             if (isNaN(circles)) {
199                 throw Haf.error('无法识别参数 degree 的值: {0}', degree);
200             }
201 
202             degree = 360 * circles;
203             return [0, degree];
204         }
205 
206         if ($.Object.isArray(degree)) {
207             return degree;
208         }
209 
210         throw Haf.error('无法识别参数 degree 的类型');
211 
212     }
213 
214     function getCenter(center) {
215 
216         if (center == null ) { //undefined 或 null
217             return [0.5, 0.5];
218         }
219 
220         var type = typeof center;
221         if (type == 'number') {
222             return [center, center];
223         }
224 
225         if (type == 'string') {
226             center = $.Math.parsePercent(center);
227             if (isNaN(center)) {
228                 throw Haf.error('无法识别参数 center 的值: {0}', center);
229             }
230             return [center, center];
231         }
232 
233         if ($.Object.isArray(center)) {
234             return center;
235         }
236 
237         throw Haf.error('无法识别参数 center 的类型');
238     }
239 
240 
241     function parse(config) {
242 
243         var rotate = config.rotate;
244         if (rotate === undefined) {
245             return null;
246         }
247 
248         var degree;
249         var center;
250 
251         if ($.Object.isPlain(rotate)) {
252             degree = rotate.degree;
253             center = rotate.center;
254         }
255         else {
256             degree = rotate;
257         }
258 
259         degree = getDegree(degree);
260         center = getCenter(center);
261 
262         return {
263             type: 2,
264             fromDegrees: degree[0],
265             toDegrees: degree[1],
266             centerX: center[0],
267             centerY: center[1]
268         };
269 
270     }
271 
272     return {
273         parse: parse
274     };
275 
276 })();

292         duration = duration == 'slow' ? 3000 :
293             duration == 'normal' ? 2000 :
294             duration == 'fast' ? 1000 :
295             duration === undefined ? 2000 : duration;
296 
297         var delay = config.delay || 0;
298 
299         return {
300             duration: duration,
301             startOffSet: delay,
302             animations: $.Array.trim(list)
303         };

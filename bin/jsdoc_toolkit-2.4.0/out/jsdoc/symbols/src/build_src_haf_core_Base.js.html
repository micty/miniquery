 49     /**
 50     * 设置 config 对象中指定成员的值。
 51     * 该方法针对 config 对象中的成员的 setter 方法。
 52     * 会触发成员的 onbeforechange 和 onchange 事件(除非在调用时指定了安静模式)。
 53     * @param {string|Object} fieldName 要设置的成员的名称。
 54         当要进行批量设置时，可传入一个 key-value 的对象。
 55     * @param [newValue] 要设置的成员的值。
 56     * @param {boolean} [isQuiet=false] 是否采用安静模式。
 57         若指定为 true，则表示采用安静模式，不触发任何事件。
 58     */
 59     set: function (fieldName, newValue, isQuiet) {
 60 
 61         var isBatch = typeof fieldName == 'object';
 62         if (isBatch) { //此时为 set({}, isQuiet);
 63             isQuiet = newValue; //参数修正
 64             newValue = undefined;
 65         }
 66 
 67         var config = this.config;
 68 
 69         if (isQuiet === true) { //安静模式，不触发事件。
 70             if (isBatch) {
 71                 $.Object.extend(config, fieldName);
 72             }
 73             else {
 74                 config[fieldName] = newValue;
 75             }
 76             return;
 77         }
 78 
 79 
 80         var self = this;
 81         var hasAnyChange = false; //指示是否有字段的值发生了改变
 82         var oldConfig = $.Object.extend({}, config); //设置前，拷贝一份作为旧的
 83 
 84         if (isBatch) { // 此时为 set({}, isQuiet);
 85             $.Object.each(fieldName, set);
 86         }
 87         else { //单个的情况
 88             set(fieldName, newValue);
 89         }
 90 
 91         if (hasAnyChange) { //最后触发总事件
 92             self.trigger('change:', [config, oldConfig]); //全部字段
 93         }
 94         
 95 
 96         //一个共用的内部方法：只针对单项，并且会触发事件。
 97         function set(fieldName, newValue) {
 98 
 99             var oldValue = self.get(fieldName);
100             var finalValue = newValue;
101 
102             var beforeEventName = 'beforechange:' + fieldName;
103 
104             if ($.Event.has(self, beforeEventName)) { //已绑定了 beforechange 事件
105 
106                 //最后一个处理函数的返回值就是最终的返回值
107                 finalValue = self.trigger(beforeEventName, [newValue, oldValue]);
108 
109                 //事件处理器返回 undefined，则使用回原来的值
110                 if (finalValue === undefined) {
111                     finalValue = oldValue;
112                 }
113             }
114 
115             //只有有一个字段发生了变化，hasAnyChange 就一直保持为 true。
116             hasAnyChange = hasAnyChange || (finalValue !== oldValue); 
117 
118             //值发生改变
119             if (finalValue !== oldValue) { //这里不能用 hasAnyChange，因它可能是上个字段的结果。
120                 config[fieldName] = finalValue;
121                 self.trigger('change:' + fieldName, [finalValue, oldValue]); //指定字段
122             }
123         }

128     /**
129     * 调用父类的指定方法，参数传递形式为类似 call。
130     * 只能由具有继承关系的实例方法去调用，而且调用方式为 this.callSuper(arguments)
131     */
132     callSuper: function ($arguments) {
133 
134         var caller = $arguments.callee;
135         var className = Meta.get(caller, 'className');
136         //if (!className || !ClassManager.isInstanceOf(this, className)) {
137         //    throw Haf.error('callSuper 必须在同一个类体系中的实例方法内被调用');
138         //}
139         
140         //下面的逻辑才是真正的主流程
141         var SuperClass = ClassManager.getClass(className, 1); //取得父类的构造函数
142         
143         var methodName = Meta.get(caller, 'methodName');
144 
145         //这里一定要从原型上取得方法引用，因为这样可以使用继承特性。
146         //不能从原始的 data 中取，因为该 data 是孤立的，它可能不包含该方法。
147         var fn = SuperClass.prototype[methodName]; //可能是 SuperClass 自己的方法，也可能是继承下来的。
148 
149         if (!fn) {
150             throw Haf.error('父类中不存在名为 {0} 的方法', methodName);
151             return;
152         }
153 
154         //var args = $.Array.parse(arguments).slice(1);
155         var args = Array.prototype.slice.call(arguments, 1);
156         return fn.apply(this, args);
157         
158     },

160     /**
161     * 给 config 对象中的成员绑定一个 beforechange 事件。
162     * 该事件会在 config 对象中的成员的值更新前触发，
163     * 事件处理函数必须返回一个值作为该成员最终的值，否则继续使用原来的新值。
164     */
165     onbeforechange: function (key, fn) {
166 
167         var self = this;
168         var $Event = $.Event;
169 
170         if ($.Object.isPlain(key)) { //针对 { a: fnA, b: fnB } 批量设置的情况
171 
172             var maps = key; //换个名称更容易理解
173 
174             $.Object.each(maps, function (key, fn) {
175                 $Event.bind(self, 'beforechange:' + key, fn);
176             });
177         }
178         else {
179             $Event.bind(self, 'beforechange:' + key, fn);
180         }
181 
182     },

184     /**
185     * 给 config 对象中的成员绑定一个 change 事件。
186     * 该事件会在 config 对象中的成员的值更新后触发，以前通知值已更新。
187     */
188     onchange: function (key, fn) {
189 
190         var self = this;
191 
192         if ($.Object.isPlain(key)) { //重截 { a: fnA, b: fnB } 批量设置的情况
193 
194             var maps = key; //换个名称更容易理解
195 
196             $.Object.each(maps, function (key, fn) {
197                 $.Event.bind(self, 'change:' + key, fn);
198             });
199         }
200         else if (typeof key == 'function') { //重载 onchange(fn) 的情况
201 
202             fn = key;
203             $.Event.bind(self, 'change:', fn);
204         }
205         else if ($.Object.isArray(key)) {
206 
207             var keys = $.Array.toObject(keys, keys); // 建立起 {a:'a', b:'b'} 这样的关系，可避免搜索数组
208             $.Event.bind(self, 'change:', function (newConfig, oldConfig) {
209                 $.Object.each(newConfig, function (key, value) {
210                     if (key in keys) {
211                         fn.call(self, newConfig, oldConfig);
212                         return false; //break;
213                     }
214                 });
215             });
216         }
217         else {
218             $.Event.bind(self, 'change:' + key, fn);
219         }
220 
221     },

223     /**
224     * 给本实例绑定事件处理函数。
225     * @param {Object|string} eventName 要绑定的事件名称。
226         如果为 Object，则表示批量绑定。
227     * @param {function} handler 事件处理函数，函数内部的 this 指向本实例。
228     * @param {boolean} [isOnce=false] 指示是否只执行一次事件处理函数，
229         如果指定为 true，则执行事件处理函数后会将该处理函数从事件列表中移除。
230     */
231     on: function (eventName, handler, isOnce) {
232 
233         var self = this;
234 
235         if (typeof eventName == 'object') { //此时为 on( {...}, isOnce );
236 
237             var listeners = eventName;
238             isOnce = handler;
239             handler = undefined;
240 
241             $.Object.each(listeners, function (key, value) {
242 
243                 if (typeof value == 'function') { //此时为 { eventName: fn }
244                     self.bind(key, value, isOnce);
245                 }
246                 else if (typeof value == 'object') { //此时为 { eventName: {...} }
247                     var fn = value.fn;
248                     if (typeof fn != 'function') {
249                         throw Haf.error('成员 {0} 的值为一个对象时，必须指定该对象中的 fn 为一个函数', key);
250                     }
251                     var single = 'single' in value ? value.single : isOnce;
252                     self.bind(key, fn, single);
253                 }
254                 else {
255                     throw Haf.error('无法识别 key 为 {0} 的所对应的值', key);
256                 }
257             });
258         }
259         else {
260             self.bind(eventName, handler, isOnce);
261         }

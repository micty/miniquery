  4 /**
  5 * 组件管理器，在实例化/销毁组件时，需要在这里进行注册/取消注册。
  6 * 支持根据 selector 查找组件。
  7 */
  8 var ComponentManager = (function ($, Haf, ClassManager) {
  9 
 10     var id$component = {},
 11 
 12         xtype$id$component = {},
 13 
 14         //判断mode的正则，mode有两种：>和空格
 15         modeRe = /^(\s?([>\^])\s?|\s|$)/,
 16 
 17         //选择符表达式中分离单个选择符的正则
 18         tokenRe = /^(#)?([\w\-]+|\*)/;
 19 
 20     //Matches a token with possibly (true|false) appended for the "shallow" parameter
 21     //这是原st的正则
 22     //tokenRe = /^(#)?([\w\-]+|\*)(?:\((true|false)\))?/,
 23 
 24     /**
 25      * 从items中过滤出xtype为指定参数的组件实例数组
 26      */
 27     function filterByXType(items, xtype) {
 28         var result = [],
 29             i = 0,
 30             length = items.length,
 31             candidate;
 32         for (; i < length; i++) {
 33             candidate = items[i];
 34             if (candidate.xtype == xtype) {
 35                 result.push(candidate);
 36             }
 37         }
 38         return result;
 39     }
 40     /**
 41      * 从items中过滤出id为指定参数的组件实例数组
 42      */
 43     function filterById(items, id) {
 44         var result = [],
 45             i = 0,
 46             length = items.length,
 47             candidate;
 48         for (; i < length; i++) {
 49             candidate = items[i];
 50             if (candidate.get('id') === id) {
 51                 result.push(candidate);
 52             }
 53         }
 54         return result;
 55     }
 56     /**
 57      * 获取容器container的下级组件实例，deep为false表示只获取一级，true表示获取多级
 58      */
 59     function getChildren(container, deep) {
 60         var result = [],
 61             children = container.getItems ? container.getItems() : null,
 62             child, i;
 63         if (children && children.length > 0) {
 64             result = result.concat(children);
 65             if (deep) {
 66                 for (i = 0; i < children.length; i++) {
 67                     child = children[i];
 68                     result = result.concat(getChildren(child, true));
 69                 }
 70             }
 71         }
 72         return result;
 73     }
 74     /**
 75      * 获取items的下级组件实例，mode为>表示直接下级，其他表示所有下级
 76      */
 77     function getItems(items, mode) {
 78         var result = [],
 79             i = 0,
 80             length = items.length,
 81             candidate,
 82             deep = mode !== '>';
 83 
 84         for (; i < length; i++) {
 85             candidate = items[i];
 86             result = result.concat(getChildren(candidate, deep));
 87         }
 88         return result;
 89     }
 90     /**
 91      * 解析selector，返回解析结果，对于"queryformview #fieldset1 textfield"
 92      * 这样的选择符，解析后的结果为：
 93      * [
 94      *      {
 95      *          method:filterByXType,
 96      *          args:['queryformview']
 97      *      },
 98      *      {
 99      *          mode:' '
100      *      },
101      *      {
102      *          method:filterById,
103      *          args:['fieldset1']
104      *      },
105      *      {
106      *          mode:' '
107      *      },
108      *      {
109      *          method:filterByXType,
110      *          args:['textfield']
111      *      }
112      * ]
113      */
114     function parseSelector(selector) {
115         var operations = [],
116             lastSelector,
117             tokenMatch,
118             matchedChar,
119             modeMatch;
120 
121         // We are going to parse the beginning of the selector over and
122         // over again, slicing off the selector any portions we converted into an
123         // operation, until it is an empty string.
124         while (selector && lastSelector !== selector) {
125             lastSelector = selector;
126 
127             // First we check if we are dealing with a token like #, * or an xtype
128             tokenMatch = selector.match(tokenRe);
129 
130             if (tokenMatch) {
131                 matchedChar = tokenMatch[1];
132 
133                 // If the token is prefixed with a # we push a filterById operation to our stack
134                 if (matchedChar === '#') {
135                     operations.push({
136                         method: filterById,
137                         args: [Haf.String.trim(tokenMatch[2])]
138                     });
139                 }
140                     // If the token is a * or an xtype string, we push a filterByXType
141                     // operation to the stack.
142                 else {
143                     operations.push({
144                         method: filterByXType,
145                         args: [Haf.String.trim(tokenMatch[2])]
146                     });
147                 }
148 
149                 // Now we slice of the part we just converted into an operation
150                 selector = selector.replace(tokenMatch[0], '');
151             }
152             modeMatch = selector.match(modeRe);
153 
154             // Now we are going to check for a mode change. This means a space
155             // or a > to determine if we are going to select all the children
156             // of the currently matched items, or a ^ if we are going to use the
157             // ownerCt axis as the candidate source.
158             if (modeMatch && modeMatch[1]) { // Assignment, and test for truthiness!
159                 operations.push({
160                     mode: modeMatch[2] || modeMatch[1]
161                 });
162                 selector = selector.replace(modeMatch[0], '');
163             }
164         }
165         return operations;
166     }
167     /**
168      * 对items,执行operations操作，返回符合条件的结果
169      */
170     function executeOperations(items, operations) {
171         var i = 0,
172             length = operations.length,
173             operation,
174             workingItems = items;
175 
176         // We are going to loop over our operations and take care of them
177         // one by one.
178         for (; i < length; i++) {
179             operation = operations[i];
180 
181             // The mode operation requires some custom handling.
182             // All other operations essentially filter down our current
183             // working items, while mode replaces our current working
184             // items by getting children from each one of our current
185             // working items. The type of mode determines the type of
186             // children we get. (e.g. > only gets direct children)
187             if (operation.mode) {
188                 workingItems = getItems(workingItems, operation.mode);
189             } else {
190                 workingItems = operation.method.apply(this, [workingItems].concat(operation.args));
191             }
192 
193             // If this is the last operation, it means our current working
194             // items are the final matched items. Thus return them!
195             if (i === length - 1) {
196                 return workingItems;
197             }
198         }
199         return [];
200     }

206     return {
207         /**
208          * 注册组件，在每个组件create时(Haf.view.Component.initialize时)，会在这里注册
209          * @param container
210          * @param component
211          */
212         register: function (component) {
213             var id = component.get('id');
214             var xtype = component.xtype;
215 
216             if (id$component[id]) {
217                 throw Haf.error('该 id 值已给占用: {0}', id);
218             }
219 
220             id$component[id] = component;
221 
222             var list = xtype$id$component[xtype];
223             if (!list) {
224                 list = xtype$id$component[xtype] = {};
225             }
226 
227             list[id] = component;
228         },
229         /**
230          * 取消注册，在组件销毁时，要进行取消注册
231          * @param component
232          */
233         unregister: function (component) {
234             var id = component.get('id');
235             var xtype = component.xtype;
236             delete id$component[id];
237             delete xtype$id$component[xtype][id];
238         },
239         /**
240          * 根据组件id获取组件实例，类似Ext.getCmp
241          * @param compId
242          * @return {*}
243          */
244         getCmp: function (compId) {
245             return id$component[compId];
246         },
247 
248         get: function (componentId) {
249             return id$component[componentId];
250         },
251 
252         /**
253          * 根据组件xtype返回该组件创建的所有实例，返回值中以id为key，实例为value
254          * 可以用于调试组件有多个实例
255          * @param xtype
256          * @return {*}
257          */
258         getCmpsByXtype: function (xtype) {
259             return xtype$id$component[xtype];
260         },
261         /**
262          * 在container下按照selector选择器进行选择，如果没有置顶container，
263          * 则在viewport(组件书的根)下查找
264          * @param root
265          * @param selector
266          * selector支持的选择符：
267          * 1.xtype选择符
268          * 2.id选择符
269          *
270          * 支持直接下级选择符(>)间接下级选择符(空格),例如：selector1 selector2
271          */
272         query: function (selector, container) {
273             var root = container || Haf.Viewport,
274                 deepItems = getChildren(root, true);
275             operations = parseSelector(selector, container);
276             return executeOperations(deepItems, operations);
277         },
278 
279         /**
280         * 规格化指定的对象，并返回等价的组件实例。
281         * @param {Object|Component|string} component 要进行规格化的对象。 
282             如果是一个纯对象，则会根据其 xtype 值创建组件实例，并返回它的引用；
283             如果是一个组件实例，则不做处理，直接返回该参数；
284             如果是一个字符串，则当成组件的 id 进行查找，然后返回该组件实例的引用。
285         */
286         normalize: function (value) {
287 
288             if ($.Object.isPlain(value)) { //纯对象 {}
289                 return Haf.create(value);
290             }
291 
292             if ($.Object.isNonNull(value)) { //非空的对象
293                 return ComponentHelper.isValidComponent( value ) ? value : null;
294             }
295 
296             if ($.Object.isString(value)) { //字符串 id
297                 return ComponentManager.get(value);
298             }
299 
300             throw Haf.error('无法识别参数 value');
301         }
302 
303         
304     };

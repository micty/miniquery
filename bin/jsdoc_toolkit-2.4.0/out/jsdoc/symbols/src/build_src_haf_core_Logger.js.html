  4 /**
  5 * 日志类
  6 * @namespace
  7 */
  8 var Logger = (function ($, console) {
  9 
 10     if (!console) {
 11         console = {
 12             log: function (s) {
 13 
 14             }
 15         };
 16     }
 17 
 18 
 19     var DEBUG = 1,
 20         INFO = 2,
 21         ERROR = 4,
 22         logLevel = 1;
 23 
 24     /**
 25     * @inner
 26     */
 27     function log($arguments) {
 28 
 29         var formater = $arguments[0];
 30         var args = Array.prototype.slice.call($arguments, 1);
 31         var msg = $.String.format(formater, args);
 32 
 33         console.log(msg);
 34     }
 35 
 36     return {
 37 
 38         /**
 39         * debug用于记录最详细的日志，主要用于开发期调试，是给开发人员看的。一般在产品发布时，不会输出
 40         * 调用前，需要首先调用isDebug判断一下
 41         * @param msg
 42         */
 43         debug: function (msg) {
 44             if (logLevel <= DEBUG) {
 45                 log(arguments);
 46             }
 47         },
 48 
 49         /**
 50         * info用于记录程序逻辑的大致走向，带有一定的业务含义，产品发布后，需要输出
 51         * @param msg
 52         */
 53         info: function (msg) {
 54             if (logLevel <= INFO) {
 55                 log(arguments);
 56             }
 57         },
 58 
 59         /**
 60         * error记录发生异常时的信息，必须记录
 61         * @param msg
 62         */
 63         error: function (msg) {
 64 
 65             log(arguments);
 66         },
 67 
 68 
 69         /**
 70         * 设置日志级别
 71         * @param level
 72         * 1:输出debug,info,error三个方法的日志
 73         * 2:输入info,error的日志
 74         * 4:只输入error日志
 75         */
 76         setLevel: function (level) {
 77             logLevel = level;
 78         },
 79 
 80         /**
 81         * 是否需要输出debug日志，增加此方法的目的是减少debug(msg)中msg的计算
 82         * @return {Boolean}
 83         */
 84         isDebug: function () {
 85             return logLevel <= DEBUG;
 86         }
 87     };
 88 
 89 
 90 })(Haf, global.console);

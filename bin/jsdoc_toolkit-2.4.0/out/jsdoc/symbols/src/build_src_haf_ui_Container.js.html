  9 /**
 10  * 容器类。
 11  * 所有容器类组件的基类，无法直接实例化。
 12  */
 13 Haf.define('Haf.view.Container', {
 14     extend: 'Haf.view.Component',
 15     xtype: 'container',
 16     'abstract': true,
 17 
 18     /**与 native 层对应的 config 对象，仅用于展示数据，不会实际影响 native 层*/
 19     nativeConfig: {
 20         id: 0,
 21         xtype: '',
 22         props: {},
 23         style: {},
 24         childs: []
 25     },
 26 
 27     config: {
 28         defaults: {},
 29         items: []
 30     },
 31 
 32 
 33     initialize: function (config, noCallSuper) {
 34 
 35         if (noCallSuper !== true) { //为 true 时，不调用 callSuper，主要是针对 Viewport
 36             this.callSuper(arguments, config);
 37         }
 38 
 39 
 40         //收集子组件的实例，这里不要写到类的定义中，因为那是写到原型上的。
 41         //为了私有数据的保护，不写在 this.items 中，而是当成元数据。
 42         Meta.set(this, 'items', []);
 43 
 44         this.nativeConfig.childs = []; //分配个数组
 45 
 46         var items = this.get('items');
 47         if (items) { //创建子组件
 48             this.add(items);
 49         }
 50 
 51     },
 52 
 53     /**
 54     * 获取所有的子组件，返回一个新的数组。
 55     * 该接口是提供给外部使用的，本类内部请使用 Meta.get(this, 'items')
 56     */
 57     getItems: function () {
 58         var items = Meta.get(this, 'items');
 59         return items.slice(0); //拷贝一份数组，避免使用者修改原 items 的元素。
 60     },
 61 
 62     /**
 63     * 获取当前容器的直接下级子组件的个数。
 64     */
 65     getCount: function () {
 66         var items = Meta.get(this, 'items');
 67         return items.length;
 68     },
 69 
 70     /**
 71     * 向容器内添加一个或多个子组件。
 72     * @param {Component|Object|Array} item 要添加的子组件。
 73         可以是一个组件实例，也可以是一个纯对象，也可以是一个数组。
 74     */
 75     add: function (item) {
 76 
 77         var self = this;
 78 
 79         if ($.Object.isArray(item)) {
 80             return $.Array.map(item, function (item, index) {
 81                 return self.addItem(item) || null;
 82             });
 83         }
 84 
 85         return this.addItem(item);
 86     },
 87 
 88     /**
 89     * 向容器内添加一个子组件。
 90     * 该方法只针对单项，不重载批量的情况。
 91     * 这是一个受保护的模板方法，仅供当前类和子类调用，子类可重写以提供自己的实现。
 92     * 开发者应该调用 add 方法。
 93     * @param {Component|Object|string} item 要添加的子组件。
 94         可以是一个组件实例，也可以是一个纯对象，也可以是组件实例的 id 值。
 95     * @param {function} [fnNativeAdd=Hae.UI.addComponent2Container] 向容器添加子组件的 native 方法。
 96         当子类需要调用自己的 native 方法时，可以提供该参数。
 97         该参数方法会接受到两个参数：item 的 nativeId 和 当前容器组件的 nativeId。
 98     * @return 返回被添加后的子组件实例。
 99     */
100     addItem: function (item, fnNativeAdd) {
101 
102         var defaults = this.get('defaults');
103         var isInstance = true; //指示 item 是否已给是组件实例。 先假设是。
104 
105         if ($.Object.isPlain(item)) {
106             isInstance = false;
107             item = $.Object.extend({}, defaults, item);
108         }
109 
110         item = ComponentManager.normalize(item);
111 
112         if (!item) {
113             throw Haf.error('参数 item 不是(或不能从它创建出)一个有效的组件。');
114         }
115 
116         if (this.contains(item)) {
117             throw Haf.error('参数 item 所对应的子组件已在容器中，不能重复添加。');
118         }
119 
120         if (isInstance) { //已是组件实例，则更新 UI
121             defaults = $.Object.remove(defaults, ['xtype', 'id']);
122             item.updateUI(defaults);
123         }
124 
125         var items = Meta.get(this, 'items');
126         items.push(item);
127 
128         Meta.set(item, 'parent', this); //设置 item 的 parent 
129 
130         var childs = this.nativeConfig.childs;
131         childs.push(item.nativeConfig);
132 
133 
134         var itemId = item.getNativeId();
135         var containerId = this.getNativeId();
136 
137         if (fnNativeAdd) {
138             fnNativeAdd.call(this, itemId, containerId);
139         }
140         else {
141             //注意，不要用一个变量缓存 Hae.UI.addComponent2Container，
142             //因为它的 this 指向 Hae.UI，它不是一个真正的静态方法。
143             Hae.UI.addComponent2Container(itemId, containerId);
144         }
145 
146         //Logger.debug('容器添加子组件: {0} + {1}', this.getGuid(), item.getGuid());
147 
148 
149         return item;
150     },
151 
152     /**
153     * 从容器中移除一个子组件。
154     * @param {Component|string|number} item 要移除的子组件。
155     *   可以是一个组件实例，或一个组件的　id 字符串，或子组件中容器中的索引位置。
156     * @param {function} [fnNativeRemove=Hae.UI.removeComponent4Container] 向容器添加子组件的 native 方法。
157         当子类需要调用自己的 native 方法移除子组件时，应该重定本方法，可以提供该参数。
158         该参数方法会接受到两个参数：item 的 nativeId 和 当前容器组件的 nativeId。
159     */
160     remove: function (item, fnNativeRemove) {
161 
162         item = this.getItem(item);
163 
164         if (!item) {
165             throw Haf.error('容器中不存在该组件');
166         }
167 
168 
169         var items = Meta.get(this, 'items');
170         items = $.Array.remove(items, item);  //从集合中移除
171         Meta.set(this, 'items', items);
172 
173 
174         var childs = this.nativeConfig.childs;
175         this.nativeConfig.childs = $.Array.remove(childs, item.nativeConfig);
176 
177 
178 
179         var itemId = item.getNativeId();
180         var containerId = this.getNativeId();
181 
182         if (fnNativeRemove) {
183             fnNativeRemove.call(this, itemId, containerId);
184         }
185         else {
186             Hae.UI.removeComponent4Container(itemId, containerId);
187         }
188 
189         //Logger.debug('容器移除子组件: {0} - {1}', containerId, itemId);
190 
191         return item;
192 
193     },
194 
195     /**
196     * 移除所有子组件
197     */
198     removeAll: function () {
199 
200         var self = this;
201         var items = Meta.get(this, 'items');
202 
203         $.Array.each(items, function (item, index) {
204 
205             self.remove(item);
206             
207         });
208     },
209 
210     push: function (item) {
211         return this.add(item);
212     },
213 
214     pop: function () {
215 
216         var count = this.getCount();
217         if (count > 0) {
218             return this.remove(count - 1); //移除最后一项
219         }
220 
221         return null;
222     },
223 
224 
225     getItem: function (item) {
226 
227         var items = Meta.get(this, 'items');
228         var type = typeof item;
229 
230         if (type == 'number') {
231             return items[item];
232         }
233 
234         item =
235             type == 'string' ? ComponentManager.get(item) :
236             //type == 'number' ? items[item] :
237             ComponentHelper.isValidComponent(item) ? item : null;
238 
239         return item && $.Array.contains(items, item) ? item : null;
240     },
241 
242     getLastItem: function () {
243         var count = this.getCount();
244         return count > 0 ? this.getItem(count - 1) : null;
245     },
246 
247     /**
248     * 检测本容器实例是否包含指定的子组件。
249     * @param {Component|string|number} item 要进行检测的子组件。
250         可以是一个组件实例；
251         或一个字符串，表示组件的 id；
252         或一个数字，表示该组件在容器中的索引值。
253     * @return {boolean} 如果容器中包含该子组件则返回 true；否则返回 false。
254     */
255     contains: function (item) {
256         return !!this.getItem(item);
257     },
258 
259     /**
260     * 销毁本实例，同时执行一些清理操作。
261     * 该方法会释放所有关联的资源，包括元数据和事件，并且会从组件树中移除本组件。
262     */
263     dispose: function () {
264 
265         //销毁子组件。
266         var items = Meta.get(this, 'items');
267 
268         $.Array.each(items, function (item, index) {
269             item.dispose();
270         });
271 
272         //销毁自己
273         this.callSuper(arguments);
274 
275     }

279 });

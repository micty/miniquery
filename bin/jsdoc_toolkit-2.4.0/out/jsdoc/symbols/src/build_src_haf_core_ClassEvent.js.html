  3 /**
  4 * 针对创建类的事件通知
  5 * @namespace
  6 */
  7 var ClassEvent = (function ($, ClassManager) {
  8 
  9     var guid = 'ClassEvent.eventTarget.' + $.String.random();
 10 
 11     //为了提高事件绑定速度
 12     var eventTarget = {
 13         toString: function () {
 14             return guid;
 15         }
 16     };
 17 
 18 
 19     function ondefine(className, fn) {
 20 
 21         var eventName = 'defineClass:';
 22 
 23         var type = typeof className;
 24         if (type == 'function') { //此时为 ondefine(fn)
 25             fn = className;
 26             className = undefined;
 27         }
 28         else if (type == 'string') {
 29             eventName += className;
 30         }
 31         else {
 32             throw Haf.error('无法识别参数 className 的类型');
 33         }
 34 
 35         $.Event.bind(eventTarget, eventName, fn);
 36 
 37 
 38         if (className) { //此时为 ondefine(className, fn)
 39             var F = ClassManager.getClass(className, false); //这里不要自动去加载类
 40 
 41             if (F) { //已经定义了该类，立即触发
 42                 var data = ClassManager.getData(className);
 43                 fn(data, className); //这里不用 trigger(className)，因为这样会把别的 handler 也触发
 44 
 45                 trigger(); //触发总事件
 46             }
 47         }
 48 
 49     }
 50 
 51     //仅供 define 方法内部调用
 52     function trigger(className) {
 53 
 54         var args;
 55 
 56         if (className) {
 57 
 58             var F = ClassManager.getClass(className, false);
 59             if (!F) {
 60                 throw Haf.error('不存在 className 为 {0} 的类', className);
 61             }
 62 
 63             var data = ClassManager.getData(className);
 64             args = [data, className];
 65             $.Event.trigger(eventTarget, 'defineClass:' + className, args);
 66         }
 67 
 68         //最后触发总事件
 69         $.Event.trigger(eventTarget, 'defineClass:', args);
 70 
 71     }
 72 
 73 
 74 
 75     function ondefinextype(xtype, fn) {
 76 
 77         var eventName = 'defineXtype:';
 78 
 79         var type = typeof xtype;
 80         if (type == 'function') { //此时为 ondefinextype( fn )
 81             fn = xtype;
 82             xtype = undefined;
 83         }
 84         else if (type == 'string') {
 85             eventName += xtype;
 86         }
 87         else {
 88             throw Haf.error('无法识别参数 xtype 的类型');
 89         }
 90 
 91         $.Event.bind(eventTarget, eventName, fn);
 92 
 93         if (xtype) { //此时为 ondefinextype(xtype, fn)
 94 
 95             var className = ClassManager.getClassNameByXtype(xtype);
 96             if (className) { //已经定义了该 xtype，立即触发
 97                 var data = ClassManager.getData(className);
 98                 fn(data, className); //这里不用 triggerXtype(xtype)，因为这样会把别的 handler 也触发
 99                 triggerXtype(); //触发总事件
100             }
101         }
102     }
103 
104     //仅供 define 方法内部调用
105     function triggerXtype(xtype) {
106 
107         var args;
108 
109         if (xtype) {
110 
111             var className = ClassManager.getClassNameByXtype(xtype);
112             if (!className) {
113                 throw Haf.error('不在存在 xtype 为 {0} 的类', xtype);
114             }
115 
116             var data = ClassManager.getData(className);
117             args = [data, className];
118 
119             $.Event.trigger(eventTarget, 'defineXtype:' + xtype, args);
120         }
121 
122         //最后触发总事件
123         $.Event.trigger(eventTarget, 'defineXtype:', args);
124 
125     }
126 
127     return {
128         trigger: trigger,
129         ondefine: ondefine,
130         triggerXtype: triggerXtype,
131         ondefinextype: ondefinextype
132     };
133 })(Haf, ClassManager);
